61 // std::plus-family operates on operands with identical types - we need to
62 // support all the (vectype op float) combinations in vector maths.
63 // Providing T(float) would open the way to endless implicit conversions.
65 namespace Intern {
66 template <typename T0, typename T1, typename TRES = T0> struct plus {
67 TRES operator() (const T0& t0, const T1& t1) const {
68 return t0+t1;
69 }
70 };
71 template <typename T0, typename T1, typename TRES = T0> struct minus {
72 TRES operator() (const T0& t0, const T1& t1) const {
73 return t0-t1;
74 }
75 };
76 template <typename T0, typename T1, typename TRES = T0> struct multiplies {
77 TRES operator() (const T0& t0, const T1& t1) const {
78 return t0*t1;
79 }
80 };
81 template <typename T0, typename T1, typename TRES = T0> struct divides {
82 TRES operator() (const T0& t0, const T1& t1) const {
83 return t0/t1;
84 }
85 };
86 }

226 template <template <typename t> class op> static Vertex BinaryOp(const Vertex& v0, const Vertex& v1) {
227 // this is a heavy task for the compiler to optimize ... *pray*
228
229 Vertex res;
230 res.position = op<aiVector3D>()(v0.position,v1.position);
231 res.normal = op<aiVector3D>()(v0.normal,v1.normal);
232 res.tangent = op<aiVector3D>()(v0.tangent,v1.tangent);
233 res.bitangent = op<aiVector3D>()(v0.bitangent,v1.bitangent);
234
235 for (unsigned int i = 0; i < AI_MAX_NUMBER_OF_TEXTURECOORDS; ++i) {
236 res.texcoords[i] = op<aiVector3D>()(v0.texcoords[i],v1.texcoords[i]);
237 }
238 for (unsigned int i = 0; i < AI_MAX_NUMBER_OF_COLOR_SETS; ++i) {
239 res.colors[i] = op<aiColor4D>()(v0.colors[i],v1.colors[i]);
240 }
241 return res;
