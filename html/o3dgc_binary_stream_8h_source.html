291 void WriteUIntASCII(unsigned long value)
292 {
293 if (value >= O3DGC_BINARY_STREAM_MAX_SYMBOL0)
294 {
295 m_stream.PushBack(O3DGC_BINARY_STREAM_MAX_SYMBOL0);
296 value -= O3DGC_BINARY_STREAM_MAX_SYMBOL0;
297 unsigned char a, b;
298 do
299 {
300 a = ((value & O3DGC_BINARY_STREAM_MAX_SYMBOL1) << 1);
301 b = ( (value >>= O3DGC_BINARY_STREAM_BITS_PER_SYMBOL1) > 0);
302 a += b;
303 m_stream.PushBack(a);
304 } while (b);
305 }
306 else
307 {
308 m_stream.PushBack((unsigned char) value);
309 }
