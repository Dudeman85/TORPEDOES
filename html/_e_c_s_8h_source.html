124 //Removes a component from an entity
125 void RemoveComponent(Entity entity) override
126 {
127 //Keep track of the deleted component's index, and the entity of the last component in the array
128 uint32_t deletedIndex = entityToIndex[entity];
129 Entity lastEntity = indexToEntity[components.size() - 1];
130
131 //Move the last element to the deleted index
132 components[entityToIndex[entity]] = components.back();
133
134 //Update the maps for the moved component
135 entityToIndex[lastEntity] = deletedIndex;
136 indexToEntity[deletedIndex] = lastEntity;
137
138 //Remove the deleted component from the maps
139 entityToIndex.erase(entity);
140 indexToEntity.erase(components.size() - 1);
141 components.pop_back();
142 }
143 };

145 //Implementation internal function. Called whenever an entity's signature changes
146 void _OnEntitySignatureChanged(Entity entity)
147 {
148 const Signature& signature = entitySignatures[entity];
149
150 //Loop through every system
151 for (auto const& system : systems)
152 {
153 //If the entity's signature matches the system's signature
154 if ((signature & systemSignatures[system.first]) == systemSignatures[system.first])
155 {
156 //Add the entity to the system's set
157 system.second->entities.insert(entity);
158 }
159 else
160 {
161 //Remove the entity from the system's set
162 system.second->entities.erase(entity);
163 }
164 }
165 }

436 {
