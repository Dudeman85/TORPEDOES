59{
60 return 0.5 * (v1.x * ((double)v3.y - v2.y) + v2.x * ((double)v1.y - v3.y) + v3.x * ((double)v2.y - v1.y));

77template <typename T>
78inline bool PointInTriangle2D(const T& p0, const T& p1,const T& p2, const T& pp)
79{
80 // Point in triangle test using baryzentric coordinates
81 const aiVector2D v0 = p1 - p0;
82 const aiVector2D v1 = p2 - p0;
83 const aiVector2D v2 = pp - p0;
84
85 double dot00 = v0 * v0;
86 double dot01 = v0 * v1;
87 double dot02 = v0 * v2;
88 double dot11 = v1 * v1;
89 double dot12 = v1 * v2;
90
91 const double invDenom = 1 / (dot00 * dot11 - dot01 * dot01);
92 dot11 = (dot11 * dot02 - dot01 * dot12) * invDenom;
93 dot00 = (dot00 * dot12 - dot01 * dot02) * invDenom;
94
95 return (dot11 > 0) && (dot00 > 0) && (dot11 + dot00 < 1);
96}

106template <typename T>
107inline bool IsCCW(T* in, size_t npoints) {
108 double aa, bb, cc, b, c, theta;
109 double convex_turn;
110 double convex_sum = 0;
111
112 ai_assert(npoints >= 3);
113
114 for (size_t i = 0; i < npoints - 2; i++) {
115 aa = ((in[i+2].x - in[i].x) * (in[i+2].x - in[i].x)) +
116 ((-in[i+2].y + in[i].y) * (-in[i+2].y + in[i].y));
117
118 bb = ((in[i+1].x - in[i].x) * (in[i+1].x - in[i].x)) +
119 ((-in[i+1].y + in[i].y) * (-in[i+1].y + in[i].y));
120
121 cc = ((in[i+2].x - in[i+1].x) *
122 (in[i+2].x - in[i+1].x)) +
123 ((-in[i+2].y + in[i+1].y) *
124 (-in[i+2].y + in[i+1].y));
125
126 b = std::sqrt(bb);
127 c = std::sqrt(cc);
128 theta = std::acos((bb + cc - aa) / (2 * b * c));
129
130 if (OnLeftSideOfLine2D(in[i],in[i+2],in[i+1])) {
131 // if (convex(in[i].x, in[i].y,
132 // in[i+1].x, in[i+1].y,
133 // in[i+2].x, in[i+2].y)) {
134 convex_turn = AI_MATH_PI_F - theta;
135 convex_sum += convex_turn;
136 }
137 else {
138 convex_sum -= AI_MATH_PI_F - theta;
139 }
140 }
141 aa = ((in[1].x - in[npoints-2].x) *
142 (in[1].x - in[npoints-2].x)) +
143 ((-in[1].y + in[npoints-2].y) *

152 ((-in[1].y + in[0].y) * (-in[1].y + in[0].y));
153
154 b = std::sqrt(bb);

158 //if (convex(in[npoints-2].x, in[npoints-2].y,

162 convex_turn = AI_MATH_PI_F - theta;

188template <int ofs_x, int ofs_y, int ofs_z, typename TReal>
189inline void NewellNormal (aiVector3t<TReal>& out, int num, TReal* x, TReal* y, TReal* z)
190{
191 // Duplicate the first two vertices at the end
192 x[(num+0)*ofs_x] = x[0];
193 x[(num+1)*ofs_x] = x[ofs_x];
194
195 y[(num+0)*ofs_y] = y[0];
196 y[(num+1)*ofs_y] = y[ofs_y];
197
198 z[(num+0)*ofs_z] = z[0];
199 z[(num+1)*ofs_z] = z[ofs_z];
200
201 TReal sum_xy = 0.0, sum_yz = 0.0, sum_zx = 0.0;
202
203 TReal *xptr = x +ofs_x, *xlow = x, *xhigh = x + ofs_x*2;
204 TReal *yptr = y +ofs_y, *ylow = y, *yhigh = y + ofs_y*2;
205 TReal *zptr = z +ofs_z, *zlow = z, *zhigh = z + ofs_z*2;
206
207 for (int tmp=0; tmp < num; tmp++) {
208 sum_xy += (*xptr) * ( (*yhigh) - (*ylow) );
209 sum_yz += (*yptr) * ( (*zhigh) - (*zlow) );
210 sum_zx += (*zptr) * ( (*xhigh) - (*xlow) );
211
212 xptr += ofs_x;
213 xlow += ofs_x;
214 xhigh += ofs_x;
215
216 yptr += ofs_y;
217 ylow += ofs_y;
218 yhigh += ofs_y;
219
220 zptr += ofs_z;
221 zlow += ofs_z;
222 zhigh += ofs_z;
223 }
224 out = aiVector3t<TReal>(sum_yz,sum_zx,sum_xy);
225}
