61class MemoryIOStream : public IOStream {
62public:
63 MemoryIOStream (const uint8_t* buff, size_t len, bool own = false)
64 : buffer (buff)
65 , length(len)
66 , pos((size_t)0)
67 , own(own) {
68 // empty
69 }
70
71 ~MemoryIOStream () {
72 if(own) {
73 delete[] buffer;
74 }
75 }
76
77 // -------------------------------------------------------------------
78 // Read from stream
79 size_t Read(void* pvBuffer, size_t pSize, size_t pCount) {
80 ai_assert(nullptr != pvBuffer);
81 ai_assert(0 != pSize);
82
83 const size_t cnt = std::min( pCount, (length-pos) / pSize);
84 const size_t ofs = pSize * cnt;
85
86 ::memcpy(pvBuffer,buffer+pos,ofs);
87 pos += ofs;
88
89 return cnt;
90 }
91
92 // -------------------------------------------------------------------
93 // Write to stream
94 size_t Write(const void* /*pvBuffer*/, size_t /*pSize*/,size_t /*pCount*/) {
95 ai_assert(false); // won't be needed
96 return 0;
97 }
98
99 // -------------------------------------------------------------------
100 // Seek specific position
101 aiReturn Seek(size_t pOffset, aiOrigin pOrigin) {
102 if (aiOrigin_SET == pOrigin) {
103 if (pOffset > length) {
104 return AI_FAILURE;
105 }
106 pos = pOffset;
107 } else if (aiOrigin_END == pOrigin) {
108 if (pOffset > length) {
109 return AI_FAILURE;
110 }
111 pos = length-pOffset;
112 } else {
113 if (pOffset+pos > length) {
114 return AI_FAILURE;
115 }
116 pos += pOffset;
117 }
118 return AI_SUCCESS;
119 }
120
121 // -------------------------------------------------------------------
122 // Get current seek position
123 size_t Tell() const {
124 return pos;
125 }
126
127 // -------------------------------------------------------------------
128 // Get size of file
129 size_t FileSize() const {
130 return length;
131 }
132
133 // -------------------------------------------------------------------
134 // Flush file contents
135 void Flush() {
136 ai_assert(false); // won't be needed
137 }
138
139private:
140 const uint8_t* buffer;
141 size_t length,pos;
142 bool own;
143};

147class MemoryIOSystem : public IOSystem {
148public:
150 MemoryIOSystem(const uint8_t* buff, size_t len, IOSystem* io)
151 : buffer(buff)
152 , length(len)
153 , existing_io(io)
154 , created_streams() {
155 // empty
156 }
157
159 ~MemoryIOSystem() {
160 }
161
162 // -------------------------------------------------------------------
164 bool Exists(const char* pFile) const override {
165 if (0 == strncmp( pFile, AI_MEMORYIO_MAGIC_FILENAME, AI_MEMORYIO_MAGIC_FILENAME_LENGTH ) ) {
166 return true;
167 }
168 return existing_io ? existing_io->Exists(pFile) : false;
169 }
170
171 // -------------------------------------------------------------------
173 char getOsSeparator() const override {
174 return existing_io ? existing_io->getOsSeparator()
175 : '/'; // why not? it doesn't care
176 }
177
178 // -------------------------------------------------------------------
180 IOStream* Open(const char* pFile, const char* pMode = "rb") override {
181 if ( 0 == strncmp( pFile, AI_MEMORYIO_MAGIC_FILENAME, AI_MEMORYIO_MAGIC_FILENAME_LENGTH ) ) {
182 created_streams.emplace_back(new MemoryIOStream(buffer, length));
183 return created_streams.back();
184 }
185 return existing_io ? existing_io->Open(pFile, pMode) : NULL;
186 }
187
188 // -------------------------------------------------------------------
190 void Close( IOStream* pFile) override {
191 auto it = std::find(created_streams.begin(), created_streams.end(), pFile);
192 if (it != created_streams.end()) {
193 delete pFile;
194 created_streams.erase(it);
195 } else if (existing_io) {
196 existing_io->Close(pFile);
197 }
198 }
199
200 // -------------------------------------------------------------------
202 bool ComparePaths(const char* one, const char* second) const override {
203 return existing_io ? existing_io->ComparePaths(one, second) : false;
204 }
205
206 bool PushDirectory( const std::string &path ) override {
207 return existing_io ? existing_io->PushDirectory(path) : false;
208 }
209
210 const std::string &CurrentDirectory() const override {
211 static std::string empty;
212 return existing_io ? existing_io->CurrentDirectory() : empty;
213 }
214
215 size_t StackSize() const override {
216 return existing_io ? existing_io->StackSize() : 0;
217 }
218
219 bool PopDirectory() override {
220 return existing_io ? existing_io->PopDirectory() : false;
221 }
222
223 bool CreateDirectory( const std::string &path ) override {
224 return existing_io ? existing_io->CreateDirectory(path) : false;
225 }
226
227 bool ChangeDirectory( const std::string &path ) override {
228 return existing_io ? existing_io->ChangeDirectory(path) : false;
229 }
230
231 bool DeleteFile( const std::string &file ) override {
232 return existing_io ? existing_io->DeleteFile(file) : false;
233 }
234
235private:
236 const uint8_t* buffer;
237 size_t length;
238 IOSystem* existing_io;
239 std::vector<IOStream*> created_streams;
240};
