94 /* read an integer */
95 static FT_Long
96 cff_parse_integer( FT_Byte* start,
97 FT_Byte* limit )
98 {
99 FT_Byte* p = start;
100 FT_Int v = *p++;
101 FT_Long val = 0;
102
103
104 if ( v == 28 )
105 {
106 if ( p + 2 > limit && limit >= p )
107 goto Bad;
108
109 val = (FT_Short)( ( (FT_UShort)p[0] << 8 ) | p[1] );
110 }
111 else if ( v == 29 )
112 {
113 if ( p + 4 > limit && limit >= p )
114 goto Bad;
115
116 val = (FT_Long)( ( (FT_ULong)p[0] << 24 ) |
117 ( (FT_ULong)p[1] << 16 ) |
118 ( (FT_ULong)p[2] << 8 ) |
119 (FT_ULong)p[3] );
120 }
121 else if ( v < 247 )
122 {
123 val = v - 139;
124 }
125 else if ( v < 251 )
126 {
127 if ( p + 1 > limit && limit >= p )
128 goto Bad;
129
130 val = ( v - 247 ) * 256 + p[0] + 108;
131 }
132 else
133 {
134 if ( p + 1 > limit && limit >= p )
135 goto Bad;
136
137 val = -( v - 251 ) * 256 - p[0] - 108;
138 }
139
140 Exit:
141 return val;
142
143 Bad:
144 val = 0;
145 FT_TRACE4(( "!!!END OF DATA:!!!" ));
146 goto Exit;
147 }

181 /* read a real */
182 static FT_Fixed
183 cff_parse_real( FT_Byte* start,
184 FT_Byte* limit,
185 FT_Long power_ten,
186 FT_Long* scaling )
187 {
188 FT_Byte* p = start;
189 FT_Int nib;
190 FT_UInt phase;
191
192 FT_Long result, number, exponent;
193 FT_Int sign = 0, exponent_sign = 0, have_overflow = 0;
194 FT_Long exponent_add, integer_length, fraction_length;
195
196
197 if ( scaling )
198 *scaling = 0;
199
200 result = 0;
201
202 number = 0;
203 exponent = 0;
204
205 exponent_add = 0;
206 integer_length = 0;
207 fraction_length = 0;
208
209 /* First of all, read the integer part. */
210 phase = 4;
211
212 for (;;)
213 {
214 /* If we entered this iteration with phase == 4, we need to */
215 /* read a new byte. This also skips past the initial 0x1E. */
216 if ( phase )
217 {
218 p++;
219
220 /* Make sure we don't read past the end. */
221 if ( p + 1 > limit && limit >= p )
222 goto Bad;
223 }
224
225 /* Get the nibble. */
226 nib = (FT_Int)( p[0] >> phase ) & 0xF;
227 phase = 4 - phase;
228
229 if ( nib == 0xE )
230 sign = 1;
231 else if ( nib > 9 )
232 break;
233 else
234 {
235 /* Increase exponent if we can't add the digit. */
236 if ( number >= 0xCCCCCCCL )
237 exponent_add++;
238 /* Skip leading zeros. */
239 else if ( nib || number )
240 {
241 integer_length++;
242 number = number * 10 + nib;
243 }
244 }
245 }
246
247 /* Read fraction part, if any. */
248 if ( nib == 0xA )
249 for (;;)
250 {
251 /* If we entered this iteration with phase == 4, we need */
252 /* to read a new byte. */
253 if ( phase )
254 {
255 p++;
256
257 /* Make sure we don't read past the end. */
258 if ( p + 1 > limit && limit >= p )
259 goto Bad;
260 }
261
262 /* Get the nibble. */
263 nib = ( p[0] >> phase ) & 0xF;
264 phase = 4 - phase;
265 if ( nib >= 10 )
266 break;
267
268 /* Skip leading zeros if possible. */
269 if ( !nib && !number )
270 exponent_add--;
271 /* Only add digit if we don't overflow. */
272 else if ( number < 0xCCCCCCCL && fraction_length < 9 )
273 {
274 fraction_length++;
275 number = number * 10 + nib;
276 }
277 }
278
279 /* Read exponent, if any. */
280 if ( nib == 12 )
281 {
282 exponent_sign = 1;
283 nib = 11;
284 }
285
286 if ( nib == 11 )
287 {
288 for (;;)
289 {
290 /* If we entered this iteration with phase == 4, */
291 /* we need to read a new byte. */
292 if ( phase )
293 {
294 p++;
295
296 /* Make sure we don't read past the end. */
297 if ( p + 1 > limit && limit >= p )
298 goto Bad;
299 }
300
301 /* Get the nibble. */
302 nib = ( p[0] >> phase ) & 0xF;
303 phase = 4 - phase;
304 if ( nib >= 10 )
305 break;
306
307 /* Arbitrarily limit exponent. */
308 if ( exponent > 1000 )
309 have_overflow = 1;
310 else
311 exponent = exponent * 10 + nib;
312 }
313
314 if ( exponent_sign )
315 exponent = -exponent;
316 }
317
318 if ( !number )
319 goto Exit;
320
321 if ( have_overflow )
322 {
323 if ( exponent_sign )
324 goto Underflow;
325 else
326 goto Overflow;
327 }
328
329 /* We don't check `power_ten' and `exponent_add'. */
330 exponent += power_ten + exponent_add;
331
332 if ( scaling )
333 {
334 /* Only use `fraction_length'. */
335 fraction_length += integer_length;
336 exponent += integer_length;
337
338 if ( fraction_length <= 5 )
339 {
340 if ( number > 0x7FFFL )
341 {
342 result = FT_DivFix( number, 10 );
343 *scaling = exponent - fraction_length + 1;
344 }
345 else
346 {
347 if ( exponent > 0 )
348 {
349 FT_Long new_fraction_length, shift;
350
351
352 /* Make `scaling' as small as possible. */
353 new_fraction_length = FT_MIN( exponent, 5 );
354 shift = new_fraction_length - fraction_length;
355
356 if ( shift > 0 )
357 {
358 exponent -= new_fraction_length;
359 number *= power_tens[shift];
360 if ( number > 0x7FFFL )
361 {
362 number /= 10;
363 exponent += 1;
364 }
365 }
366 else
367 exponent -= fraction_length;
368 }
369 else
370 exponent -= fraction_length;
371
372 result = (FT_Long)( (FT_ULong)number << 16 );
373 *scaling = exponent;
374 }
375 }
376 else
377 {
378 if ( ( number / power_tens[fraction_length - 5] ) > 0x7FFFL )
379 {
380 result = FT_DivFix( number, power_tens[fraction_length - 4] );
381 *scaling = exponent - 4;
382 }
383 else
384 {
385 result = FT_DivFix( number, power_tens[fraction_length - 5] );
386 *scaling = exponent - 5;
387 }
388 }
389 }
390 else
391 {
392 integer_length += exponent;
393 fraction_length -= exponent;
394
395 if ( integer_length > 5 )
396 goto Overflow;
397 if ( integer_length < -5 )
398 goto Underflow;
399
400 /* Remove non-significant digits. */
401 if ( integer_length < 0 )
402 {
403 number /= power_tens[-integer_length];
404 fraction_length += integer_length;
405 }
406
407 /* this can only happen if exponent was non-zero */
408 if ( fraction_length == 10 )
409 {
410 number /= 10;
411 fraction_length -= 1;
412 }
413
414 /* Convert into 16.16 format. */
415 if ( fraction_length > 0 )
416 {
417 if ( ( number / power_tens[fraction_length] ) > 0x7FFFL )
418 goto Exit;
419
420 result = FT_DivFix( number, power_tens[fraction_length] );
421 }
422 else
423 {
424 number *= power_tens[-fraction_length];
425
426 if ( number > 0x7FFFL )
427 goto Overflow;
428
429 result = (FT_Long)( (FT_ULong)number << 16 );
430 }
431 }
432
433 Exit:
434 if ( sign )
435 result = -result;
436
437 return result;
438
439 Overflow:
440 result = 0x7FFFFFFFL;
441 FT_TRACE4(( "!!!OVERFLOW:!!!" ));
442 goto Exit;
443
444 Underflow:
445 result = 0;
446 FT_TRACE4(( "!!!UNDERFLOW:!!!" ));
447 goto Exit;
448
449 Bad:
450 result = 0;
451 FT_TRACE4(( "!!!END OF DATA:!!!" ));
452 goto Exit;
453 }

456 /* read a number, either integer or real */
457 FT_LOCAL_DEF( FT_Long )
458 cff_parse_num( CFF_Parser parser,
459 FT_Byte** d )
460 {
461 if ( **d == 30 )
462 {
463 /* binary-coded decimal is truncated to integer */
464 return cff_parse_real( *d, parser->limit, 0, NULL ) >> 16;
465 }
466
467 else if ( **d == 255 )
468 {
469 /* 16.16 fixed-point is used internally for CFF2 blend results. */
470 /* Since these are trusted values, a limit check is not needed. */
471
472 /* After the 255, 4 bytes give the number. */
473 /* The blend value is converted to integer, with rounding; */
474 /* due to the right-shift we don't need the lowest byte. */
475#if 0
476 return (FT_Short)(
477 ( ( ( (FT_UInt32)*( d[0] + 1 ) << 24 ) |
478 ( (FT_UInt32)*( d[0] + 2 ) << 16 ) |
479 ( (FT_UInt32)*( d[0] + 3 ) << 8 ) |
480 (FT_UInt32)*( d[0] + 4 ) ) + 0x8000U ) >> 16 );
481#else
482 return (FT_Short)(
483 ( ( ( (FT_UInt32)*( d[0] + 1 ) << 16 ) |
484 ( (FT_UInt32)*( d[0] + 2 ) << 8 ) |
485 (FT_UInt32)*( d[0] + 3 ) ) + 0x80U ) >> 8 );
486#endif
487 }
488
489 else
490 return cff_parse_integer( *d, parser->limit );
491 }

494 /* read a floating point number, either integer or real */
495 static FT_Fixed
496 do_fixed( CFF_Parser parser,
497 FT_Byte** d,
498 FT_Long scaling )
499 {
500 if ( **d == 30 )
501 return cff_parse_real( *d, parser->limit, scaling, NULL );
502 else
503 {
504 FT_Long val = cff_parse_integer( *d, parser->limit );
505
506
507 if ( scaling )
508 {
509 if ( FT_ABS( val ) > power_ten_limits[scaling] )
510 {
511 val = val > 0 ? 0x7FFFFFFFL : -0x7FFFFFFFL;
512 goto Overflow;
513 }
514
515 val *= power_tens[scaling];
516 }
517
518 if ( val > 0x7FFF )
519 {
520 val = 0x7FFFFFFFL;
521 goto Overflow;
522 }
523 else if ( val < -0x7FFF )
524 {
525 val = -0x7FFFFFFFL;
526 goto Overflow;
527 }
528
529 return (FT_Long)( (FT_ULong)val << 16 );
530
531 Overflow:
532 FT_TRACE4(( "!!!OVERFLOW:!!!" ));
533 return val;
534 }
535 }

558 /* read a floating point number, either integer or real, */
559 /* and return it as precise as possible -- `scaling' returns */
560 /* the scaling factor (as a power of 10) */
561 static FT_Fixed
562 cff_parse_fixed_dynamic( CFF_Parser parser,
563 FT_Byte** d,
564 FT_Long* scaling )
565 {
566 FT_ASSERT( scaling );
567
568 if ( **d == 30 )
569 return cff_parse_real( *d, parser->limit, 0, scaling );
570 else
571 {
572 FT_Long number;
573 FT_Int integer_length;
574
575
576 number = cff_parse_integer( *d, parser->limit );
577
578 if ( number > 0x7FFFL )
579 {
580 for ( integer_length = 5; integer_length < 10; integer_length++ )
581 if ( number < power_tens[integer_length] )
582 break;
583
584 if ( ( number / power_tens[integer_length - 5] ) > 0x7FFFL )
585 {
586 *scaling = integer_length - 4;
587 return FT_DivFix( number, power_tens[integer_length - 4] );
588 }
589 else
590 {
591 *scaling = integer_length - 5;
592 return FT_DivFix( number, power_tens[integer_length - 5] );
593 }
594 }
595 else
596 {
597 *scaling = 0;
598 return (FT_Long)( (FT_ULong)number << 16 );
599 }
600 }

604 static FT_Error
605 cff_parse_font_matrix( CFF_Parser parser )
606 {
607 CFF_FontRecDict dict = (CFF_FontRecDict)parser->object;
608 FT_Matrix* matrix = &dict->font_matrix;
609 FT_Vector* offset = &dict->font_offset;
610 FT_ULong* upm = &dict->units_per_em;
611 FT_Byte** data = parser->stack;
612
613
614 if ( parser->top >= parser->stack + 6 )
615 {
616 FT_Fixed values[6];
617 FT_Long scalings[6];
618
619 FT_Long min_scaling, max_scaling;
620 int i;
621
622
623 dict->has_font_matrix = TRUE;
624
625 /* We expect a well-formed font matrix, that is, the matrix elements */
626 /* `xx' and `yy' are of approximately the same magnitude. To avoid */
627 /* loss of precision, we use the magnitude of the largest matrix */
628 /* element to scale all other elements. The scaling factor is then */
629 /* contained in the `units_per_em' value. */
630
631 max_scaling = FT_LONG_MIN;
632 min_scaling = FT_LONG_MAX;
633
634 for ( i = 0; i < 6; i++ )
635 {
636 values[i] = cff_parse_fixed_dynamic( parser, data++, &scalings[i] );
637 if ( values[i] )
638 {
639 if ( scalings[i] > max_scaling )
640 max_scaling = scalings[i];
641 if ( scalings[i] < min_scaling )
642 min_scaling = scalings[i];
643 }
644 }
645
646 if ( max_scaling < -9 ||
647 max_scaling > 0 ||
648 ( max_scaling - min_scaling ) < 0 ||
649 ( max_scaling - min_scaling ) > 9 )
650 {
651 FT_TRACE1(( "cff_parse_font_matrix:"
652 " strange scaling values (minimum %ld, maximum %ld),\n",
653 min_scaling, max_scaling ));
654 FT_TRACE1(( " "
655 " using default matrix\n" ));
656 goto Unlikely;
657 }
658
659 for ( i = 0; i < 6; i++ )
660 {
661 FT_Fixed value = values[i];
662 FT_Long divisor, half_divisor;
663
664
665 if ( !value )
666 continue;
667
668 divisor = power_tens[max_scaling - scalings[i]];
669 half_divisor = divisor >> 1;
670
671 if ( value < 0 )
672 {
673 if ( FT_LONG_MIN + half_divisor < value )
674 values[i] = ( value - half_divisor ) / divisor;
675 else
676 values[i] = FT_LONG_MIN / divisor;
677 }
678 else
679 {
680 if ( FT_LONG_MAX - half_divisor > value )
681 values[i] = ( value + half_divisor ) / divisor;
682 else
683 values[i] = FT_LONG_MAX / divisor;
684 }
685 }
686
687 matrix->xx = values[0];
688 matrix->yx = values[1];
689 matrix->xy = values[2];
690 matrix->yy = values[3];
691 offset->x = values[4];
692 offset->y = values[5];
693
694 *upm = (FT_ULong)power_tens[-max_scaling];
695
696 FT_TRACE4(( " [%f %f %f %f %f %f]\n",
697 (double)matrix->xx / (double)*upm / 65536,
698 (double)matrix->xy / (double)*upm / 65536,
699 (double)matrix->yx / (double)*upm / 65536,
700 (double)matrix->yy / (double)*upm / 65536,
701 (double)offset->x / (double)*upm / 65536,
702 (double)offset->y / (double)*upm / 65536 ));
703
704 if ( !FT_Matrix_Check( matrix ) )
705 {
706 FT_TRACE1(( "cff_parse_font_matrix:"
707 " degenerate values, using default matrix\n" ));
708 goto Unlikely;
709 }
710
711 return FT_Err_Ok;
712 }
713 else
714 return FT_THROW( Stack_Underflow );
715
716 Unlikely:
717 /* Return default matrix in case of unlikely values. */
718
719 matrix->xx = 0x10000L;
720 matrix->yx = 0;
721 matrix->xy = 0;
722 matrix->yy = 0x10000L;
723 offset->x = 0;
724 offset->y = 0;
725 *upm = 1;
726
727 return FT_Err_Ok;
728 }

1130 FT_LOCAL_DEF( FT_Error )
1131 cff_parser_run( CFF_Parser parser,
1132 FT_Byte* start,
1133 FT_Byte* limit )
1134 {
1135 FT_Byte* p = start;
1136 FT_Error error = FT_Err_Ok;
1137
1138#ifdef CFF_CONFIG_OPTION_OLD_ENGINE
1139 PSAux_Service psaux;
1140
1141 FT_Library library = parser->library;
1142 FT_Memory memory = library->memory;
1143#endif
1144
1145 parser->top = parser->stack;
1146 parser->start = start;
1147 parser->limit = limit;
1148 parser->cursor = start;
1149
1150 while ( p < limit )
1151 {
1152 FT_UInt v = *p;
1153
1154
1155 /* Opcode 31 is legacy MM T2 operator, not a number. */
1156 /* Opcode 255 is reserved and should not appear in fonts; */
1157 /* it is used internally for CFF2 blends. */
1158 if ( v >= 27 && v != 31 && v != 255 )
1159 {
1160 /* it's a number; we will push its position on the stack */
1161 if ( (FT_UInt)( parser->top - parser->stack ) >= parser->stackSize )
1162 goto Stack_Overflow;
1163
1164 *parser->top++ = p;
1165
1166 /* now, skip it */
1167 if ( v == 30 )
1168 {
1169 /* skip real number */
1170 p++;
1171 for (;;)
1172 {
1173 /* An unterminated floating point number at the */
1174 /* end of a dictionary is invalid but harmless. */
1175 if ( p >= limit )
1176 goto Exit;
1177 v = p[0] >> 4;
1178 if ( v == 15 )
1179 break;
1180 v = p[0] & 0xF;
1181 if ( v == 15 )
1182 break;
1183 p++;
1184 }
1185 }
1186 else if ( v == 28 )
1187 p += 2;
1188 else if ( v == 29 )
1189 p += 4;
1190 else if ( v > 246 )
1191 p += 1;
1192 }
1193#ifdef CFF_CONFIG_OPTION_OLD_ENGINE
1194 else if ( v == 31 )
1195 {
1196 /* a Type 2 charstring */
1197
1198 CFF_Decoder decoder;
1199 CFF_FontRec cff_rec;
1200 FT_Byte* charstring_base;
1201 FT_ULong charstring_len;
1202
1203 FT_Fixed* stack;
1204 FT_Byte* q = NULL;
1205
1206
1207 charstring_base = ++p;
1208
1209 /* search `endchar' operator */
1210 for (;;)
1211 {
1212 if ( p >= limit )
1213 goto Exit;
1214 if ( *p == 14 )
1215 break;
1216 p++;
1217 }
1218
1219 charstring_len = (FT_ULong)( p - charstring_base ) + 1;
1220
1221 /* construct CFF_Decoder object */
1222 FT_ZERO( &decoder );
1223 FT_ZERO( &cff_rec );
1224
1225 cff_rec.top_font.font_dict.num_designs = parser->num_designs;
1226 cff_rec.top_font.font_dict.num_axes = parser->num_axes;
1227 decoder.cff = &cff_rec;
1228
1229 psaux = (PSAux_Service)FT_Get_Module_Interface( library, "psaux" );
1230 if ( !psaux )
1231 {
1232 FT_ERROR(( "cff_parser_run: cannot access `psaux' module\n" ));
1233 error = FT_THROW( Missing_Module );
1234 goto Exit;
1235 }
1236
1237 error = psaux->cff_decoder_funcs->parse_charstrings_old(
1238 &decoder, charstring_base, charstring_len, 1 );
1239 if ( error )
1240 goto Exit;
1241
1242 /* Now copy the stack data in the temporary decoder object, */
1243 /* converting it back to charstring number representations */
1244 /* (this is ugly, I know). */
1245 /* The maximum required size is 5 bytes per stack element. */
1246 if ( FT_QALLOC( q, (FT_Long)( 2 * sizeof ( FT_ListNode ) ) +
1247 5 * ( decoder.top - decoder.stack ) ) )
1248 goto Exit;
1249
1250 FT_List_Add( &parser->t2_strings, (FT_ListNode)q );
1251
1252 q += 2 * sizeof ( FT_ListNode );
1253
1254 for ( stack = decoder.stack; stack < decoder.top; stack++ )
1255 {
1256 FT_Long num = *stack;
1257
1258
1259 if ( (FT_UInt)( parser->top - parser->stack ) >= parser->stackSize )
1260 goto Stack_Overflow;
1261
1262 *parser->top++ = q;
1263
1264 if ( num & 0xFFFFU )
1265 {
1266 *q++ = 255;
1267 *q++ = (FT_Byte)( ( num >> 24 ) & 0xFF );
1268 *q++ = (FT_Byte)( ( num >> 16 ) & 0xFF );
1269 *q++ = (FT_Byte)( ( num >> 8 ) & 0xFF );
1270 *q++ = (FT_Byte)( ( num ) & 0xFF );
1271 }
1272 else
1273 {
1274 num >>= 16;
1275
1276 if ( -107 <= num && num <= 107 )
1277 *q++ = (FT_Byte)( num + 139 );
1278 else if ( 108 <= num && num <= 1131 )
1279 {
1280 *q++ = (FT_Byte)( ( ( num - 108 ) >> 8 ) + 247 );
1281 *q++ = (FT_Byte)( ( num - 108 ) & 0xFF );
1282 }
1283 else if ( -1131 <= num && num <= -108 )
1284 {
1285 *q++ = (FT_Byte)( ( ( -num - 108 ) >> 8 ) + 251 );
1286 *q++ = (FT_Byte)( ( -num - 108) & 0xFF );
1287 }
1288 else
1289 {
1290 *q++ = 28;
1291 *q++ = (FT_Byte)( num >> 8 );
1292 *q++ = (FT_Byte)( num & 0xFF );
1293 }
1294 }
1295 }
1296 }
1297#endif /* CFF_CONFIG_OPTION_OLD_ENGINE */
1298 else
1299 {
1300 /* This is not a number, hence it's an operator. Compute its code */
1301 /* and look for it in our current list. */
1302
1303 FT_UInt code;
1304 FT_UInt num_args;
1305 const CFF_Field_Handler* field;
1306
1307
1308 if ( (FT_UInt)( parser->top - parser->stack ) >= parser->stackSize )
1309 goto Stack_Overflow;
1310
1311 num_args = (FT_UInt)( parser->top - parser->stack );
1312 *parser->top = p;
1313 code = v;
1314
1315 if ( v == 12 )
1316 {
1317 /* two byte operator */
1318 p++;
1319 if ( p >= limit )
1320 goto Syntax_Error;
1321
1322 code = 0x100 | p[0];
1323 }
1324 code = code | parser->object_code;
1325
1326 for ( field = cff_field_handlers; field->kind; field++ )
1327 {
1328 if ( field->code == (FT_Int)code )
1329 {
1330 /* we found our field's handler; read it */
1331 FT_Long val;
1332 FT_Byte* q = (FT_Byte*)parser->object + field->offset;
1333
1334
1335#ifdef FT_DEBUG_LEVEL_TRACE
1336 FT_TRACE4(( " %s", field->id ));
1337#endif
1338
1339 /* check that we have enough arguments -- except for */
1340 /* delta encoded arrays, which can be empty */
1341 if ( field->kind != cff_kind_delta && num_args < 1 )
1342 goto Stack_Underflow;
1343
1344 switch ( field->kind )
1345 {
1346 case cff_kind_bool:
1347 case cff_kind_string:
1348 case cff_kind_num:
1349 val = cff_parse_num( parser, parser->stack );
1350 goto Store_Number;
1351
1352 case cff_kind_fixed:
1353 val = cff_parse_fixed( parser, parser->stack );
1354 goto Store_Number;
1355
1356 case cff_kind_fixed_thousand:
1357 val = cff_parse_fixed_scaled( parser, parser->stack, 3 );
1358
1359 Store_Number:
1360 switch ( field->size )
1361 {
1362 case (8 / FT_CHAR_BIT):
1363 *(FT_Byte*)q = (FT_Byte)val;
1364 break;
1365
1366 case (16 / FT_CHAR_BIT):
1367 *(FT_Short*)q = (FT_Short)val;
1368 break;
1369
1370 case (32 / FT_CHAR_BIT):
1371 *(FT_Int32*)q = (FT_Int)val;
1372 break;
1373
1374 default: /* for 64-bit systems */
1375 *(FT_Long*)q = val;
1376 }
1377
1378#ifdef FT_DEBUG_LEVEL_TRACE
1379 switch ( field->kind )
1380 {
1381 case cff_kind_bool:
1382 FT_TRACE4(( " %s\n", val ? "true" : "false" ));
1383 break;
1384
1385 case cff_kind_string:
1386 FT_TRACE4(( " %ld (SID)\n", val ));
1387 break;
1388
1389 case cff_kind_num:
1390 FT_TRACE4(( " %ld\n", val ));
1391 break;
1392
1393 case cff_kind_fixed:
1394 FT_TRACE4(( " %f\n", (double)val / 65536 ));
1395 break;
1396
1397 case cff_kind_fixed_thousand:
1398 FT_TRACE4(( " %f\n", (double)val / 65536 / 1000 ));
1399 break;
1400
1401 default:
1402 ; /* never reached */
1403 }
1404#endif
1405
1406 break;
1407
1408 case cff_kind_delta:
1409 {
1410 FT_Byte* qcount = (FT_Byte*)parser->object +
1411 field->count_offset;
1412
1413 FT_Byte** data = parser->stack;
1414
1415
1416 if ( num_args > field->array_max )
1417 num_args = field->array_max;
1418
1419 FT_TRACE4(( " [" ));
1420
1421 /* store count */
1422 *qcount = (FT_Byte)num_args;
1423
1424 val = 0;
1425 while ( num_args > 0 )
1426 {
1427 val = ADD_LONG( val, cff_parse_num( parser, data++ ) );
1428 switch ( field->size )
1429 {
1430 case (8 / FT_CHAR_BIT):
1431 *(FT_Byte*)q = (FT_Byte)val;
1432 break;
1433
1434 case (16 / FT_CHAR_BIT):
1435 *(FT_Short*)q = (FT_Short)val;
1436 break;
1437
1438 case (32 / FT_CHAR_BIT):
1439 *(FT_Int32*)q = (FT_Int)val;
1440 break;
1441
1442 default: /* for 64-bit systems */
1443 *(FT_Long*)q = val;
1444 }
1445
1446 FT_TRACE4(( " %ld", val ));
1447
1448 q += field->size;
1449 num_args--;
1450 }
1451
1452 FT_TRACE4(( "]\n" ));
1453 }
1454 break;
1455
1456 default: /* callback or blend */
1457 error = field->reader( parser );
1458 if ( error )
1459 goto Exit;
1460 }
1461 goto Found;
1462 }
1463 }
1464
1465 /* this is an unknown operator, or it is unsupported; */

1476 } /* while ( p < limit ) */
1477
1478 Exit:
1479 return error;
