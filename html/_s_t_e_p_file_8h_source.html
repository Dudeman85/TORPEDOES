771 template <typename T,uint64_t min_cnt, uint64_t max_cnt>
772 struct InternGenericConvertList {
773 void operator()(ListOf<T, min_cnt, max_cnt>& out, const std::shared_ptr< const EXPRESS::DataType >& inp_base, const STEP::DB& db) {
774
775 const EXPRESS::LIST* inp = dynamic_cast<const EXPRESS::LIST*>(inp_base.get());
776 if (!inp) {
777 throw TypeError("type error reading aggregate");
778 }
779
780 // XXX is this really how the EXPRESS notation ([?:3],[1:3]) is intended?
781 if (max_cnt && inp->GetSize() > max_cnt) {
782 ASSIMP_LOG_WARN("too many aggregate elements");
783 }
784 else if (inp->GetSize() < min_cnt) {
785 ASSIMP_LOG_WARN("too few aggregate elements");
786 }
787
788 out.reserve(inp->GetSize());
789 for(size_t i = 0; i < inp->GetSize(); ++i) {
790
791 out.push_back( typename ListOf<T, min_cnt, max_cnt>::OutScalar() );
792 try{
793 GenericConvert(out.back(),(*inp)[i], db);
794 }
795 catch(const TypeError& t) {
796 throw TypeError(t.what() +std::string(" of aggregate"));
797 }
798 }
799 }
800 };

802 template <typename T>
803 struct InternGenericConvert< Lazy<T> > {
804 void operator()(Lazy<T>& out, const std::shared_ptr< const EXPRESS::DataType >& in_base, const STEP::DB& db) {
805 const EXPRESS::ENTITY* in = dynamic_cast<const EXPRESS::ENTITY*>(in_base.get());
806 if (!in) {
807 throw TypeError("type error reading entity");
808 }
809 out = Couple<T>(db).GetObject(*in);
810 }
