146template<class T>
147inline
148bool IOStreamBuffer<T>::open( IOStream *stream ) {
149 // file still opened!
150 if ( nullptr != m_stream ) {
151 return false;
152 }
153
154 // Invalid stream pointer
155 if ( nullptr == stream ) {
156 return false;
157 }
158
159 m_stream = stream;
160 m_filesize = m_stream->FileSize();
161 if ( m_filesize == 0 ) {
162 return false;
163 }
164 if ( m_filesize < m_cacheSize ) {
165 m_cacheSize = m_filesize;
166 }
167
168 m_numBlocks = m_filesize / m_cacheSize;
169 if ( ( m_filesize % m_cacheSize ) > 0 ) {
170 m_numBlocks++;
171 }
172
173 return true;
174}

208bool IOStreamBuffer<T>::readNextBlock() {
209 m_stream->Seek( m_filePos, aiOrigin_SET );

242template<class T>
243inline
244bool IOStreamBuffer<T>::getNextDataLine( std::vector<T> &buffer, T continuationToken ) {
245 buffer.resize( m_cacheSize );
246 if ( m_cachePos >= m_cacheSize || 0 == m_filePos ) {
247 if ( !readNextBlock() ) {
248 return false;
249 }
250 }
251
252 bool continuationFound( false );
253 size_t i = 0;
254 for( ;; ) {
255 if ( continuationToken == m_cache[ m_cachePos ] ) {
256 continuationFound = true;
257 ++m_cachePos;
258 }
259 if ( IsLineEnd( m_cache[ m_cachePos ] ) ) {
260 if ( !continuationFound ) {
261 // the end of the data line
262 break;
263 } else {
264 // skip line end
265 while ( m_cache[m_cachePos] != '\n') {
266 ++m_cachePos;
267 }
268 ++m_cachePos;
269 continuationFound = false;
270 }
271 }
272
273 buffer[ i ] = m_cache[ m_cachePos ];
274 ++m_cachePos;
275 ++i;
276 if (m_cachePos >= size()) {
277 break;
278 }
279 if ( m_cachePos >= m_cacheSize ) {
280 if ( !readNextBlock() ) {
281 return false;
282 }
283 }
284 }
285
286 buffer[ i ] = '\n';
287 ++m_cachePos;
288
289 return true;
290}

297template<class T>
298inline
299bool IOStreamBuffer<T>::getNextLine(std::vector<T> &buffer) {
300 buffer.resize(m_cacheSize);
301 if ( isEndOfCache( m_cachePos, m_cacheSize ) || 0 == m_filePos) {
302 if (!readNextBlock()) {
303 return false;
304 }
305 }
306
307 if (IsLineEnd(m_cache[m_cachePos])) {
308 // skip line end
309 while (m_cache[m_cachePos] != '\n') {
310 ++m_cachePos;
311 }
312 ++m_cachePos;
313 if ( isEndOfCache( m_cachePos, m_cacheSize ) ) {
314 if ( !readNextBlock() ) {
315 return false;
316 }
317 }
318 }
319
320 size_t i( 0 );
321 while (!IsLineEnd(m_cache[ m_cachePos ])) {
322 buffer[i] = m_cache[ m_cachePos ];
323 ++m_cachePos;
324 ++i;
325 if (m_cachePos >= m_cacheSize) {
326 if (!readNextBlock()) {
327 return false;
328 }
329 }
330 }
331 buffer[i] = '\n';
332 ++m_cachePos;
333
334 return true;
335}

337template<class T>
338inline
339bool IOStreamBuffer<T>::getNextBlock( std::vector<T> &buffer) {
340 // Return the last block-value if getNextLine was used before
341 if ( 0 != m_cachePos ) {
342 buffer = std::vector<T>( m_cache.begin() + m_cachePos, m_cache.end() );
343 m_cachePos = 0;
344 } else {
345 if ( !readNextBlock() ) {
346 return false;
347 }
348
349 buffer = std::vector<T>(m_cache.begin(), m_cache.end());
350 }
351
352 return true;
