167 class TriangleFans
168 {
169 public:
171 TriangleFans(long sizeTFAN = O3DGC_TFANS_MIN_SIZE_TFAN_SIZE_BUFFER,
172 long verticesSize = O3DGC_TFANS_MIN_SIZE_ALLOCATED_VERTICES_BUFFER)
173 {
174 assert(sizeTFAN > 0);
175 assert(verticesSize > 0);
176 m_numTFANs = 0;
177 m_numVertices = 0;
178 m_verticesAllocatedSize = verticesSize;
179 m_sizeTFANAllocatedSize = sizeTFAN;
180 m_sizeTFAN = new long [m_sizeTFANAllocatedSize];
181 m_vertices = new long [m_verticesAllocatedSize];
182 };
184 ~TriangleFans(void)
185 {
186 delete [] m_vertices;
187 delete [] m_sizeTFAN;
188 };
189
190 O3DGCErrorCode Allocate(long sizeTFAN, long verticesSize)
191 {
192 assert(sizeTFAN > 0);
193 assert(verticesSize > 0);
194 m_numTFANs = 0;
195 m_numVertices = 0;
196 if (m_verticesAllocatedSize < verticesSize)
197 {
198 delete [] m_vertices;
199 m_verticesAllocatedSize = verticesSize;
200 m_vertices = new long [m_verticesAllocatedSize];
201 }
202 if (m_sizeTFANAllocatedSize < sizeTFAN)
203 {
204 delete [] m_sizeTFAN;
205 m_sizeTFANAllocatedSize = sizeTFAN;
206 m_sizeTFAN = new long [m_sizeTFANAllocatedSize];
207 }
208 return O3DGC_OK;
209 };
210 O3DGCErrorCode Clear()
211 {
212 m_numTFANs = 0;
213 m_numVertices = 0;
214 return O3DGC_OK;
215 }
216 O3DGCErrorCode AddVertex(long vertex)
217 {
218 assert(m_numTFANs >= 0);
219 assert(m_numTFANs < m_sizeTFANAllocatedSize);
220 assert(m_numVertices >= 0);
221 ++m_numVertices;
222 if (m_numVertices == m_verticesAllocatedSize)
223 {
224 m_verticesAllocatedSize *= 2;
225 long * tmp = m_vertices;
226 m_vertices = new long [m_verticesAllocatedSize];
227 memcpy(m_vertices, tmp, sizeof(long) * m_numVertices);
228 delete [] tmp;
229 }
230 m_vertices[m_numVertices-1] = vertex;
231 ++m_sizeTFAN[m_numTFANs-1];
232 return O3DGC_OK;
233 }
234 O3DGCErrorCode AddTFAN()
235 {
236 assert(m_numTFANs >= 0);
237 ++m_numTFANs;
238 if (m_numTFANs == m_sizeTFANAllocatedSize)
239 {
240 m_sizeTFANAllocatedSize *= 2;
241 long * tmp = m_sizeTFAN;
242 m_sizeTFAN = new long [m_sizeTFANAllocatedSize];
243 memcpy(m_sizeTFAN, tmp, sizeof(long) * m_numTFANs);
244 delete [] tmp;
245 }
246 m_sizeTFAN[m_numTFANs-1] = (m_numTFANs > 1) ? m_sizeTFAN[m_numTFANs-2] : 0;
247 return O3DGC_OK;
248 }
249 long Begin(long tfan) const
250 {
251 assert(tfan < m_numTFANs);
252 assert(tfan >= 0);
253 return (tfan>0)?m_sizeTFAN[tfan-1]:0;
254 }
255 long End(long tfan) const
256 {
257 assert(tfan < m_numTFANs);
258 assert(tfan >= 0);
259 return m_sizeTFAN[tfan];
260 }
261 long GetVertex(long vertex) const
262 {
263 assert(vertex < m_numVertices);
264 assert(vertex >= 0);
265 return m_vertices[vertex];
266 }
267 long GetTFANSize(long tfan) const
268 {
269 return End(tfan) - Begin(tfan);
270 }
271 long GetNumTFANs() const
272 {
273 return m_numTFANs;
274 }
275 long GetNumVertices() const
276 {
277 return m_numVertices;
278 }
279

281 long m_verticesAllocatedSize;
282 long m_sizeTFANAllocatedSize;
283 long m_numTFANs;
284 long m_numVertices;
285 long * m_vertices;
