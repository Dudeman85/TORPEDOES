381 // Finds the item in the given library by its reference, throws if not found
382 template <typename Type>
383 const Type& ColladaParser::ResolveLibraryReference( const std::map<std::string, Type>& pLibrary, const std::string& pURL) const
384 {
385 typename std::map<std::string, Type>::const_iterator it = pLibrary.find( pURL);
386 if( it == pLibrary.end())
387 ThrowException( Formatter::format() << "Unable to resolve library reference \"" << pURL << "\"." );
388 return it->second;
389 }
