228 static std::unique_ptr<VmdMotion> LoadFromStream(std::ifstream *stream)
229 {
230
231 char buffer[30];
232 auto result = mmd::make_unique<VmdMotion>();
233
234 // magic and version
235 stream->read((char*) buffer, 30);
236 if (strncmp(buffer, "Vocaloid Motion Data", 20))
237 {
238 std::cerr << "invalid vmd file." << std::endl;
239 return nullptr;
240 }
241 result->version = std::atoi(buffer + 20);
242
243 // name
244 stream->read(buffer, 20);
245 result->model_name = std::string(buffer);
246
247 // bone frames
248 int bone_frame_num;
249 stream->read((char*) &bone_frame_num, sizeof(int));
250 result->bone_frames.resize(bone_frame_num);
251 for (int i = 0; i < bone_frame_num; i++)
252 {
253 result->bone_frames[i].Read(stream);
254 }
255
256 // face frames
257 int face_frame_num;
258 stream->read((char*) &face_frame_num, sizeof(int));
259 result->face_frames.resize(face_frame_num);
260 for (int i = 0; i < face_frame_num; i++)
261 {
262 result->face_frames[i].Read(stream);
263 }
264
265 // camera frames
266 int camera_frame_num;
267 stream->read((char*) &camera_frame_num, sizeof(int));
268 result->camera_frames.resize(camera_frame_num);
269 for (int i = 0; i < camera_frame_num; i++)
270 {
271 result->camera_frames[i].Read(stream);
272 }
273
274 // light frames
275 int light_frame_num;
276 stream->read((char*) &light_frame_num, sizeof(int));
277 result->light_frames.resize(light_frame_num);
278 for (int i = 0; i < light_frame_num; i++)
279 {
280 result->light_frames[i].Read(stream);
281 }
282
283 // unknown2
284 stream->read(buffer, 4);
285
286 // ik frames
287 if (stream->peek() != std::ios::traits_type::eof())
288 {
289 int ik_num;
290 stream->read((char*) &ik_num, sizeof(int));
291 result->ik_frames.resize(ik_num);
292 for (int i = 0; i < ik_num; i++)
293 {
294 result->ik_frames[i].Read(stream);
295 }
296 }
297
298 if (stream->peek() != std::ios::traits_type::eof())
299 {
300 std::cerr << "vmd stream has unknown data." << std::endl;
301 }
302
303 return result;
304 }
