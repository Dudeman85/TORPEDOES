211template<class T>
212Ref<T> LazyDict<T>::Get(const char* id)
213{
214 id = T::TranslateId(mAsset, id);
215
216 typename Dict::iterator it = mObjsById.find(id);
217 if (it != mObjsById.end()) { // already created?
218 return Ref<T>(mObjs, it->second);
219 }
220
221 // read it from the JSON object
222 if (!mDict) {
223 throw DeadlyImportError("GLTF: Missing section \"" + std::string(mDictId) + "\"");
224 }
225
226 Value::MemberIterator obj = mDict->FindMember(id);
227 if (obj == mDict->MemberEnd()) {
228 throw DeadlyImportError("GLTF: Missing object with id \"" + std::string(id) + "\" in \"" + mDictId + "\"");
229 }
230 if (!obj->value.IsObject()) {
231 throw DeadlyImportError("GLTF: Object with id \"" + std::string(id) + "\" is not a JSON object");
232 }
233
234 // create an instance of the given type
235 T* inst = new T();
236 inst->id = id;
237 ReadMember(obj->value, "name", inst->name);
238 inst->Read(obj->value, mAsset);
239 return Add(inst);
240}

242template<class T>
243Ref<T> LazyDict<T>::Add(T* obj)
244{
245 unsigned int idx = unsigned(mObjs.size());
246 mObjs.push_back(obj);
247 mObjsById[obj->id] = idx;
248 mAsset.mUsedIds[obj->id] = true;
249 return Ref<T>(mObjs, idx);
250}

252template<class T>
253Ref<T> LazyDict<T>::Create(const char* id)
254{
255 Asset::IdMap::iterator it = mAsset.mUsedIds.find(id);
256 if (it != mAsset.mUsedIds.end()) {
257 throw DeadlyImportError("GLTF: two objects with the same ID exist");
258 }
259 T* inst = new T();
260 inst->id = id;
261 return Add(inst);
262}

289inline void Buffer::Read(Value& obj, Asset& r)
290{
291 size_t statedLength = MemberOrDefault<size_t>(obj, "byteLength", 0);
292 byteLength = statedLength;
293
294 Value* it = FindString(obj, "uri");
295 if (!it) {
296 if (statedLength > 0) {
297 throw DeadlyImportError("GLTF: buffer with non-zero length missing the \"uri\" attribute");
298 }
299 return;
300 }
301
302 const char* uri = it->GetString();
303
304 glTFCommon::Util::DataURI dataURI;
305 if (ParseDataURI(uri, it->GetStringLength(), dataURI)) {
306 if (dataURI.base64) {
307 uint8_t* data = 0;
308 this->byteLength = Util::DecodeBase64(dataURI.data, dataURI.dataLength, data);
309 this->mData.reset(data, std::default_delete<uint8_t[]>());
310
311 if (statedLength > 0 && this->byteLength != statedLength) {
312 throw DeadlyImportError("GLTF: buffer \"" + id + "\", expected " + to_string(statedLength) +
313 " bytes, but found " + to_string(dataURI.dataLength));
314 }
315 }
316 else { // assume raw data
317 if (statedLength != dataURI.dataLength) {
318 throw DeadlyImportError("GLTF: buffer \"" + id + "\", expected " + to_string(statedLength) +
319 " bytes, but found " + to_string(dataURI.dataLength));
320 }
321
322 this->mData.reset(new uint8_t[dataURI.dataLength], std::default_delete<uint8_t[]>());
323 memcpy( this->mData.get(), dataURI.data, dataURI.dataLength );
324 }
325 }
326 else { // Local file
327 if (byteLength > 0) {
328 std::string dir = !r.mCurrentAssetDir.empty() ? (r.mCurrentAssetDir + "/") : "";
329
330 IOStream* file = r.OpenFile(dir + uri, "rb");
331 if (file) {
332 bool ok = LoadFromStream(*file, byteLength);
333 delete file;
334
335 if (!ok)
336 throw DeadlyImportError("GLTF: error while reading referenced file \"" + std::string(uri) + "\"" );
337 }
338 else {
339 throw DeadlyImportError("GLTF: could not open referenced file \"" + std::string(uri) + "\"");
340 }
341 }
342 }
343}

345inline bool Buffer::LoadFromStream(IOStream& stream, size_t length, size_t baseOffset)
346{
347 byteLength = length ? length : stream.FileSize();
348
349 if (baseOffset) {
350 stream.Seek(baseOffset, aiOrigin_SET);
351 }
352
353 mData.reset(new uint8_t[byteLength], std::default_delete<uint8_t[]>());
354
355 if (stream.Read(mData.get(), byteLength, 1) != 1) {
356 return false;
357 }
358 return true;
359}

361inline void Buffer::EncodedRegion_Mark(const size_t pOffset, const size_t pEncodedData_Length, uint8_t* pDecodedData, const size_t pDecodedData_Length, const std::string& pID)
362{
363 // Check pointer to data
364 if(pDecodedData == nullptr) throw DeadlyImportError("GLTF: for marking encoded region pointer to decoded data must be provided.");
365
366 // Check offset
367 if(pOffset > byteLength)
368 {
369 const uint8_t val_size = 32;
370
371 char val[val_size];
372
373 ai_snprintf(val, val_size, "%llu", (long long)pOffset);
374 throw DeadlyImportError(std::string("GLTF: incorrect offset value (") + val + ") for marking encoded region.");
375 }
376
377 // Check length
378 if((pOffset + pEncodedData_Length) > byteLength)
379 {
380 const uint8_t val_size = 64;
381
382 char val[val_size];
383
384 ai_snprintf(val, val_size, "%llu, %llu", (long long)pOffset, (long long)pEncodedData_Length);
385 throw DeadlyImportError(std::string("GLTF: encoded region with offset/length (") + val + ") is out of range.");
386 }
387
388 // Add new region
389 EncodedRegion_List.push_back(new SEncodedRegion(pOffset, pEncodedData_Length, pDecodedData, pDecodedData_Length, pID));
390 // And set new value for "byteLength"
391 byteLength += (pDecodedData_Length - pEncodedData_Length);
392}

412inline bool Buffer::ReplaceData(const size_t pBufferData_Offset, const size_t pBufferData_Count, const uint8_t* pReplace_Data, const size_t pReplace_Count)
413{
414const size_t new_data_size = byteLength + pReplace_Count - pBufferData_Count;
415
416uint8_t* new_data;
417
418 if((pBufferData_Count == 0) || (pReplace_Count == 0) || (pReplace_Data == nullptr)) return false;
419
420 new_data = new uint8_t[new_data_size];
421 // Copy data which place before replacing part.
422 memcpy(new_data, mData.get(), pBufferData_Offset);
423 // Copy new data.
424 memcpy(&new_data[pBufferData_Offset], pReplace_Data, pReplace_Count);
425 // Copy data which place after replacing part.
426 memcpy(&new_data[pBufferData_Offset + pReplace_Count], &mData.get()[pBufferData_Offset + pBufferData_Count], pBufferData_Offset);
427 // Apply new data
428 mData.reset(new_data, std::default_delete<uint8_t[]>());
429 byteLength = new_data_size;
430
431 return true;
432}

434inline size_t Buffer::AppendData(uint8_t* data, size_t length)
435{
436 size_t offset = this->byteLength;
437 Grow(length);

442inline void Buffer::Grow(size_t amount)
443{
444 if (amount <= 0) return;
445 if (capacity >= byteLength + amount)
446 {
447 byteLength += amount;
448 return;
449 }
450
451 // Shift operation is standard way to divide integer by 2, it doesn't cast it to float back and forth, also works for odd numbers,
452 // originally it would look like: static_cast<size_t>(capacity * 1.5f)
453 capacity = std::max(capacity + (capacity >> 1), byteLength + amount);
454
455 uint8_t* b = new uint8_t[capacity];
456 if (mData) memcpy(b, mData.get(), byteLength);
457 mData.reset(b, std::default_delete<uint8_t[]>());
458 byteLength += amount;
459}

511inline uint8_t* Accessor::GetPointer()
512{
513 if (!bufferView || !bufferView->buffer) return 0;
514 uint8_t* basePtr = bufferView->buffer->GetPointer();
515 if (!basePtr) return 0;
516
517 size_t offset = byteOffset + bufferView->byteOffset;
518
519 // Check if region is encoded.
520 if(bufferView->buffer->EncodedRegion_Current != nullptr)
521 {
522 const size_t begin = bufferView->buffer->EncodedRegion_Current->Offset;
523 const size_t end = begin + bufferView->buffer->EncodedRegion_Current->DecodedData_Length;
524
525 if((offset >= begin) && (offset < end))
526 return &bufferView->buffer->EncodedRegion_Current->DecodedData[offset - begin];
527 }
528
529 return basePtr + offset;
530}

533 inline void CopyData(size_t count,
534 const uint8_t* src, size_t src_stride,
535 uint8_t* dst, size_t dst_stride)
536 {
537 if (src_stride == dst_stride) {
538 memcpy(dst, src, count * src_stride);
539 }
540 else {
541 size_t sz = std::min(src_stride, dst_stride);
542 for (size_t i = 0; i < count; ++i) {
543 memcpy(dst, src, sz);
544 if (sz < dst_stride) {
545 memset(dst + sz, 0, dst_stride - sz);
546 }
547 src += src_stride;
548 dst += dst_stride;
549 }
550 }
551 }

554template<class T>
555bool Accessor::ExtractData(T*& outData)
556{
557 uint8_t* data = GetPointer();
558 if (!data) return false;
559
560 const size_t elemSize = GetElementSize();
561 const size_t totalSize = elemSize * count;
562
563 const size_t stride = byteStride ? byteStride : elemSize;
564
565 const size_t targetElemSize = sizeof(T);
566 ai_assert(elemSize <= targetElemSize);
567
568 ai_assert(count*stride <= bufferView->byteLength);
569
570 outData = new T[count];
571 if (stride == elemSize && targetElemSize == elemSize) {
572 memcpy(outData, data, totalSize);
573 }
574 else {
575 for (size_t i = 0; i < count; ++i) {
576 memcpy(outData + i, data + i*stride, elemSize);
577 }
578 }
579
580 return true;
581}

583inline void Accessor::WriteData(size_t count, const void* src_buffer, size_t src_stride)
584{
585 uint8_t* buffer_ptr = bufferView->buffer->GetPointer();
586 size_t offset = byteOffset + bufferView->byteOffset;
587
588 size_t dst_stride = GetNumComponents() * GetBytesPerComponent();
589
590 const uint8_t* src = reinterpret_cast<const uint8_t*>(src_buffer);
591 uint8_t* dst = reinterpret_cast< uint8_t*>(buffer_ptr + offset);
592
593 ai_assert(dst + count*dst_stride <= buffer_ptr + bufferView->buffer->byteLength);
594 CopyData(count, src, src_stride, dst, dst_stride);
595}

629{

631 if (Value* extensions = FindObject(obj, "extensions")) {
632 if (r.extensionsUsed.KHR_binary_glTF) {
633 if (Value* ext = FindObject(*extensions, "KHR_binary_glTF")) {
634
635 width = MemberOrDefault(*ext, "width", 0);
636 height = MemberOrDefault(*ext, "height", 0);
637
638 ReadMember(*ext, "mimeType", mimeType);
639
640 const char* bufferViewId;
641 if (ReadMember(*ext, "bufferView", bufferViewId)) {
642 Ref<BufferView> bv = r.bufferViews.Get(bufferViewId);
643 if (bv) {
644 mDataLength = bv->byteLength;
645 mData.reset(new uint8_t[mDataLength]);
646 memcpy(mData.get(), bv->buffer->GetPointer() + bv->byteOffset, mDataLength);
647 }
648 }
649 }
650 }
651 }
652
653 if (!mDataLength) {
654 if (Value* uri = FindString(obj, "uri")) {
655 const char* uristr = uri->GetString();
656
657 glTFCommon::Util::DataURI dataURI;
658 if (ParseDataURI(uristr, uri->GetStringLength(), dataURI)) {
659 mimeType = dataURI.mediaType;
660 if (dataURI.base64) {
661 uint8_t *ptr = nullptr;
662 mDataLength = glTFCommon::Util::DecodeBase64(dataURI.data, dataURI.dataLength, ptr);
663 mData.reset(ptr);
664 }
665 }
666 else {
667 this->uri = uristr;
668 }
669 }
670 }

811 }
812 else if ((pos = Compare(attr, "NORMAL"))) {
813 v = &(p.attributes.normal);
814 }
815 else if ((pos = Compare(attr, "TEXCOORD"))) {
816 v = &(p.attributes.texcoord);
817 }

819 v = &(p.attributes.color);
820 }
821 else if ((pos = Compare(attr, "JOINT"))) {
822 v = &(p.attributes.joint);
823 }
824 else if ((pos = Compare(attr, "JOINTMATRIX"))) {

835inline void Mesh::Read(Value& pJSON_Object, Asset& pAsset_Root)
836{
837 /****************** Mesh primitives ******************/
838 if (Value* primitives = FindArray(pJSON_Object, "primitives")) {
839 this->primitives.resize(primitives->Size());
840 for (unsigned int i = 0; i < primitives->Size(); ++i) {
841 Value& primitive = (*primitives)[i];
842
843 Primitive& prim = this->primitives[i];
844 prim.mode = MemberOrDefault(primitive, "mode", PrimitiveMode_TRIANGLES);
845
846 if (Value* attrs = FindObject(primitive, "attributes")) {
847 for (Value::MemberIterator it = attrs->MemberBegin(); it != attrs->MemberEnd(); ++it) {
848 if (!it->value.IsString()) continue;
849 const char* attr = it->name.GetString();
850 // Valid attribute semantics include POSITION, NORMAL, TEXCOORD, COLOR, JOINT, JOINTMATRIX,
851 // and WEIGHT.Attribute semantics can be of the form[semantic]_[set_index], e.g., TEXCOORD_0, TEXCOORD_1, etc.
852
853 int undPos = 0;
854 Mesh::AccessorList* vec = 0;
855 if (GetAttribVector(prim, attr, vec, undPos)) {
856 size_t idx = (attr[undPos] == '_') ? atoi(attr + undPos + 1) : 0;
857 if ((*vec).size() <= idx) (*vec).resize(idx + 1);
858 (*vec)[idx] = pAsset_Root.accessors.Get(it->value.GetString());
859 }
860 }
861 }
862
863 if (Value* indices = FindString(primitive, "indices")) {
864 prim.indices = pAsset_Root.accessors.Get(indices->GetString());
865 }
866
867 if (Value* material = FindString(primitive, "material")) {
868 prim.material = pAsset_Root.materials.Get(material->GetString());
869 }
870 }
871 }
872
873 /****************** Mesh extensions ******************/
874 Value* json_extensions = FindObject(pJSON_Object, "extensions");
875
876 if(json_extensions == nullptr) goto mr_skip_extensions;
877
878 for(Value::MemberIterator it_memb = json_extensions->MemberBegin(); it_memb != json_extensions->MemberEnd(); it_memb++)
879 {
880#ifdef ASSIMP_IMPORTER_GLTF_USE_OPEN3DGC
881 if(it_memb->name.GetString() == std::string("Open3DGC-compression"))
882 {
883 // Search for compressed data.
884 // Compressed data contain description of part of "buffer" which is encoded. This part must be decoded and
885 // new data will replace old encoded part by request. In fact \"compressedData\" is kind of "accessor" structure.
886 Value* comp_data = FindObject(it_memb->value, "compressedData");
887
888 if(comp_data == nullptr) throw DeadlyImportError("GLTF: \"Open3DGC-compression\" must has \"compressedData\".");
889
890 ASSIMP_LOG_INFO("GLTF: Decompressing Open3DGC data.");
891
892 /************** Read data from JSON-document **************/
893 #define MESH_READ_COMPRESSEDDATA_MEMBER(pFieldName, pOut) \
894 if(!ReadMember(*comp_data, pFieldName, pOut)) \
895 { \
896 throw DeadlyImportError(std::string("GLTF: \"compressedData\" must has \"") + pFieldName + "\"."); \
897 }
898
899 const char* mode_str;
900 const char* type_str;
901 ComponentType component_type;
902 SCompression_Open3DGC* ext_o3dgc = new SCompression_Open3DGC;
903
904 MESH_READ_COMPRESSEDDATA_MEMBER("buffer", ext_o3dgc->Buffer);
905 MESH_READ_COMPRESSEDDATA_MEMBER("byteOffset", ext_o3dgc->Offset);
906 MESH_READ_COMPRESSEDDATA_MEMBER("componentType", component_type);
907 MESH_READ_COMPRESSEDDATA_MEMBER("type", type_str);
908 MESH_READ_COMPRESSEDDATA_MEMBER("count", ext_o3dgc->Count);
909 MESH_READ_COMPRESSEDDATA_MEMBER("mode", mode_str);
910 MESH_READ_COMPRESSEDDATA_MEMBER("indicesCount", ext_o3dgc->IndicesCount);
911 MESH_READ_COMPRESSEDDATA_MEMBER("verticesCount", ext_o3dgc->VerticesCount);
912
913 #undef MESH_READ_COMPRESSEDDATA_MEMBER
914
915 // Check some values
916 if(strcmp(type_str, "SCALAR")) throw DeadlyImportError("GLTF: only \"SCALAR\" type is supported for compressed data.");
917 if(component_type != ComponentType_UNSIGNED_BYTE) throw DeadlyImportError("GLTF: only \"UNSIGNED_BYTE\" component type is supported for compressed data.");
918
919 // Set read/write data mode.
920 if(strcmp(mode_str, "binary") == 0)
921 ext_o3dgc->Binary = true;
922 else if(strcmp(mode_str, "ascii") == 0)
923 ext_o3dgc->Binary = false;
924 else
925 throw DeadlyImportError(std::string("GLTF: for compressed data supported modes is: \"ascii\", \"binary\". Not the: \"") + mode_str + "\".");
926
927 /************************ Decoding ************************/
928 Decode_O3DGC(*ext_o3dgc, pAsset_Root);
929 Extension.push_back(ext_o3dgc);// store info in mesh extensions list.
930 }// if(it_memb->name.GetString() == "Open3DGC-compression")
931 else
932#endif
933 {
934 throw DeadlyImportError(std::string("GLTF: Unknown mesh extension: \"") + it_memb->name.GetString() + "\".");
935 }
936 }// for(Value::MemberIterator it_memb = json_extensions->MemberBegin(); it_memb != json_extensions->MemberEnd(); json_extensions++)
937
938mr_skip_extensions:
939
940 return;// After label some operators must be present.

944inline void Mesh::Decode_O3DGC(const SCompression_Open3DGC& pCompression_Open3DGC, Asset& pAsset_Root)
945{
946typedef unsigned short IndicesType;
947
948o3dgc::SC3DMCDecoder<IndicesType> decoder;
949o3dgc::IndexedFaceSet<IndicesType> ifs;
950o3dgc::BinaryStream bstream;
951uint8_t* decoded_data;
952size_t decoded_data_size = 0;
953Ref<Buffer> buf = pAsset_Root.buffers.Get(pCompression_Open3DGC.Buffer);
954
955 // Read data from buffer and place it in BinaryStream for decoder.
956 // Just "Count" because always is used type equivalent to uint8_t.
957 bstream.LoadFromBuffer(&buf->GetPointer()[pCompression_Open3DGC.Offset], static_cast<unsigned long>(pCompression_Open3DGC.Count));
958
959 // After decoding header we can get size of primitives.
960 if(decoder.DecodeHeader(ifs, bstream) != o3dgc::O3DGC_OK) throw DeadlyImportError("GLTF: can not decode Open3DGC header.");
961
962 /****************** Get sizes of arrays and check sizes ******************/
963 // Note. See "Limitations for meshes when using Open3DGC-compression".
964
965 // Indices
966 size_t size_coordindex = ifs.GetNCoordIndex() * 3;// See float attributes note.
967
968 if(primitives[0].indices->count != size_coordindex)
969 throw DeadlyImportError("GLTF: Open3DGC. Compressed indices count (" + to_string(size_coordindex) +
970 ") not equal to uncompressed (" + to_string(primitives[0].indices->count) + ").");
971
972 size_coordindex *= sizeof(IndicesType);
973 // Coordinates
974 size_t size_coord = ifs.GetNCoord();// See float attributes note.
975
976 if(primitives[0].attributes.position[0]->count != size_coord)
977 throw DeadlyImportError("GLTF: Open3DGC. Compressed positions count (" + to_string(size_coord) +
978 ") not equal to uncompressed (" + to_string(primitives[0].attributes.position[0]->count) + ").");
979
980 size_coord *= 3 * sizeof(float);
981 // Normals
982 size_t size_normal = ifs.GetNNormal();// See float attributes note.
983
984 if(primitives[0].attributes.normal[0]->count != size_normal)
985 throw DeadlyImportError("GLTF: Open3DGC. Compressed normals count (" + to_string(size_normal) +
986 ") not equal to uncompressed (" + to_string(primitives[0].attributes.normal[0]->count) + ").");
987
988 size_normal *= 3 * sizeof(float);
989 // Additional attributes.
990 std::vector<size_t> size_floatattr;
991 std::vector<size_t> size_intattr;
992
993 size_floatattr.resize(ifs.GetNumFloatAttributes());
994 size_intattr.resize(ifs.GetNumIntAttributes());
995
996 decoded_data_size = size_coordindex + size_coord + size_normal;
997 for(size_t idx = 0, idx_end = size_floatattr.size(), idx_texcoord = 0; idx < idx_end; idx++)
998 {
999 // size = number_of_elements * components_per_element * size_of_component.
1000 // Note. But as you can see above, at first we are use this variable in meaning "count". After checking count of objects...
1001 size_t tval = ifs.GetNFloatAttribute(static_cast<unsigned long>(idx));
1002
1003 switch(ifs.GetFloatAttributeType(static_cast<unsigned long>(idx)))
1004 {
1005 case o3dgc::O3DGC_IFS_FLOAT_ATTRIBUTE_TYPE_TEXCOORD:
1006 // Check situation when encoded data contain texture coordinates but primitive not.
1007 if(idx_texcoord < primitives[0].attributes.texcoord.size())
1008 {
1009 if(primitives[0].attributes.texcoord[idx]->count != tval)
1010 throw DeadlyImportError("GLTF: Open3DGC. Compressed texture coordinates count (" + to_string(tval) +
1011 ") not equal to uncompressed (" + to_string(primitives[0].attributes.texcoord[idx]->count) + ").");
1012
1013 idx_texcoord++;
1014 }
1015 else
1016 {
1017 ifs.SetNFloatAttribute(static_cast<unsigned long>(idx), 0ul);// Disable decoding this attribute.
1018 }
1019
1020 break;
1021 default:
1022 throw DeadlyImportError("GLTF: Open3DGC. Unsupported type of float attribute: " + to_string(ifs.GetFloatAttributeType(static_cast<unsigned long>(idx))));
1023 }
1024
1025 tval *= ifs.GetFloatAttributeDim(static_cast<unsigned long>(idx)) * sizeof(o3dgc::Real);// After checking count of objects we can get size of array.
1026 size_floatattr[idx] = tval;
1027 decoded_data_size += tval;
1028 }
1029
1030 for(size_t idx = 0, idx_end = size_intattr.size(); idx < idx_end; idx++)
1031 {
1032 // size = number_of_elements * components_per_element * size_of_component. See float attributes note.
1033 size_t tval = ifs.GetNIntAttribute(static_cast<unsigned long>(idx));
1034 switch( ifs.GetIntAttributeType(static_cast<unsigned long>(idx) ) )
1035 {
1036 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_UNKOWN:
1037 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_INDEX:
1038 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_JOINT_ID:
1039 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_INDEX_BUFFER_ID:
1040 break;
1041
1042 default:
1043 throw DeadlyImportError("GLTF: Open3DGC. Unsupported type of int attribute: " + to_string(ifs.GetIntAttributeType(static_cast<unsigned long>(idx))));
1044 }
1045
1046 tval *= ifs.GetIntAttributeDim(static_cast<unsigned long>(idx)) * sizeof(long);// See float attributes note.
1047 size_intattr[idx] = tval;
1048 decoded_data_size += tval;
1049 }
1050
1051 // Create array for decoded data.
1052 decoded_data = new uint8_t[decoded_data_size];
1053
1054 /****************** Set right array regions for decoder ******************/
1055
1056 auto get_buf_offset = [](Ref<Accessor>& pAccessor) -> size_t { return pAccessor->byteOffset + pAccessor->bufferView->byteOffset; };
1057
1058 // Indices
1059 ifs.SetCoordIndex((IndicesType* const)(decoded_data + get_buf_offset(primitives[0].indices)));
1060 // Coordinates
1061 ifs.SetCoord((o3dgc::Real* const)(decoded_data + get_buf_offset(primitives[0].attributes.position[0])));
1062 // Normals
1063 if(size_normal)
1064 {
1065 ifs.SetNormal((o3dgc::Real* const)(decoded_data + get_buf_offset(primitives[0].attributes.normal[0])));
1066 }
1067
1068 for(size_t idx = 0, idx_end = size_floatattr.size(), idx_texcoord = 0; idx < idx_end; idx++)
1069 {
1070 switch(ifs.GetFloatAttributeType(static_cast<unsigned long>(idx)))
1071 {
1072 case o3dgc::O3DGC_IFS_FLOAT_ATTRIBUTE_TYPE_TEXCOORD:
1073 if(idx_texcoord < primitives[0].attributes.texcoord.size())
1074 {
1075 // See above about absent attributes.
1076 ifs.SetFloatAttribute(static_cast<unsigned long>(idx), (o3dgc::Real* const)(decoded_data + get_buf_offset(primitives[0].attributes.texcoord[idx])));
1077 idx_texcoord++;
1078 }
1079
1080 break;
1081 default:
1082 throw DeadlyImportError("GLTF: Open3DGC. Unsupported type of float attribute: " + to_string(ifs.GetFloatAttributeType(static_cast<unsigned long>(idx))));
1083 }
1084 }
1085
1086 for(size_t idx = 0, idx_end = size_intattr.size(); idx < idx_end; idx++) {
1087 switch(ifs.GetIntAttributeType(static_cast<unsigned int>(idx))) {
1088 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_UNKOWN:
1089 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_INDEX:
1090 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_JOINT_ID:
1091 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_INDEX_BUFFER_ID:
1092 break;
1093
1094 // ifs.SetIntAttribute(idx, (long* const)(decoded_data + get_buf_offset(primitives[0].attributes.joint)));
1095 default:
1096 throw DeadlyImportError("GLTF: Open3DGC. Unsupported type of int attribute: " + to_string(ifs.GetIntAttributeType(static_cast<unsigned long>(idx))));
1097 }
1098 }
1099
1100 //
1101 // Decode data
1102 //
1103 if ( decoder.DecodePayload( ifs, bstream ) != o3dgc::O3DGC_OK ) {
1104 throw DeadlyImportError( "GLTF: can not decode Open3DGC data." );
1105 }
1106
1107 // Set encoded region for "buffer".
1108 buf->EncodedRegion_Mark(pCompression_Open3DGC.Offset, pCompression_Open3DGC.Count, decoded_data, decoded_data_size, id);
1109 // No. Do not delete "output_data". After calling "EncodedRegion_Mark" bufferView is owner of "output_data".
1110 // "delete [] output_data;"
1111}

1295inline void Asset::ReadBinaryHeader(IOStream& stream)
1296{
1297 GLB_Header header;
1298 if (stream.Read(&header, sizeof(header), 1) != 1) {
1299 throw DeadlyImportError("GLTF: Unable to read the file header");
1300 }
1301
1302 if (strncmp((char*)header.magic, AI_GLB_MAGIC_NUMBER, sizeof(header.magic)) != 0) {
1303 throw DeadlyImportError("GLTF: Invalid binary glTF file");
1304 }
1305
1306 AI_SWAP4(header.version);
1307 asset.version = to_string(header.version);
1308 if (header.version != 1) {
1309 throw DeadlyImportError("GLTF: Unsupported binary glTF version");
1310 }
1311
1312 AI_SWAP4(header.sceneFormat);

1328inline void Asset::Load(const std::string& pFile, bool isBinary)
1329{
1330 mCurrentAssetDir.clear();
1331 int pos = std::max(int(pFile.rfind('/')), int(pFile.rfind('\\')));
1332 if (pos != int(std::string::npos)) mCurrentAssetDir = pFile.substr(0, pos + 1);
1333
1334 shared_ptr<IOStream> stream(OpenFile(pFile.c_str(), "rb", true));
1335 if (!stream) {
1336 throw DeadlyImportError("GLTF: Could not open file for reading");
1337 }
1338
1339 // is binary? then read the header
1340 if (isBinary) {
1341 SetAsBinary(); // also creates the body buffer
1342 ReadBinaryHeader(*stream);
1343 }
1344 else {
1345 mSceneLength = stream->FileSize();
1346 mBodyLength = 0;
1347 }
1348
1349
1350 // read the scene data
1351
1352 std::vector<char> sceneData(mSceneLength + 1);
1353 sceneData[mSceneLength] = '\0';
1354
1355 if (stream->Read(&sceneData[0], 1, mSceneLength) != mSceneLength) {
1356 throw DeadlyImportError("GLTF: Could not read the file contents");
1357 }
1358
1359
1360 // parse the JSON document
1361
1362 Document doc;
1363 doc.ParseInsitu(&sceneData[0]);
1364
1365 if (doc.HasParseError()) {
1366 char buffer[32];
1367 ai_snprintf(buffer, 32, "%d", static_cast<int>(doc.GetErrorOffset()));

1369 + GetParseError_En(doc.GetParseError()));
1370 }
1371
1372 if (!doc.IsObject()) {
1373 throw DeadlyImportError("GLTF: JSON document root must be a JSON object");
1374 }
1375
1376 // Fill the buffer instance for the current file embedded contents
1377 if (mBodyLength > 0) {
1378 if (!mBodyBuffer->LoadFromStream(*stream, mBodyLength, mBodyOffset)) {
1379 throw DeadlyImportError("GLTF: Unable to read gltf file");
1380 }
1381 }
1382
1383
1384 // Load the metadata
1385 asset.Read(doc);
1386 ReadExtensionsUsed(doc);
1387
1388 // Prepare the dictionaries
1389 for (size_t i = 0; i < mDicts.size(); ++i) {
1390 mDicts[i]->AttachToDocument(doc);
1391 }
1392
1393
1394
1395 // Read the "scene" property, which specifies which scene to load
1396 // and recursively load everything referenced by it
1397 if (Value* scene = FindString(doc, "scene")) {
1398 this->scene = scenes.Get(scene->GetString());
1399 }
1400
1401 // Clean up
1402 for (size_t i = 0; i < mDicts.size(); ++i) {
1403 mDicts[i]->DetachFromDocument();
