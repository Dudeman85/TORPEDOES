220 // Reads the current xml node
221 void parseCurrentNode()
222 {
223 char_type* start = P;
224
225 // move forward until '<' found
226 while(*P != L'<' && *P)
227 ++P;
228
229 if (!*P)
230 return;
231
232 if (P - start > 0)
233 {
234 // we found some text, store it
235 if (setText(start, P))
236 return;
237 }
238
239 ++P;
240
241 // based on current token, parse and report next element
242 switch(*P)
243 {
244 case L'/':
245 parseClosingXMLElement();
246 break;
247 case L'?':
248 ignoreDefinition();
249 break;
250 case L'!':
251 if (!parseCDATA())
252 parseComment();
253 break;
254 default:
255 parseOpeningXMLElement();
256 break;
257 }
258 }

262 bool setText(char_type* start, char_type* end)
263 {
264 // check if text is more than 2 characters, and if not, check if there is
265 // only white space, so that this text won't be reported
266 if (end - start < 3)
267 {
268 char_type* p = start;
269 for(; p != end; ++p)
270 if (!isWhiteSpace(*p))
271 break;
272
273 if (p == end)
274 return false;
275 }
276
277 // set current text to the parsed text, and replace xml special characters
278 core::string<char_type> s(start, (int)(end - start));
279 NodeName = replaceSpecialCharacters(s);
280
281 // current XML node type is text
282 CurrentNodeType = EXN_TEXT;
283
284 return true;
285 }

304 {

331 void parseOpeningXMLElement()
332 {
333 CurrentNodeType = EXN_ELEMENT;
334 IsEmptyElement = false;
335 Attributes.clear();
336
337 // find name
338 const char_type* startName = P;
339
340 // find end of element
341 while(*P != L'>' && !isWhiteSpace(*P))
342 ++P;
343
344 const char_type* endName = P;
345
346 // find Attributes
347 while(*P != L'>')
348 {
349 if (isWhiteSpace(*P))
350 ++P;
351 else
352 {
353 if (*P != L'/')
354 {
355 // we've got an attribute
356
357 // read the attribute names
358 const char_type* attributeNameBegin = P;
359
360 while(!isWhiteSpace(*P) && *P != L'=')
361 ++P;
362
363 const char_type* attributeNameEnd = P;
364 ++P;
365
366 // read the attribute value
367 // check for quotes and single quotes, thx to murphy
368 while( (*P != L'\"') && (*P != L'\'') && *P)
369 ++P;
370
371 if (!*P) // malformatted xml file
372 return;
373
374 const char_type attributeQuoteChar = *P;
375
376 ++P;
377 const char_type* attributeValueBegin = P;
378
379 while(*P != attributeQuoteChar && *P)
380 ++P;
381
382 if (!*P) // malformatted xml file
383 return;
384
385 const char_type* attributeValueEnd = P;
386 ++P;
387
388 SAttribute attr;
389 attr.Name = core::string<char_type>(attributeNameBegin,
390 (int)(attributeNameEnd - attributeNameBegin));
391
392 core::string<char_type> s(attributeValueBegin,
393 (int)(attributeValueEnd - attributeValueBegin));
394
395 attr.Value = replaceSpecialCharacters(s);
396 Attributes.push_back(attr);
397 }
398 else
399 {
400 // tag is closed directly
401 ++P;
402 IsEmptyElement = true;
403 break;
404 }
405 }
406 }
407
408 // check if this tag is closing directly
409 if (endName > startName && *(endName-1) == L'/')
410 {
411 // directly closing tag
412 IsEmptyElement = true;
413 endName--;
414 }
415
416 NodeName = core::string<char_type>(startName, (int)(endName - startName));
417
418 ++P;
419 }

423 void parseClosingXMLElement()
424 {
425 CurrentNodeType = EXN_ELEMENT_END;
426 IsEmptyElement = false;
427 Attributes.clear();
428
429 ++P;

434

437 --P;

444 bool parseCDATA()
445 {
446 if (*(P+1) != L'[')
447 return false;
448
449 CurrentNodeType = EXN_CDATA;
450
451 // skip '<![CDATA['
452 int count=0;
453 while( *P && count<8 )
454 {
455 ++P;
456 ++count;
457 }
458
459 if (!*P)
460 return true;
461
462 char_type *cDataBegin = P;
463 char_type *cDataEnd = 0;
464
465 // find end of CDATA
466 while(*P && !cDataEnd)
467 {
468 if (*P == L'>' &&
469 (*(P-1) == L']') &&
470 (*(P-2) == L']'))
471 {
472 cDataEnd = P - 2;
473 }
474
475 ++P;
476 }
477
478 if ( cDataEnd )
479 NodeName = core::string<char_type>(cDataBegin, (int)(cDataEnd - cDataBegin));
480 else
481 NodeName = "";
482
483 return true;
484 }

509 // replaces xml special characters in a string and creates a new one
510 core::string<char_type> replaceSpecialCharacters(
511 core::string<char_type>& origstr)
512 {
513 int pos = origstr.findFirst(L'&');
514 int oldPos = 0;
515
516 if (pos == -1)
517 return origstr;
518
519 core::string<char_type> newstr;
520
521 while(pos != -1 && pos < origstr.size()-2)
522 {
523 // check if it is one of the special characters
524
525 int specialChar = -1;
526 for (int i=0; i<(int)SpecialCharacters.size(); ++i)
527 {
528 const char_type* p = &origstr.c_str()[pos]+1;
529
530 if (equalsn(&SpecialCharacters[i][1], p, SpecialCharacters[i].size()-1))
531 {
532 specialChar = i;
533 break;
534 }
535 }
536
537 if (specialChar != -1)
538 {
539 newstr.append(origstr.subString(oldPos, pos - oldPos));
540 newstr.append(SpecialCharacters[specialChar][0]);
541 pos += SpecialCharacters[specialChar].size();
542 }
543 else
544 {
545 newstr.append(origstr.subString(oldPos, pos - oldPos + 1));
546 pos += 1;
547 }
548
549 // find next &
550 oldPos = pos;
551 pos = origstr.findNext(L'&', pos);
552 }
553
554 if (oldPos < origstr.size()-1)
555 newstr.append(origstr.subString(oldPos, origstr.size()-oldPos));
556
557 return newstr;
558 }

563 bool readFile(IFileReadCallBack* callback)
564 {
565 int size = callback->getSize();
566 size += 4; // We need two terminating 0's at the end.
567 // For ASCII we need 1 0's, for UTF-16 2, for UTF-32 4.
568
569 char* data8 = new char[size];
570
571 if (!callback->read(data8, size-4))
572 {
573 delete [] data8;
574 return false;
575 }
576
577 // add zeros at end
578
579 data8[size-1] = 0;
580 data8[size-2] = 0;
581 data8[size-3] = 0;
582 data8[size-4] = 0;
583
584 char16* data16 = reinterpret_cast<char16*>(data8);
585 char32* data32 = reinterpret_cast<char32*>(data8);
586
587 // now we need to convert the data to the desired target format
588 // based on the byte order mark.
589
590 const unsigned char UTF8[] = {0xEF, 0xBB, 0xBF}; // 0xEFBBBF;
591 const int UTF16_BE = 0xFFFE;
592 const int UTF16_LE = 0xFEFF;
593 const int UTF32_BE = 0xFFFE0000;
594 const int UTF32_LE = 0x0000FEFF;
595
596 // check source for all utf versions and convert to target data format
597
598 if (size >= 4 && data32[0] == (char32)UTF32_BE)
599 {
600 // UTF-32, big endian
601 SourceFormat = ETF_UTF32_BE;
602 convertTextData(data32+1, data8, (size/4)); // data32+1 because we need to skip the header
603 }
604 else
605 if (size >= 4 && data32[0] == (char32)UTF32_LE)
606 {
607 // UTF-32, little endian
608 SourceFormat = ETF_UTF32_LE;
609 convertTextData(data32+1, data8, (size/4)); // data32+1 because we need to skip the header
610 }
611 else
612 if (size >= 2 && data16[0] == UTF16_BE)
613 {
614 // UTF-16, big endian
615 SourceFormat = ETF_UTF16_BE;
616 convertTextData(data16+1, data8, (size/2)); // data16+1 because we need to skip the header
617 }
618 else
619 if (size >= 2 && data16[0] == UTF16_LE)
620 {
621 // UTF-16, little endian
622 SourceFormat = ETF_UTF16_LE;
623 convertTextData(data16+1, data8, (size/2)); // data16+1 because we need to skip the header
624 }
625 else
626 if (size >= 3 && data8[0] == UTF8[0] && data8[1] == UTF8[1] && data8[2] == UTF8[2])
627 {
628 // UTF-8
629 SourceFormat = ETF_UTF8;
630 convertTextData(data8+3, data8, size); // data8+3 because we need to skip the header
631 }
632 else
633 {
634 // ASCII
635 SourceFormat = ETF_ASCII;
636 convertTextData(data8, data8, size);
637 }
638
639 return true;
640 }

648 template<class src_char_type>
649 void convertTextData(src_char_type* source, char* pointerToStore, int sizeWithoutHeader)
650 {
651 // convert little to big endian if necessary
652 if (sizeof(src_char_type) > 1 &&
653 isLittleEndian(TargetFormat) != isLittleEndian(SourceFormat))
654 convertToLittleEndian(source);
655
656 // check if conversion is necessary:
657 if (sizeof(src_char_type) == sizeof(char_type))
658 {
659 // no need to convert
660 TextBegin = (char_type*)source;
661 TextData = (char_type*)pointerToStore;
662 TextSize = sizeWithoutHeader;
663 }
664 else
665 {
666 // convert source into target data format.
667 // TODO: implement a real conversion. This one just
668 // copies bytes. This is a problem when there are
669 // unicode symbols using more than one character.
670
671 TextData = new char_type[sizeWithoutHeader];
672
673 // MSVC debugger complains here about loss of data ...
674 size_t numShift = sizeof( char_type) * 8;
675 assert(numShift < 64);
676 const src_char_type cc = (src_char_type)(((uint64_t(1u) << numShift) - 1));
677 for (int i=0; i<sizeWithoutHeader; ++i)
678 TextData[i] = char_type( source[i] & cc);
679
680 TextBegin = TextData;
681 TextSize = sizeWithoutHeader;
682
683 // delete original data because no longer needed
684 delete [] pointerToStore;
685 }

689 template<class src_char_type>
690 void convertToLittleEndian(src_char_type* t)
691 {
692 if (sizeof(src_char_type) == 4)
693 {
694 // 32 bit
695
696 while(*t)
697 {
698 *t = ((*t & 0xff000000) >> 24) |
699 ((*t & 0x00ff0000) >> 8) |
700 ((*t & 0x0000ff00) << 8) |
701 ((*t & 0x000000ff) << 24);
702 ++t;
703 }
704 }
705 else
706 {
707 // 16 bit
708
709 while(*t)
710 {
711 *t = (*t >> 8) | (*t << 8);
712 ++t;
713 }
714 }

751 bool equalsn(const char_type* str1, const char_type* str2, int len)
752 {
753 int i;
754 for(i=0; str1[i] && str2[i] && i < len; ++i)
755 if (str1[i] != str2[i])
756 return false;
757
758 // if one (or both) of the strings was smaller then they
759 // are only equal if they have the same lenght
760 return (i == len) || (str1[i] == 0 && str2[i] == 0);
761 }
