420 // Gets a new value from EAX call,
421 // validates it,
422 // updates the deferred value,
423 // updates a dirty flag.
424 template<
425 typename TValidator,
426 EaxDirtyFlags TDirtyBit,
427 typename TMemberResult,
428 typename TProps,
429 typename TState>
430 void eax_defer(const EaxCall& call, TState& state, TMemberResult TProps::*member) noexcept
431 {
432 const auto& src = call.get_value<ContextException, const TMemberResult>();
433 TValidator{}(src);
434 const auto& dst_i = state.i.*member;
435 auto& dst_d = state.d.*member;
436 dst_d = src;
437
438 if(dst_i != dst_d)
439 eax_df_ |= TDirtyBit;
440 }

442 template<
443 EaxDirtyFlags TDirtyBit,
444 typename TMemberResult,
445 typename TProps,
446 typename TState>
447 void eax_context_commit_property(TState& state, EaxDirtyFlags& dst_df,
448 TMemberResult TProps::*member) noexcept
449 {
450 if((eax_df_ & TDirtyBit) != EaxDirtyFlags{})
451 {
452 dst_df |= TDirtyBit;
453 const auto& src_d = state.d.*member;
454 state.i.*member = src_d;
455 eax_.*member = src_d;
456 }
