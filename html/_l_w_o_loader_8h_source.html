445inline int LWOImporter::ReadVSizedIntLWO2(uint8_t*& inout)
446{
447 int i;
448 int c = *inout;inout++;
449 if(c != 0xFF)
450 {
451 i = c << 8;
452 c = *inout;inout++;
453 i |= c;
454 }
455 else
456 {
457 c = *inout;inout++;
458 i = c << 16;
459 c = *inout;inout++;
460 i |= c << 8;
461 c = *inout;inout++;
462 i |= c;
463 }
464 return i;
465}

468inline void LWOImporter::GetS0(std::string& out,unsigned int max)
469{
470 unsigned int iCursor = 0;
471 const char*sz = (const char*)mFileBuffer;
472 while (*mFileBuffer)
473 {
474 if (++iCursor > max)
475 {
476 ASSIMP_LOG_WARN("LWO: Invalid file, string is is too long");
477 break;
478 }
479 ++mFileBuffer;
480 }
481 size_t len = (size_t) ((const char*)mFileBuffer-sz);
482 out = std::string(sz,len);
483 mFileBuffer += (len&0x1 ? 1 : 2);
484}
