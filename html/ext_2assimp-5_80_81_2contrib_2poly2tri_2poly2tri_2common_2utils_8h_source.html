64Orientation Orient2d(const Point& pa, const Point& pb, const Point& pc)
65{
66 double detleft = (pa.x - pc.x) * (pb.y - pc.y);
67 double detright = (pa.y - pc.y) * (pb.x - pc.x);
68 double val = detleft - detright;
69 if (val > -EPSILON && val < EPSILON) {
70 return COLLINEAR;
71 } else if (val > 0) {
72 return CCW;
73 }
74 return CW;
75}

111bool InScanArea(const Point& pa, const Point& pb, const Point& pc, const Point& pd)
112{
113 double oadb = (pa.x - pb.x)*(pd.y - pb.y) - (pd.x - pb.x)*(pa.y - pb.y);
114 if (oadb >= -EPSILON) {
115 return false;
116 }
117
118 double oadc = (pa.x - pc.x)*(pd.y - pc.y) - (pd.x - pc.x)*(pa.y - pc.y);
119 if (oadc <= EPSILON) {
120 return false;
121 }
122 return true;
123}
