176//-------------------------------------------------------------------------------
177template <class type, class intype>
178type clamp(intype in)
179{
180 // for unsigned types only ...
181 intype mask = (0x1u << (sizeof(type)*8))-1;
182 return (type)std::max((intype)0,std::min(in,mask));
183}
