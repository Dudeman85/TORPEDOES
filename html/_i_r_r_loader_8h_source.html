150 explicit Node(ET t)
151 : type (t)
152 , scaling (1.0,1.0,1.0) // assume uniform scaling by default
153 , parent()
154 , framesPerSecond (0.0)
155 , id()
156 , sphereRadius (1.0)
157 , spherePolyCountX (100)
158 , spherePolyCountY (100)
159 {
160
161 // Generate a default name for the node
162 char buffer[128];
163 static int cnt;
164 ai_snprintf(buffer, 128, "IrrNode_%i",cnt++);
165 name = std::string(buffer);
166
167 // reserve space for up to 5 materials
168 materials.reserve(5);
169
170 // reserve space for up to 5 children
171 children.reserve(5);
172 }
173
174 // Transformation of the node
175 aiVector3D position, rotation, scaling;
176
177 // Name of the node
178 std::string name;
179
180 // List of all child nodes
181 std::vector<Node*> children;
182
183 // Parent node
184 Node* parent;
185
186 // Animated meshes: frames per second
187 // 0.f if not specified
188 ai_real framesPerSecond;
189
190 // Meshes: path to the mesh to be loaded
191 std::string meshPath;
192 unsigned int id;
193
194 // Meshes: List of materials to be assigned
195 // along with their corresponding material flags
196 std::vector< std::pair<aiMaterial*, unsigned int> > materials;
197
198 // Spheres: radius of the sphere to be generates
199 ai_real sphereRadius;
200
201 // Spheres: Number of polygons in the x,y direction
202 unsigned int spherePolyCountX,spherePolyCountY;
203
204 // List of all animators assigned to the node
205 std::list<Animator> animators;
