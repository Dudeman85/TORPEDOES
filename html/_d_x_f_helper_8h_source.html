60// read pairs of lines, parse group code and value and provide utilities
61// to convert the data to the target data type.
62// do NOT skip empty lines. In DXF files, they count as valid data.
63class LineReader {
64public:
65 LineReader(StreamReaderLE& reader)
66 : splitter(reader,false,true)
67 , groupcode( 0 )
68 , value()
69 , end() {
70 // empty
71 }
72
73 // -----------------------------------------
74 bool Is(int gc, const char* what) const {
75 return groupcode == gc && !strcmp(what,value.c_str());
76 }
77
78 // -----------------------------------------
79 bool Is(int gc) const {
80 return groupcode == gc;
81 }
82
83 // -----------------------------------------
84 int GroupCode() const {
85 return groupcode;
86 }
87
88 // -----------------------------------------
89 const std::string& Value() const {
90 return value;
91 }
92
93 // -----------------------------------------
94 bool End() const {
95 return !((bool)*this);
96 }
97
98 // -----------------------------------------
99 unsigned int ValueAsUnsignedInt() const {
100 return strtoul10(value.c_str());
101 }
102
103 // -----------------------------------------
104 int ValueAsSignedInt() const {
105 return strtol10(value.c_str());
106 }
107
108 // -----------------------------------------
109 float ValueAsFloat() const {
110 return fast_atof(value.c_str());
111 }
112
113 // -----------------------------------------
115 LineReader& operator++() {
116 if (end) {
117 if (end == 1) {
118 ++end;
119 }
120 return *this;
121 }
122
123 try {
124 groupcode = strtol10(splitter->c_str());
125 splitter++;
126
127 value = *splitter;
128 splitter++;
129
130 // automatically skip over {} meta blocks (these are for application use
131 // and currently not relevant for Assimp).
132 if (value.length() && value[0] == '{') {
133
134 size_t cnt = 0;
135 for(;splitter->length() && splitter->at(0) != '}'; splitter++, cnt++);
136
137 splitter++;
138 ASSIMP_LOG_DEBUG((Formatter::format("DXF: skipped over control group ("),cnt," lines)"));
139 }
140 } catch(std::logic_error&) {
141 ai_assert(!splitter);
142 }
143 if (!splitter) {
144 end = 1;
145 }
146 return *this;
147 }
148
149 // -----------------------------------------
150 LineReader& operator++(int) {
151 return ++(*this);
152 }
153
154
155 // -----------------------------------------
156 operator bool() const {
157 return end <= 1;
158 }
159
160private:
161 LineSplitter splitter;
162 int groupcode;
163 std::string value;
164 int end;
165};
