60 FT_CALLBACK_DEF( FT_Error )
61 tt_property_set( FT_Module module, /* TT_Driver */
62 const char* property_name,
63 const void* value,
64 FT_Bool value_is_string )
65 {
66 FT_Error error = FT_Err_Ok;
67 TT_Driver driver = (TT_Driver)module;
68
69#ifndef FT_CONFIG_OPTION_ENVIRONMENT_PROPERTIES
70 FT_UNUSED( value_is_string );
71#endif
72
73
74 if ( !ft_strcmp( property_name, "interpreter-version" ) )
75 {
76 FT_UInt interpreter_version;
77
78
79#ifdef FT_CONFIG_OPTION_ENVIRONMENT_PROPERTIES
80 if ( value_is_string )
81 {
82 const char* s = (const char*)value;
83
84
85 interpreter_version = (FT_UInt)ft_strtol( s, NULL, 10 );
86 }
87 else
88#endif
89 {
90 FT_UInt* iv = (FT_UInt*)value;
91
92
93 interpreter_version = *iv;
94 }
95
96 switch ( interpreter_version )
97 {
98 case TT_INTERPRETER_VERSION_35:
99 driver->interpreter_version = TT_INTERPRETER_VERSION_35;
100 break;
101
102 case TT_INTERPRETER_VERSION_38:
103#ifdef TT_SUPPORT_SUBPIXEL_HINTING_INFINALITY
104 driver->interpreter_version = TT_INTERPRETER_VERSION_38;
105 break;
106#endif
107
108 case TT_INTERPRETER_VERSION_40:
109#ifdef TT_SUPPORT_SUBPIXEL_HINTING_MINIMAL
110 driver->interpreter_version = TT_INTERPRETER_VERSION_40;
111 break;
112#endif
113
114 default:
115 error = FT_ERR( Unimplemented_Feature );
116 }
117
118 return error;
119 }
120
121 FT_TRACE2(( "tt_property_set: missing property `%s'\n",
122 property_name ));
123 return FT_THROW( Missing_Property );
124 }

175 /**************************************************************************
176 *
177 * @Function:
178 * tt_get_kerning
179 *
180 * @Description:
181 * A driver method used to return the kerning vector between two
182 * glyphs of the same face.
183 *
184 * @Input:
185 * face ::
186 * A handle to the source face object.
187 *
188 * left_glyph ::
189 * The index of the left glyph in the kern pair.
190 *
191 * right_glyph ::
192 * The index of the right glyph in the kern pair.
193 *
194 * @Output:
195 * kerning ::
196 * The kerning vector. This is in font units for
197 * scalable formats, and in pixels for fixed-sizes
198 * formats.
199 *
200 * @Return:
201 * FreeType error code. 0 means success.
202 *
203 * @Note:
204 * Only horizontal layouts (left-to-right & right-to-left) are
205 * supported by this function. Other layouts, or more sophisticated
206 * kernings, are out of scope of this method (the basic driver
207 * interface is meant to be simple).
208 *
209 * They can be implemented by format-specific interfaces.
210 */
211 FT_CALLBACK_DEF( FT_Error )
212 tt_get_kerning( FT_Face face, /* TT_Face */
213 FT_UInt left_glyph,
214 FT_UInt right_glyph,
215 FT_Vector* kerning )
216 {
217 TT_Face ttface = (TT_Face)face;
218 SFNT_Service sfnt = (SFNT_Service)ttface->sfnt;
219
220
221 kerning->x = 0;
222 kerning->y = 0;
223
224 if ( sfnt )
225 kerning->x = sfnt->get_kerning( ttface, left_glyph, right_glyph );
226
227 return 0;
228 }

231 FT_CALLBACK_DEF( FT_Error )
232 tt_get_advances( FT_Face face, /* TT_Face */
233 FT_UInt start,
234 FT_UInt count,
235 FT_Int32 flags,
236 FT_Fixed *advances )
237 {
238 FT_UInt nn;
239 TT_Face ttface = (TT_Face)face;
240
241
242 /* XXX: TODO: check for sbits */
243
244 if ( flags & FT_LOAD_VERTICAL_LAYOUT )
245 {
246#ifdef TT_CONFIG_OPTION_GX_VAR_SUPPORT
247 /* no fast retrieval for blended MM fonts without VVAR table */
248 if ( ( FT_IS_NAMED_INSTANCE( face ) || FT_IS_VARIATION( face ) ) &&
249 !( ttface->variation_support & TT_FACE_FLAG_VAR_VADVANCE ) )
250 return FT_THROW( Unimplemented_Feature );
251#endif
252
253 for ( nn = 0; nn < count; nn++ )
254 {
255 FT_Short tsb;
256 FT_UShort ah;
257
258
259 /* since we don't need `tsb', we use zero for `yMax' parameter */
260 TT_Get_VMetrics( ttface, start + nn, 0, &tsb, &ah );
261 advances[nn] = ah;
262 }
263 }
264 else
265 {
266#ifdef TT_CONFIG_OPTION_GX_VAR_SUPPORT
267 /* no fast retrieval for blended MM fonts without HVAR table */
268 if ( ( FT_IS_NAMED_INSTANCE( face ) || FT_IS_VARIATION( face ) ) &&
269 !( ttface->variation_support & TT_FACE_FLAG_VAR_HADVANCE ) )
270 return FT_THROW( Unimplemented_Feature );
271#endif
272
273 for ( nn = 0; nn < count; nn++ )
274 {
275 FT_Short lsb;
276 FT_UShort aw;
277
278
279 TT_Get_HMetrics( ttface, start + nn, &lsb, &aw );
280 advances[nn] = aw;
281 }
282 }
283
284 return FT_Err_Ok;
285 }

303 FT_CALLBACK_DEF( FT_Error )
304 tt_size_select( FT_Size size,
305 FT_ULong strike_index )
306 {
307 TT_Face ttface = (TT_Face)size->face;
308 TT_Size ttsize = (TT_Size)size;
309 FT_Error error = FT_Err_Ok;
310
311
312 ttsize->strike_index = strike_index;
313
314 if ( FT_IS_SCALABLE( size->face ) )
315 {
316 /* use the scaled metrics, even when tt_size_reset fails */
317 FT_Select_Metrics( size->face, strike_index );
318
319 tt_size_reset( ttsize ); /* ignore return value */
320 }
321 else
322 {
323 SFNT_Service sfnt = (SFNT_Service)ttface->sfnt;
324 FT_Size_Metrics* size_metrics = &size->metrics;
325
326
327 error = sfnt->load_strike_metrics( ttface,
328 strike_index,
329 size_metrics );
330 if ( error )
331 ttsize->strike_index = 0xFFFFFFFFUL;
332 }
333
334 return error;
335 }

337#endif /* TT_CONFIG_OPTION_EMBEDDED_BITMAPS */

340 FT_CALLBACK_DEF( FT_Error )
341 tt_size_request( FT_Size size,
342 FT_Size_Request req )
343 {
344 TT_Size ttsize = (TT_Size)size;
345 FT_Error error = FT_Err_Ok;
346
347
348#ifdef TT_CONFIG_OPTION_EMBEDDED_BITMAPS
349
350 if ( FT_HAS_FIXED_SIZES( size->face ) )
351 {
352 TT_Face ttface = (TT_Face)size->face;
353 SFNT_Service sfnt = (SFNT_Service)ttface->sfnt;
354 FT_ULong strike_index;
355
356
357 error = sfnt->set_sbit_strike( ttface, req, &strike_index );
358
359 if ( error )
360 ttsize->strike_index = 0xFFFFFFFFUL;
361 else
362 return tt_size_select( size, strike_index );
363 }
364
365#endif /* TT_CONFIG_OPTION_EMBEDDED_BITMAPS */
366
367 {
368 FT_Error err = FT_Request_Metrics( size->face, req );
369
370
371 if ( err )
372 {
373 error = err;
374 goto Exit;
375 }
376 }
377
378 if ( FT_IS_SCALABLE( size->face ) )
379 {
380 error = tt_size_reset( ttsize );
381
382#ifdef TT_USE_BYTECODE_INTERPRETER
383 /* for the `MPS' bytecode instruction we need the point size */
384 if ( !error )
385 {
386 FT_UInt resolution =
387 ttsize->metrics->x_ppem > ttsize->metrics->y_ppem
388 ? req->horiResolution
389 : req->vertResolution;
390
391
392 /* if we don't have a resolution value, assume 72dpi */
393 if ( req->type == FT_SIZE_REQUEST_TYPE_SCALES ||
394 !resolution )
395 resolution = 72;
396
397 ttsize->point_size = FT_MulDiv( ttsize->ttmetrics.ppem,
398 64 * 72,
399 resolution );
400 }
401#endif
402 }
403
404 Exit:
405 return error;
406 }

409 /**************************************************************************
410 *
411 * @Function:
412 * tt_glyph_load
413 *
414 * @Description:
415 * A driver method used to load a glyph within a given glyph slot.
416 *
417 * @Input:
418 * slot ::
419 * A handle to the target slot object where the glyph
420 * will be loaded.
421 *
422 * size ::
423 * A handle to the source face size at which the glyph
424 * must be scaled, loaded, etc.
425 *
426 * glyph_index ::
427 * The index of the glyph in the font file.
428 *
429 * load_flags ::
430 * A flag indicating what to load for this glyph. The
431 * FT_LOAD_XXX constants can be used to control the
432 * glyph loading process (e.g., whether the outline
433 * should be scaled, whether to load bitmaps or not,
434 * whether to hint the outline, etc).
435 *
436 * @Return:
437 * FreeType error code. 0 means success.
438 */
439 FT_CALLBACK_DEF( FT_Error )
440 tt_glyph_load( FT_GlyphSlot slot, /* TT_GlyphSlot */
441 FT_Size size, /* TT_Size */
442 FT_UInt glyph_index,
443 FT_Int32 load_flags )
444 {
445 TT_GlyphSlot ttslot = (TT_GlyphSlot)slot;
446 TT_Size ttsize = (TT_Size)size;
447 FT_Face face = ttslot->face;
448 FT_Error error;
449
450
451 if ( !slot )
452 return FT_THROW( Invalid_Slot_Handle );
453
454 if ( !size )
455 return FT_THROW( Invalid_Size_Handle );
456
457 if ( !face )
458 return FT_THROW( Invalid_Face_Handle );
459
460#ifdef FT_CONFIG_OPTION_INCREMENTAL
461 if ( glyph_index >= (FT_UInt)face->num_glyphs &&
462 !face->internal->incremental_interface )
463#else
464 if ( glyph_index >= (FT_UInt)face->num_glyphs )
465#endif
466 return FT_THROW( Invalid_Argument );
467
468 if ( load_flags & FT_LOAD_NO_HINTING )
469 {
470 /* both FT_LOAD_NO_HINTING and FT_LOAD_NO_AUTOHINT */
471 /* are necessary to disable hinting for tricky fonts */
472
473 if ( FT_IS_TRICKY( face ) )
474 load_flags &= ~FT_LOAD_NO_HINTING;
475
476 if ( load_flags & FT_LOAD_NO_AUTOHINT )
477 load_flags |= FT_LOAD_NO_HINTING;
478 }
479
480 if ( load_flags & ( FT_LOAD_NO_RECURSE | FT_LOAD_NO_SCALE ) )
481 {
482 load_flags |= FT_LOAD_NO_BITMAP | FT_LOAD_NO_SCALE;
483
484 if ( !FT_IS_TRICKY( face ) )
485 load_flags |= FT_LOAD_NO_HINTING;
486 }
487
488 /* use hinted metrics only if we load a glyph with hinting */
489 ttsize->metrics = ( load_flags & FT_LOAD_NO_HINTING )
490 ? &size->metrics
491 : &ttsize->hinted_metrics;
492
493 /* now fill in the glyph slot with outline/bitmap/layered */
494 error = TT_Load_Glyph( ttsize, ttslot, glyph_index, load_flags );
495
496 /* force drop-out mode to 2 - irrelevant now */
497 /* slot->outline.dropout_mode = 2; */
498
499 return error;
500 }

608 FT_CALLBACK_DEF( FT_Module_Interface )
609 tt_get_interface( FT_Module driver, /* TT_Driver */
610 const char* tt_interface )
611 {
612 FT_Library library;
613 FT_Module_Interface result;
614 FT_Module sfntd;
615 SFNT_Service sfnt;
616
617
618 result = ft_service_list_lookup( tt_services, tt_interface );
619 if ( result )
620 return result;
621
622 if ( !driver )
623 return NULL;
624 library = driver->library;
625 if ( !library )
626 return NULL;
627
628 /* only return the default interface from the SFNT module */
629 sfntd = FT_Get_Module( library, "sfnt" );
630 if ( sfntd )
631 {
632 sfnt = (SFNT_Service)( sfntd->clazz->module_interface );
633 if ( sfnt )
634 return sfnt->get_interface( driver, tt_interface );
635 }
636
637 return 0;
638 }
