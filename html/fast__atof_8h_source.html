57// ------------------------------------------------------------------------------------
58// Convert a string in decimal format to a number
59// ------------------------------------------------------------------------------------
60inline
61unsigned int strtoul10( const char* in, const char** out=0) {
62 unsigned int value = 0;
63
64 for ( ;; ) {
65 if ( *in < '0' || *in > '9' ) {
66 break;
67 }
68
69 value = ( value * 10 ) + ( *in - '0' );
70 ++in;
71 }
72 if ( out ) {
73 *out = in;
74 }
75 return value;
76}

98// ------------------------------------------------------------------------------------
99// Convert a string in hex format to a number
100// ------------------------------------------------------------------------------------
101inline
102unsigned int strtoul16( const char* in, const char** out=0) {
103 unsigned int value( 0 );
104 for ( ;; ) {
105 if ( *in >= '0' && *in <= '9' ) {
106 value = ( value << 4u ) + ( *in - '0' );
107 } else if (*in >= 'A' && *in <= 'F') {
108 value = ( value << 4u ) + ( *in - 'A' ) + 10;
109 } else if (*in >= 'a' && *in <= 'f') {
110 value = ( value << 4u ) + ( *in - 'a' ) + 10;
111 } else {
112 break;
113 }
114 ++in;
115 }
116 if ( out ) {
117 *out = in;
118 }
119 return value;
120}

122// ------------------------------------------------------------------------------------
123// Convert just one hex digit
124// Return value is UINT_MAX if the input character is not a hex digit.
125// ------------------------------------------------------------------------------------
126inline
127unsigned int HexDigitToDecimal(char in) {
128 unsigned int out( UINT_MAX );
129 if ( in >= '0' && in <= '9' ) {
130 out = in - '0';
131 } else if ( in >= 'a' && in <= 'f' ) {
132 out = 10u + in - 'a';
133 } else if ( in >= 'A' && in <= 'F' ) {
134 out = 10u + in - 'A';
135 }
136
137 // return value is UINT_MAX if the input is not a hex digit
138 return out;
139}

141// ------------------------------------------------------------------------------------
142// Convert a hex-encoded octet (2 characters, i.e. df or 1a).

144inline
145uint8_t HexOctetToDecimal(const char* in) {

166// ------------------------------------------------------------------------------------
167// Parse a C++-like integer literal - hex and oct prefixes.
168// 0xNNNN - hex
169// 0NNN - oct
170// NNN - dec
171// ------------------------------------------------------------------------------------
172inline
173unsigned int strtoul_cppstyle( const char* in, const char** out=0) {
174 if ('0' == in[0]) {
175 return 'x' == in[1] ? strtoul16(in+2,out) : strtoul8(in+1,out);
176 }
177 return strtoul10(in, out);

180// ------------------------------------------------------------------------------------
181// Special version of the function, providing higher accuracy and safety
182// It is mainly used by fast_atof to prevent ugly and unwanted integer overflows.
183// ------------------------------------------------------------------------------------
184inline
185uint64_t strtoul10_64( const char* in, const char** out=0, unsigned int* max_inout=0) {
186 unsigned int cur = 0;
187 uint64_t value = 0;
188
189 if ( *in < '0' || *in > '9' ) {
190 throw std::invalid_argument( std::string( "The string \"" ) + in + "\" cannot be converted into a value." );
191 }
192
193 for ( ;; ) {
194 if ( *in < '0' || *in > '9' ) {
195 break;
196 }
197
198 const uint64_t new_value = ( value * (uint64_t) 10 ) + ( (uint64_t) ( *in - '0' ) );
199
200 // numeric overflow, we rely on you
201 if ( new_value < value ) {
202 ASSIMP_LOG_WARN_F( "Converting the string \"", in, "\" into a value resulted in overflow." );
203 return 0;
204 }
205
206 value = new_value;
207
208 ++in;
209 ++cur;
210
211 if (max_inout && *max_inout == cur) {
212 if (out) { /* skip to end */
213 while ( *in >= '0' && *in <= '9' ) {
214 ++in;
215 }
216 *out = in;
217 }
218
219 return value;
220 }
221 }
222 if ( out ) {
223 *out = in;
224 }
225
226 if ( max_inout ) {
227 *max_inout = cur;
228 }
229
230 return value;
231}

258template<typename Real>
259inline
260const char* fast_atoreal_move(const char* c, Real& out, bool check_comma = true) {
261 Real f = 0;
262
263 bool inv = (*c == '-');
264 if (inv || *c == '+') {
265 ++c;
266 }
267
268 if ((c[0] == 'N' || c[0] == 'n') && ASSIMP_strincmp(c, "nan", 3) == 0) {
269 out = std::numeric_limits<Real>::quiet_NaN();
270 c += 3;
271 return c;
272 }
273
274 if ((c[0] == 'I' || c[0] == 'i') && ASSIMP_strincmp(c, "inf", 3) == 0) {
275 out = std::numeric_limits<Real>::infinity();
276 if (inv) {
277 out = -out;
278 }
279 c += 3;
280 if ((c[0] == 'I' || c[0] == 'i') && ASSIMP_strincmp(c, "inity", 5) == 0) {
281 c += 5;
282 }
283 return c;
284 }
285
286 if (!(c[0] >= '0' && c[0] <= '9') &&
287 !((c[0] == '.' || (check_comma && c[0] == ',')) && c[1] >= '0' && c[1] <= '9')) {
288 throw std::invalid_argument("Cannot parse string "
289 "as real number: does not start with digit "
290 "or decimal point followed by digit.");
291 }
292
293 if (*c != '.' && (! check_comma || c[0] != ',')) {
294 f = static_cast<Real>( strtoul10_64 ( c, &c) );
295 }
296
297 if ((*c == '.' || (check_comma && c[0] == ',')) && c[1] >= '0' && c[1] <= '9') {
298 ++c;
299
300 // NOTE: The original implementation is highly inaccurate here. The precision of a single
301 // IEEE 754 float is not high enough, everything behind the 6th digit tends to be more
302 // inaccurate than it would need to be. Casting to double seems to solve the problem.
303 // strtol_64 is used to prevent integer overflow.
304
305 // Another fix: this tends to become 0 for long numbers if we don't limit the maximum
306 // number of digits to be read. AI_FAST_ATOF_RELAVANT_DECIMALS can be a value between
307 // 1 and 15.
308 unsigned int diff = AI_FAST_ATOF_RELAVANT_DECIMALS;
309 double pl = static_cast<double>( strtoul10_64 ( c, &c, &diff ));
310
311 pl *= fast_atof_table[diff];
312 f += static_cast<Real>( pl );
313 }
314 // For backwards compatibility: eat trailing dots, but not trailing commas.
315 else if (*c == '.') {
316 ++c;
317 }
318
319 // A major 'E' must be allowed. Necessary for proper reading of some DXF files.
320 // Thanks to Zhao Lei to point out that this if() must be outside the if (*c == '.' ..)
321 if (*c == 'e' || *c == 'E') {
322 ++c;
323 const bool einv = (*c=='-');
324 if (einv || *c=='+') {
325 ++c;
326 }
327
328 // The reason float constants are used here is that we've seen cases where compilers
329 // would perform such casts on compile-time constants at runtime, which would be
330 // bad considering how frequently fast_atoreal_move<float> is called in Assimp.
331 Real exp = static_cast<Real>( strtoul10_64(c, &c) );
332 if (einv) {
333 exp = -exp;
334 }
335 f *= std::pow(static_cast<Real>(10.0), exp);
336 }
337
338 if (inv) {
339 f = -f;
340 }
341 out = f;
342 return c;
343}
