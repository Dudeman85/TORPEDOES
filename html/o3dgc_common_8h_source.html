217 inline unsigned long Insert(SC3DMCTriplet e, unsigned long & nPred, SC3DMCPredictor * const list)
218 {
219 unsigned long pos = 0xFFFFFFFF;
220 bool foundOrInserted = false;
221 for (unsigned long j = 0; j < nPred; ++j)
222 {
223 if (e == list[j].m_id)
224 {
225 foundOrInserted = true;
226 break;
227 }
228 else if (e < list[j].m_id)
229 {
230 if (nPred < O3DGC_SC3DMC_MAX_PREDICTION_NEIGHBORS)
231 {
232 ++nPred;
233 }
234 for (unsigned long h = nPred-1; h > j; --h)
235 {
236 list[h] = list[h-1];
237 }
238 list[j].m_id = e;
239 pos = j;
240 foundOrInserted = true;
241 break;
242 }
243 }
244 if (!foundOrInserted && nPred < O3DGC_SC3DMC_MAX_PREDICTION_NEIGHBORS)
245 {
246 pos = nPred;
247 list[nPred++].m_id = e;
248 }
249 return pos;
250 }

252 inline void SphereToCube(const T x, const T y, const T z,
253 T & a, T & b, char & index)
254 {
255 T ax = absolute(x);
256 T ay = absolute(y);
257 T az = absolute(z);
258 if (az >= ax && az >= ay)
259 {
260 if (z >= (T)(0))
261 {
262 index = 0;
263 a = x;
264 b = y;
265 }
266 else
267 {
268 index = 1;
269 a = -x;
270 b = -y;
271 }
272 }
273 else if (ay >= ax && ay >= az)
274 {
275 if (y >= (T)(0))
276 {
277 index = 2;
278 a = z;
279 b = x;
280 }
281 else
282 {
283 index = 3;
284 a = -z;
285 b = -x;
286 }
287 }
288 else if (ax >= ay && ax >= az)
289 {
290 if (x >= (T)(0))
291 {
292 index = 4;
293 a = y;
294 b = z;
295 }
296 else
297 {
298 index = 5;
299 a = -y;
300 b = -z;
301 }

304 inline void CubeToSphere(const Real a, const Real b, const char index,
305 Real & x, Real & y, Real & z)
306 {
307 switch( index )
308 {
309 case 0:
310 x = a;
311 y = b;
312 z = (Real) sqrt(max(0.0, 1.0 - x*x-y*y));
313 break;
314 case 1:
315 x = -a;
316 y = -b;
317 z = -(Real) sqrt(max(0.0, 1.0 - x*x-y*y));
318 break;
319 case 2:
320 z = a;
321 x = b;
322 y = (Real) sqrt(max(0.0, 1.0 - x*x-z*z));
323 break;
324 case 3:
325 z = -a;
326 x = -b;
327 y = -(Real) sqrt(max(0.0, 1.0 - x*x-z*z));
328 break;
329 case 4:
330 y = a;
331 z = b;
332 x = (Real) sqrt(max(0.0, 1.0 - y*y-z*z));
333 break;
334 case 5:
335 y = -a;
336 z = -b;
337 x = -(Real) sqrt(max(0.0, 1.0 - y*y-z*z));
338 break;
339 }

349 inline void ComputeVectorMinMax(const Real * const tab,
350 unsigned long size,
351 unsigned long dim,
352 unsigned long stride,
353 Real * minTab,
354 Real * maxTab,
355 O3DGCSC3DMCQuantizationMode quantMode)
356 {
357 if (size == 0 || dim == 0)
358 {
359 return;
360 }
361 unsigned long p = 0;
362 for(unsigned long d = 0; d < dim; ++d)
363 {
364 maxTab[d] = minTab[d] = tab[p++];
365 }
366 p = stride;
367 for(unsigned long i = 1; i < size; ++i)
368 {
369 for(unsigned long d = 0; d < dim; ++d)
370 {
371 if (maxTab[d] < tab[p+d]) maxTab[d] = tab[p+d];
372 if (minTab[d] > tab[p+d]) minTab[d] = tab[p+d];
373 }
374 p += stride;
375 }
376
377 if (quantMode == O3DGC_SC3DMC_DIAG_BB)
378 {
379 Real diag = Real( 0.0 );
380 Real r;
381 for(unsigned long d = 0; d < dim; ++d)
382 {
383 r = (maxTab[d] - minTab[d]);
384 diag += r*r;
385 }
386 diag = static_cast<Real>(sqrt(diag));
387 for(unsigned long d = 0; d < dim; ++d)
388 {
389 maxTab[d] = minTab[d] + diag;
390 }
391 }
392 else if (quantMode == O3DGC_SC3DMC_MAX_ALL_DIMS)
393 {
394 Real maxr = (maxTab[0] - minTab[0]);
395 Real r;
396 for(unsigned long d = 1; d < dim; ++d)
397 {
398 r = (maxTab[d] - minTab[d]);
399 if ( r > maxr)
400 {
401 maxr = r;
402 }
403 }
404 for(unsigned long d = 0; d < dim; ++d)
405 {
406 maxTab[d] = minTab[d] + maxr;
407 }
408 }
409 }
