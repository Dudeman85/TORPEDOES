95 static FT_Long
96 FT_MulDiv( FT_Long a,
97 FT_Long b,
98 FT_Long c )
99 {
100 FT_Int s;
101 FT_Long d;
102
103
104 s = 1;
105 if ( a < 0 ) { a = -a; s = -1; }
106 if ( b < 0 ) { b = -b; s = -s; }
107 if ( c < 0 ) { c = -c; s = -s; }
108
109 d = (FT_Long)( c > 0 ? ( (FT_Int64)a * b + ( c >> 1 ) ) / c
110 : 0x7FFFFFFFL );
111
112 return ( s > 0 ) ? d : -d;
113 }
