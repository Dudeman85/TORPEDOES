255// Convert a quaternion to its usual representation
256inline void ConvertQuaternion (const aiVector3D& in, aiQuaternion& out) {
257
258 out.x = in.x;
259 out.y = in.y;
260 out.z = in.z;
261
262 const float t = 1.0f - (in.x*in.x) - (in.y*in.y) - (in.z*in.z);
263
264 if (t < 0.0f)
265 out.w = 0.0f;
266 else out.w = std::sqrt (t);
267
268 // Assimp convention.
269 out.w *= -1.f;
270}
