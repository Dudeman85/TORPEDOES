76AI_FORCE_INLINE
77unsigned int ASSIMP_itoa10( char* out, unsigned int max, int32_t number) {
78 ai_assert(NULL != out);
79
80 // write the unary minus to indicate we have a negative number
81 unsigned int written = 1u;
82 if (number < 0 && written < max) {
83 *out++ = '-';
84 ++written;
85 number = -number;
86 }
87
88 // We begin with the largest number that is not zero.
89 int32_t cur = 1000000000; // 2147483648
90 bool mustPrint = false;
91 while (written < max) {
92
93 const unsigned int digit = number / cur;
94 if (mustPrint || digit > 0 || 1 == cur) {
95 // print all future zeroe's from now
96 mustPrint = true;
97
98 *out++ = '0'+static_cast<char>(digit);
99
100 ++written;
101 number -= digit*cur;
102 if (1 == cur) {
103 break;
104 }
105 }
106 cur /= 10;
107 }
108
109 // append a terminal zero
110 *out++ = '\0';
111 return written-1;
112}

136AI_FORCE_INLINE
137int ASSIMP_stricmp(const char *s1, const char *s2) {
138 ai_assert( NULL != s1 );
139 ai_assert( NULL != s2 );
140
141#if (defined _MSC_VER)
142
143 return ::_stricmp(s1,s2);
144#elif defined( __GNUC__ )
145
146 return ::strcasecmp(s1,s2);
147#else
148
149 char c1, c2;
150 do {
151 c1 = tolower(*s1++);
152 c2 = tolower(*s2++);
153 }
154 while ( c1 && (c1 == c2) );
155 return c1 - c2;
156#endif
157}
