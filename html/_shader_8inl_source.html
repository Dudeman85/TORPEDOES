70static const std::string fragmentShader = R"(
71 #version 130
72 #define FLIP_HORIZONTAL 8u
73 #define FLIP_VERTICAL 4u
74 #define FLIP_DIAGONAL 2u
75
76 in vec2 v_texCoord;
77
78 uniform usampler2D u_lookupMap;
79 uniform sampler2D u_tileMap;
80
81 uniform vec2 u_tileSize = vec2(64.0);
82 uniform vec2 u_tilesetCount = vec2(6.0, 7.0);
83 uniform vec2 u_tilesetScale = vec2(1.0);
84
85 uniform float u_opacity = 1.0;
86
87 out vec4 colour;
88 /*fixes rounding imprecision on AMD cards*/
89 const float epsilon = 0.000005;
90
91 void main()
92 {
93 uvec2 values = texture(u_lookupMap, v_texCoord).rg;
94 if(values.r > 0u)
95 {
96 float index = float(values.r) - 1.0;
97 vec2 position = vec2(mod(index + epsilon, u_tilesetCount.x), floor((index / u_tilesetCount.x) + epsilon)) / u_tilesetCount;
98 vec2 offsetCoord = (v_texCoord * (textureSize(u_lookupMap, 0) * u_tilesetScale)) / u_tileSize;
99
100 vec2 texelSize = vec2(1.0) / textureSize(u_lookupMap, 0);
101 vec2 offset = mod(v_texCoord, texelSize);
102 vec2 ratio = offset / texelSize;
103 offset = ratio * (1.0 / u_tileSize);
104 offset *= u_tileSize / u_tilesetCount;
105
106 if(values.g != 0u)
107 {
108 vec2 tileSize = vec2(1.0) / u_tilesetCount;
109 if((values.g & FLIP_DIAGONAL) != 0u)
110 {
111 float temp = offset.x;
112 offset.x = offset.y;
113 offset.y = temp;
114 temp = tileSize.x / tileSize.y;
115 offset.x *= temp;
116 offset.y /= temp;
117 offset.x = tileSize.x - offset.x;
118 offset.y = tileSize.y - offset.y;
119 }
120 if((values.g & FLIP_VERTICAL) != 0u)
121 {
122 offset.y = tileSize.y - offset.y;
123 }
124 if((values.g & FLIP_HORIZONTAL) != 0u)
125 {
126 offset.x = tileSize.x - offset.x;
127 }
128 }
129 colour = texture(u_tileMap, position + offset);
130 colour.a = min(colour.a, u_opacity);
131 }
132 else
133 {
134 colour = vec4(0.0);
135 }
136 })";
