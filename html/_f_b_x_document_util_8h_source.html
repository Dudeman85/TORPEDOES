76template <typename T>
77inline
78const T* ProcessSimpleConnection(const Connection& con,
79 bool is_object_property_conn,
80 const char* name,
81 const Element& element,
82 const char** propNameOut = nullptr)
83{
84 if (is_object_property_conn && !con.PropertyName().length()) {
85 DOMWarning("expected incoming " + std::string(name) +
86 " link to be an object-object connection, ignoring",
87 &element
88 );
89 return nullptr;
90 }
91 else if (!is_object_property_conn && con.PropertyName().length()) {
92 DOMWarning("expected incoming " + std::string(name) +
93 " link to be an object-property connection, ignoring",
94 &element
95 );
96 return nullptr;
97 }
98
99 if(is_object_property_conn && propNameOut) {
100 // note: this is ok, the return value of PropertyValue() is guaranteed to
101 // remain valid and unchanged as long as the document exists.
102 *propNameOut = con.PropertyName().c_str();
103 }
104
105 const Object* const ob = con.SourceObject();
106 if(!ob) {
107 DOMWarning("failed to read source object for incoming " + std::string(name) +
108 " link, ignoring",
109 &element);
110 return nullptr;
111 }
112
113 return dynamic_cast<const T*>(ob);
114}
