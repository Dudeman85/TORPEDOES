95template<class T>
96inline
97bool isFloat( T *in, T *end ) {
98 if( in != end ) {
99 if( *in == '-' ) {
100 ++in;
101 }
102 }
103
104 // check for <1>.0f
105 bool result( false );
106 while( isNotEndOfToken( in, end ) ) {
107 if( *in == '.' ) {
108 result = true;
109 break;
110 }
111 result = isNumeric( *in );
112 if( !result ) {
113 return false;
114 }
115 ++in;
116 }
117
118 // check for 1<.>0f
119 if( *in == '.' ) {
120 ++in;
121 } else {
122 return false;
123 }
124
125 // check for 1.<0>f
126 while( isNotEndOfToken( in, end ) ) {
127 result = isNumeric( *in );

132 }
