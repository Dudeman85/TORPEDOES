92AI_FORCE_INLINE bool is_qnan(float in)
93{
94 // the straightforward solution does not work:
95 // return (in != in);
96 // compiler generates code like this
97 // load <in> to <register-with-different-width>
98 // compare <register-with-different-width> against <in>
99
100 // FIXME: Use <float> stuff instead? I think fpclassify needs C99
101 _IEEESingle temp;
102 memcpy(&temp, &in, sizeof(float));
103 return (temp.IEEE.Exp == (1u << 8)-1 &&
104 temp.IEEE.Frac);
105}
