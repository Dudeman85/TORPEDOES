161enum MeshChunkId

163 M_HEADER = 0x1000,
164 // char* version : Version number check
165 M_MESH = 0x3000,
166 // bool skeletallyAnimated // important flag which affects h/w buffer policies
167 // Optional M_GEOMETRY chunk
168 M_SUBMESH = 0x4000,
169 // char* materialName
170 // bool useSharedVertices
171 // unsigned int indexCount
172 // bool indexes32Bit

174 // OR
175 // unsigned short* faceVertexIndices (indexCount)
176 // M_GEOMETRY chunk (Optional: present only if useSharedVertices = false)
177 M_SUBMESH_OPERATION = 0x4010, // optional, trilist assumed if missing
178 // unsigned short operationType
179 M_SUBMESH_BONE_ASSIGNMENT = 0x4100,
180 // Optional bone weights (repeating section)
181 // unsigned int vertexIndex;
182 // unsigned short boneIndex;
183 // float weight;
184 // Optional chunk that matches a texture name to an alias
185 // a texture alias is sent to the submesh material to use this texture name

188 // char* aliasName;
189 // char* textureName;

191 M_GEOMETRY = 0x5000, // NB this chunk is embedded within M_MESH and M_SUBMESH
192 // unsigned int vertexCount
193 M_GEOMETRY_VERTEX_DECLARATION = 0x5100,
194 M_GEOMETRY_VERTEX_ELEMENT = 0x5110, // Repeating section

197 // unsigned short semantic; // VertexElementSemantic

199 // unsigned short index; // index of the semantic (for colours and texture coords)
200 M_GEOMETRY_VERTEX_BUFFER = 0x5200, // Repeating section
201 // unsigned short bindIndex; // Index to bind this buffer to

204 // raw buffer data

206 // Optional link to skeleton

208 M_MESH_BONE_ASSIGNMENT = 0x7000,
209 // Optional bone weights (repeating section)
210 // unsigned int vertexIndex;
211 // unsigned short boneIndex;
212 // float weight;
213 M_MESH_LOD = 0x8000,
214 // Optional LOD information
215 // string strategyName;
216 // unsigned short numLevels;
217 // bool manual; (true for manual alternate meshes, false for generated)

219 // Repeating section, ordered in increasing depth

221 // LOD value - this is a distance, a pixel count etc, based on strategy

223 M_MESH_LOD_MANUAL = 0x8110,

228 // Repeating section (1 per submesh)
229 // unsigned int indexCount;
230 // bool indexes32Bit
231 // unsigned short* faceIndexes; (indexCount)

233 // unsigned int* faceIndexes; (indexCount)

242 M_SUBMESH_NAME_TABLE = 0xA000,
243 // Subchunks of the name table. Each chunk contains an index & string

251 // unsigned short lodIndex
252 // bool isManual // If manual, no edge data here, loaded from manual mesh

254 // unsigned long numTriangles
255 // unsigned long numEdgeGroups
256 // Triangle* triangleList
257 // unsigned long indexSet
258 // unsigned long vertexSet
259 // unsigned long vertIndex[3]

261 // float normal[4]
262
263 M_EDGE_GROUP = 0xB110,
264 // unsigned long vertexSet
265 // unsigned long triStart
266 // unsigned long triCount
267 // unsigned long numEdges
268 // Edge* edgeList
269 // unsigned long triIndex[2]
270 // unsigned long vertIndex[2]
271 // unsigned long sharedVertIndex[2]
272 // bool degenerate
273 // Optional poses section, referred to by pose keyframes
274 M_POSES = 0xC000,
275 M_POSE = 0xC100,
276 // char* name (may be blank)
277 // unsigned short target // 0 for shared geometry,
278 // 1+ for submesh index + 1
279 // bool includesNormals [1.8+]
280 M_POSE_VERTEX = 0xC111,
281 // unsigned long vertexIndex
282 // float xoffset, yoffset, zoffset
283 // float xnormal, ynormal, znormal (optional, 1.8+)
284 // Optional vertex animation chunk
285 M_ANIMATIONS = 0xD000,
286 M_ANIMATION = 0xD100,
287 // char* name
288 // float length
289 M_ANIMATION_BASEINFO = 0xD105,
290 // [Optional] base keyframe information (pose animation only)

356enum SkeletonChunkId
357{
358 SKELETON_HEADER = 0x1000,
359 // char* version : Version number check
360 SKELETON_BLENDMODE = 0x1010, // optional
361 // unsigned short blendmode : SkeletonAnimationBlendMode
362 SKELETON_BONE = 0x2000,
363 // Repeating section defining each bone in the system.
364 // Bones are assigned indexes automatically based on their order of declaration
365 // starting with 0.
366 // char* name : name of the bone
367 // unsigned short handle : handle of the bone, should be contiguous & start at 0
368 // Vector3 position : position of this bone relative to parent
369 // Quaternion orientation : orientation of this bone relative to parent
370 // Vector3 scale : scale of this bone relative to parent
371 SKELETON_BONE_PARENT = 0x3000,
372 // Record of the parent of a single bone, used to build the node tree
373 // Repeating section, listed in Bone Index order, one per Bone
374 // unsigned short handle : child bone
375 // unsigned short parentHandle : parent bone
376 SKELETON_ANIMATION = 0x4000,
377 // A single animation for this skeleton
378 // char* name : Name of the animation
379 // float length : Length of the animation in seconds
380 SKELETON_ANIMATION_BASEINFO = 0x4010,
381 // [Optional] base keyframe information
382 // char* baseAnimationName (blank for self)
383 // float baseKeyFrameTime
384 SKELETON_ANIMATION_TRACK = 0x4100,
385 // A single animation track (relates to a single bone)
386 // Repeating section (within SKELETON_ANIMATION)
387 // unsigned short boneIndex : Index of bone to apply to
388 SKELETON_ANIMATION_TRACK_KEYFRAME = 0x4110,
389 // A single keyframe within the track
390 // Repeating section
391 // float time : The time position (seconds)
392 // Quaternion rotate : Rotation to apply at this keyframe
393 // Vector3 translate : Translation to apply at this keyframe
394 // Vector3 scale : Scale to apply at this keyframe
395 SKELETON_ANIMATION_LINK = 0x5000
396 // Link to another skeleton, to re-use its animations
397 // char* skeletonName : name of skeleton to get animations from
398 // float scale : scale to apply to trans/scale keys
399};
