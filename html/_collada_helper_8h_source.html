611struct Animation

623 ~Animation()
624 {
625 for( std::vector<Animation*>::iterator it = mSubAnims.begin(); it != mSubAnims.end(); ++it)
626 delete *it;
627 }
628
630 void CollectChannelsRecursively(std::vector<AnimationChannel> &channels)
631 {
632 channels.insert(channels.end(), mChannels.begin(), mChannels.end());
633
634 for (std::vector<Animation*>::iterator it = mSubAnims.begin(); it != mSubAnims.end(); ++it)
635 {
636 Animation *pAnim = (*it);
637
638 pAnim->CollectChannelsRecursively(channels);
639 }
640 }
641
643 void CombineSingleChannelAnimations()
644 {
645 CombineSingleChannelAnimationsRecursively(this);
646 }
647
648 void CombineSingleChannelAnimationsRecursively(Animation *pParent)
649 {
650 for (std::vector<Animation*>::iterator it = pParent->mSubAnims.begin(); it != pParent->mSubAnims.end();)
651 {
652 Animation *anim = *it;
653
654 CombineSingleChannelAnimationsRecursively(anim);
655
656 if (anim->mChannels.size() == 1)
657 {
658 pParent->mChannels.push_back(anim->mChannels[0]);
659
660 it = pParent->mSubAnims.erase(it);
661
662 delete anim;
663 }
664 else
665 {
666 ++it;
667 }
668 }
669 }
670};
