57 template<size_t N>
58 inline Value& MakeValue(Value& val, float(&r)[N], MemoryPoolAllocator<>& al) {
59 val.SetArray();
60 val.Reserve(N, al);
61 for (decltype(N) i = 0; i < N; ++i) {
62 val.PushBack(r[i], al);
63 }
64 return val;

111 inline void Write(Value& obj, Animation& a, AssetWriter& w)
112 {
113 /****************** Channels *******************/
114 Value channels;
115 channels.SetArray();
116 channels.Reserve(unsigned(a.channels.size()), w.mAl);
117
118 for (size_t i = 0; i < unsigned(a.channels.size()); ++i) {
119 Animation::Channel& c = a.channels[i];
120 Value valChannel;
121 valChannel.SetObject();
122 {
123 valChannel.AddMember("sampler", c.sampler, w.mAl);
124
125 Value valTarget;
126 valTarget.SetObject();
127 {
128 valTarget.AddMember("node", c.target.node->index, w.mAl);
129 switch (c.target.path) {
130 case AnimationPath_TRANSLATION:
131 valTarget.AddMember("path", "translation", w.mAl);
132 break;
133 case AnimationPath_ROTATION:
134 valTarget.AddMember("path", "rotation", w.mAl);
135 break;
136 case AnimationPath_SCALE:
137 valTarget.AddMember("path", "scale", w.mAl);
138 break;
139 case AnimationPath_WEIGHTS:
140 valTarget.AddMember("path", "weights", w.mAl);
141 break;
142 }
143 }
144 valChannel.AddMember("target", valTarget, w.mAl);
145 }
146 channels.PushBack(valChannel, w.mAl);
147 }
148 obj.AddMember("channels", channels, w.mAl);
149
150 /****************** Samplers *******************/
151 Value valSamplers;
152 valSamplers.SetArray();
153
154 for (size_t i = 0; i < unsigned(a.samplers.size()); ++i) {
155 Animation::Sampler& s = a.samplers[i];
156 Value valSampler;
157 valSampler.SetObject();
158 {
159 valSampler.AddMember("input", s.input->index, w.mAl);
160 switch (s.interpolation) {
161 case Interpolation_LINEAR:
162 valSampler.AddMember("path", "LINEAR", w.mAl);
163 break;
164 case Interpolation_STEP:
165 valSampler.AddMember("path", "STEP", w.mAl);
166 break;
167 case Interpolation_CUBICSPLINE:
168 valSampler.AddMember("path", "CUBICSPLINE", w.mAl);
169 break;
170 }
171 valSampler.AddMember("output", s.output->index, w.mAl);
172 }
173 valSamplers.PushBack(valSampler, w.mAl);
174 }
175 obj.AddMember("samplers", valSamplers, w.mAl);
176 }

309 {

311 pbrMetallicRoughness.SetObject();
312 {
313 WriteTex(pbrMetallicRoughness, m.pbrMetallicRoughness.baseColorTexture, "baseColorTexture", w.mAl);
314 WriteTex(pbrMetallicRoughness, m.pbrMetallicRoughness.metallicRoughnessTexture, "metallicRoughnessTexture", w.mAl);
315 WriteVec(pbrMetallicRoughness, m.pbrMetallicRoughness.baseColorFactor, "baseColorFactor", defaultBaseColor, w.mAl);
316
317 if (m.pbrMetallicRoughness.metallicFactor != 1) {
318 WriteFloat(pbrMetallicRoughness, m.pbrMetallicRoughness.metallicFactor, "metallicFactor", w.mAl);
319 }
320
321 if (m.pbrMetallicRoughness.roughnessFactor != 1) {
322 WriteFloat(pbrMetallicRoughness, m.pbrMetallicRoughness.roughnessFactor, "roughnessFactor", w.mAl);
323 }
324 }
325
326 if (!pbrMetallicRoughness.ObjectEmpty()) {
327 obj.AddMember("pbrMetallicRoughness", pbrMetallicRoughness, w.mAl);
328 }
329
330 WriteTex(obj, m.normalTexture, "normalTexture", w.mAl);
331 WriteTex(obj, m.emissiveTexture, "emissiveTexture", w.mAl);
332 WriteTex(obj, m.occlusionTexture, "occlusionTexture", w.mAl);
333 WriteVec(obj, m.emissiveFactor, "emissiveFactor", defaultEmissiveFactor, w.mAl);
334
335 if (m.alphaCutoff != 0.5) {
336 WriteFloat(obj, m.alphaCutoff, "alphaCutoff", w.mAl);
337 }
338
339 if (m.alphaMode != "OPAQUE") {
340 obj.AddMember("alphaMode", Value(m.alphaMode, w.mAl).Move(), w.mAl);

342
343 if (m.doubleSided) {
344 obj.AddMember("doubleSided", m.doubleSided, w.mAl);

348 exts.SetObject();
349

352 pbrSpecularGlossiness.SetObject();

356 //pbrSpecularGlossiness

358 WriteVec(pbrSpecularGlossiness, pbrSG.specularFactor, "specularFactor", defaultSpecularFactor, w.mAl);

362 }

369 }
370 }
371
372 if (m.unlit) {
373 Value unlit;

375 exts.AddMember("KHR_materials_unlit", unlit, w.mAl);
376 }
377
378 if (!exts.ObjectEmpty()) {
379 obj.AddMember("extensions", exts, w.mAl);
380 }

571 inline void AssetWriter::WriteFile(const char* path)
572 {
573 std::unique_ptr<IOStream> jsonOutFile(mAsset.OpenFile(path, "wt", true));
574
575 if (jsonOutFile == 0) {
576 throw DeadlyExportError("Could not open output file: " + std::string(path));
577 }
578
579 StringBuffer docBuffer;
580
581 PrettyWriter<StringBuffer> writer(docBuffer);
582 mDoc.Accept(writer);
583
584 if (jsonOutFile->Write(docBuffer.GetString(), docBuffer.GetSize(), 1) != 1) {
585 throw DeadlyExportError("Failed to write scene data!");
586 }
587
588 // Write buffer data to separate .bin files
589 for (unsigned int i = 0; i < mAsset.buffers.Size(); ++i) {
590 Ref<Buffer> b = mAsset.buffers.Get(i);
591
592 std::string binPath = b->GetURI();
593
594 std::unique_ptr<IOStream> binOutFile(mAsset.OpenFile(binPath, "wb", true));
595
596 if (binOutFile == 0) {
597 throw DeadlyExportError("Could not open output file: " + binPath);
598 }
599
600 if (b->byteLength > 0) {
601 if (binOutFile->Write(b->GetPointer(), b->byteLength, 1) != 1) {
602 throw DeadlyExportError("Failed to write binary file: " + binPath);
603 }
604 }
605 }
606 }

608 inline void AssetWriter::WriteGLBFile(const char* path)
609 {
610 std::unique_ptr<IOStream> outfile(mAsset.OpenFile(path, "wb", true));
611
612 if (outfile == 0) {
613 throw DeadlyExportError("Could not open output file: " + std::string(path));
614 }
615
616 Ref<Buffer> bodyBuffer = mAsset.GetBodyBuffer();
617 if (bodyBuffer->byteLength > 0) {
618 rapidjson::Value glbBodyBuffer;
619 glbBodyBuffer.SetObject();
620 glbBodyBuffer.AddMember("byteLength", static_cast<uint64_t>(bodyBuffer->byteLength), mAl);
621 mDoc["buffers"].PushBack(glbBodyBuffer, mAl);
622 }
623
624 // Padding with spaces as required by the spec
625 uint32_t padding = 0x20202020;
626
627 //
628 // JSON chunk
629 //
630
631 StringBuffer docBuffer;
632 Writer<StringBuffer> writer(docBuffer);
633 mDoc.Accept(writer);
634
635 uint32_t jsonChunkLength = (docBuffer.GetSize() + 3) & ~3; // Round up to next multiple of 4
636 auto paddingLength = jsonChunkLength - docBuffer.GetSize();
637
638 GLB_Chunk jsonChunk;
639 jsonChunk.chunkLength = jsonChunkLength;
640 jsonChunk.chunkType = ChunkType_JSON;
641 AI_SWAP4(jsonChunk.chunkLength);
642
643 outfile->Seek(sizeof(GLB_Header), aiOrigin_SET);
644 if (outfile->Write(&jsonChunk, 1, sizeof(GLB_Chunk)) != sizeof(GLB_Chunk)) {
645 throw DeadlyExportError("Failed to write scene data header!");
646 }
647 if (outfile->Write(docBuffer.GetString(), 1, docBuffer.GetSize()) != docBuffer.GetSize()) {
648 throw DeadlyExportError("Failed to write scene data!");
649 }
650 if (paddingLength && outfile->Write(&padding, 1, paddingLength) != paddingLength) {
651 throw DeadlyExportError("Failed to write scene data padding!");
652 }
653
654 //
655 // Binary chunk
656 //
657
658 uint32_t binaryChunkLength = 0;
659 if (bodyBuffer->byteLength > 0) {
660 binaryChunkLength = (bodyBuffer->byteLength + 3) & ~3; // Round up to next multiple of 4
661 auto paddingLength = binaryChunkLength - bodyBuffer->byteLength;
662
663 GLB_Chunk binaryChunk;
664 binaryChunk.chunkLength = binaryChunkLength;
665 binaryChunk.chunkType = ChunkType_BIN;
666 AI_SWAP4(binaryChunk.chunkLength);
667
668 size_t bodyOffset = sizeof(GLB_Header) + sizeof(GLB_Chunk) + jsonChunk.chunkLength;
669 outfile->Seek(bodyOffset, aiOrigin_SET);
670 if (outfile->Write(&binaryChunk, 1, sizeof(GLB_Chunk)) != sizeof(GLB_Chunk)) {
671 throw DeadlyExportError("Failed to write body data header!");
672 }
673 if (outfile->Write(bodyBuffer->GetPointer(), 1, bodyBuffer->byteLength) != bodyBuffer->byteLength) {
674 throw DeadlyExportError("Failed to write body data!");
675 }
676 if (paddingLength && outfile->Write(&padding, 1, paddingLength) != paddingLength) {
677 throw DeadlyExportError("Failed to write body data padding!");
678 }
679 }
680
681 //
682 // Header
683 //
684
685 GLB_Header header;
686 memcpy(header.magic, AI_GLB_MAGIC_NUMBER, sizeof(header.magic));
687
688 header.version = 2;
689 AI_SWAP4(header.version);
690
691 header.length = uint32_t(sizeof(GLB_Header) + 2 * sizeof(GLB_Chunk) + jsonChunkLength + binaryChunkLength);
692 AI_SWAP4(header.length);
693
694 outfile->Seek(0, aiOrigin_SET);

729 void AssetWriter::WriteObjects(LazyDict<T>& d)
730 {
731 if (d.mObjs.empty()) return;
732
733 Value* container = &mDoc;
734
735 if (d.mExtId) {
736 Value* exts = FindObject(mDoc, "extensions");
737 if (!exts) {
738 mDoc.AddMember("extensions", Value().SetObject().Move(), mDoc.GetAllocator());
739 exts = FindObject(mDoc, "extensions");
740 }
741
742 if (!(container = FindObject(*exts, d.mExtId))) {
743 exts->AddMember(StringRef(d.mExtId), Value().SetObject().Move(), mDoc.GetAllocator());
744 container = FindObject(*exts, d.mExtId);
745 }
746 }
747
748 Value* dict;
749 if (!(dict = FindArray(*container, d.mDictId))) {
750 container->AddMember(StringRef(d.mDictId), Value().SetArray().Move(), mDoc.GetAllocator());
751 dict = FindArray(*container, d.mDictId);
752 if (nullptr == dict) {
753 return;
754 }
755 }
756
757 for (size_t i = 0; i < d.mObjs.size(); ++i) {
758 if (d.mObjs[i]->IsSpecial()) continue;
759
760 Value obj;
761 obj.SetObject();
762
763 if (!d.mObjs[i]->name.empty()) {
764 obj.AddMember("name", StringRef(d.mObjs[i]->name.c_str()), mAl);
765 }
766
767 Write(obj, *d.mObjs[i], *this);
768
769 dict->PushBack(obj, mAl);
770 }
