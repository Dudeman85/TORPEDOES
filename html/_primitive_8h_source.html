27 Primitive(std::vector<float> vertices, std::vector<unsigned int> indices)
28 {
29 numVertices = indices.size();
30
31 //Make the Vertex Array Object, Vertex Buffer Object, and Element Buffer Object
32 glGenVertexArrays(1, &VAO);
33 glGenBuffers(1, &VBO);
34 glGenBuffers(1, &EBO);
35
36 //Bind the Vertex Array Object
37 glBindVertexArray(VAO);
38
39 //Bind the Vertex Bufer Object and set vertices
40 glBindBuffer(GL_ARRAY_BUFFER, VBO);
41 glBufferData(GL_ARRAY_BUFFER, sizeof(float) * vertices.size(), vertices.data(), GL_STATIC_DRAW);
42
43 //Bind and set indices to EBO
44 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
45 glBufferData(GL_ELEMENT_ARRAY_BUFFER, sizeof(unsigned int) * indices.size(), indices.data(), GL_STATIC_DRAW);
46
47 //Configure Vertex attribute at location 0 aka position
48 glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(float), (void*)0);
49 glEnableVertexAttribArray(0);
50
51 //Unbind all buffers and arrays
52 glBindVertexArray(0);
53 glBindBuffer(GL_ARRAY_BUFFER, 0);
54 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
55 }

77 model = glm::rotate(model, glm::radians(transform.rotation.x), glm::vec3(1.0f, 0.0f, 0.0f));
78 model = glm::rotate(model, glm::radians(transform.rotation.y), glm::vec3(0.0f, 1.0f, 0.0f));
79 model = glm::rotate(model, glm::radians(transform.rotation.z), glm::vec3(0.0f, 0.0f, 1.0f));
80 //Scale
81 model = glm::scale(model, transform.scale.ToGlm());
82
83 //Give the shader the model matrix
84 unsigned int modelLoc = glGetUniformLocation(defaultShader->ID, "model");
85 glUniformMatrix4fv(modelLoc, 1, GL_FALSE, glm::value_ptr(model));
86
87 //Give the shader the primitive's color
88 unsigned int colorLoc = glGetUniformLocation(defaultShader->ID, "color");
89 glUniform4f(colorLoc, color.x / 255, color.y / 255, color.z / 255, 1);
90
91 //Get the view and projection locations
92 unsigned int viewLoc = glGetUniformLocation(defaultShader->ID, "view");
93 unsigned int projLoc = glGetUniformLocation(defaultShader->ID, "projection");
94
95 //Give the shader the camera's view matrix
96 glUniformMatrix4fv(viewLoc, 1, GL_FALSE, glm::value_ptr(cam->GetViewMatrix()));
97 //Give the shader the camera's projection matrix
98 glUniformMatrix4fv(projLoc, 1, GL_FALSE, glm::value_ptr(cam->GetProjectionMatrix()));
99
100
101 glDrawElements(GL_LINE_LOOP, numVertices, GL_UNSIGNED_INT, 0);
