270 static glm::mat4 GetLocalTranformMatrix(ecs::Entity entity)
271 {
272 Transform& transform = ecs::GetComponent<Transform>(entity);
273
274 //Create the transform matrix
275 glm::mat4 transformMatrix = glm::mat4(1.0f);
276 //Position
277 transformMatrix = glm::translate(transformMatrix, transform.position.ToGlm());
278 //X, Y, Z euler rotations

280 transformMatrix = glm::rotate(transformMatrix, glm::radians(transform.rotation.y), glm::vec3(0.0f, 1.0f, 0.0f));

289 static glm::mat4 GetGlobalTransformMatrix(ecs::Entity entity)
290 {
291 Transform& transform = ecs::GetComponent<Transform>(entity);
292
293 //Go through all the parents and add them to a list
294 std::vector<ecs::Entity> parents;
295 parents.push_back(entity);
296 ecs::Entity currentParent = transform.parent;
297 while (currentParent != 0)
298 {
299 parents.push_back(currentParent);
300
301 currentParent = ecs::GetComponent<Transform>(currentParent).parent;
302 }
303
304 //Create the transform matrix
305 glm::mat4 transformMatrix = GetLocalTranformMatrix(parents.back());
306
307 //Go through each parent from root up
308 for (int i = parents.size() - 2; i >= 0; i--)
309 {
310 transformMatrix *= GetLocalTranformMatrix(parents[i]);
311 }
312
313 return transformMatrix;
314 }

317 static std::vector<Vector3> ApplyTransforms(std::vector<Vector3> vertices, Vector3 rotation, Vector3 scale, Vector3 position)
318 {
319 //Create the transform matrix
320 glm::mat4 transform = glm::mat4(1.0f);
321 //Position
322 transform = glm::translate(transform, position.ToGlm());
323 //X, Y, Z euler rotations
324 transform = glm::rotate(transform, glm::radians(rotation.x), glm::vec3(1.0f, 0.0f, 0.0f));
325 transform = glm::rotate(transform, glm::radians(rotation.y), glm::vec3(0.0f, 1.0f, 0.0f));
326 transform = glm::rotate(transform, glm::radians(rotation.z), glm::vec3(0.0f, 0.0f, 1.0f));
327 //Scale
328 transform = glm::scale(transform, scale.ToGlm());
329
330 std::vector<Vector3> transformedVerts;
331 //For each vertice apply the transforms
332 for (int i = 0; i < vertices.size(); i++)
333 {
334 //Many type conversions later we have applied transform matrix
335 glm::vec3 glmVert = glm::vec3(glm::vec4(vertices[i].ToGlm(), 0) * transform);
336 Vector3 vert(glmVert.x, glmVert.y, glmVert.z);
337 transformedVerts.push_back(vert);
338 }
339
340 return transformedVerts;
341 }

344 static std::vector<Vector2> ApplyTransforms2D(std::vector<Vector2> vertices, float rotation, Vector2 scale, Vector2 position)
345 {
346 std::vector<Vector2> transformedVerts;
347 //For each vertice apply scale and rotation
348 for (int i = 0; i < vertices.size(); i++)
349 {
350 //Apply transform to the polygon collider
351 Vector2 transformedVert = vertices[i];
352 //Rotate
353 float angle = Radians(rotation);
354 transformedVert.x = vertices[i].x * cos(angle) - vertices[i].y * sin(angle);
355 transformedVert.y = vertices[i].x * sin(angle) + vertices[i].y * cos(angle);
356 //Scale
357 transformedVert *= scale;
358 //Move
359 transformedVert += position;
360
361 transformedVerts.push_back(transformedVert);
362 }
363 return transformedVerts;
364 }
