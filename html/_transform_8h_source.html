290 //Get the global Transform matrix of an entity after all parent transforms have been applied
291 static glm::mat4 GetGlobalTransformMatrix(ecs::Entity entity)
292 {
293 Transform& transform = ecs::GetComponent<Transform>(entity);
294
295 //Go through all the parents and add them to a list
296 std::vector<ecs::Entity> parents;
297 parents.push_back(entity);
298 ecs::Entity currentParent = transform.parent;
299 while (currentParent != 0)
300 {
301 parents.push_back(currentParent);
302
303 currentParent = ecs::GetComponent<Transform>(currentParent).parent;
304 }
305
306 //Create the transform matrix
307 glm::mat4 transformMatrix = GetLocalTranformMatrix(parents.back());
308
309 //Go through each parent from root up
310 for (int i = parents.size() - 2; i >= 0; i--)
311 {
312 transformMatrix *= GetLocalTranformMatrix(parents[i]);
313 }
314
315 return transformMatrix;
316 }

319 static std::vector<Vector3> ApplyTransforms(std::vector<Vector3> vertices, Vector3 rotation, Vector3 scale, Vector3 position)
320 {
321 //Create the transform matrix
322 glm::mat4 transform = glm::mat4(1.0f);
323 //Position
324 transform = glm::translate(transform, position.ToGlm());
325 //X, Y, Z euler rotations
326 transform = glm::rotate(transform, glm::radians(rotation.x), glm::vec3(1.0f, 0.0f, 0.0f));
327 transform = glm::rotate(transform, glm::radians(rotation.y), glm::vec3(0.0f, 1.0f, 0.0f));
328 transform = glm::rotate(transform, glm::radians(rotation.z), glm::vec3(0.0f, 0.0f, 1.0f));
329 //Scale
330 transform = glm::scale(transform, scale.ToGlm());
331
332 std::vector<Vector3> transformedVerts;
333 //For each vertice apply the transforms
334 for (int i = 0; i < vertices.size(); i++)
335 {
336 //Many type conversions later we have applied transform matrix
337 glm::vec3 glmVert = glm::vec3(glm::vec4(vertices[i].ToGlm(), 0) * transform);
338 Vector3 vert(glmVert.x, glmVert.y, glmVert.z);
339 transformedVerts.push_back(vert);
340 }
341
342 return transformedVerts;
