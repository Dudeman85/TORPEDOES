82 Material(const Material &other) = default;
83 Material &operator=(const Material &other) = default;
84
85
87 Material(Material &&other) AI_NO_EXCEPT
88 : D3DS::Material(std::move(other))
89 , avSubMaterials(std::move(other.avSubMaterials))
90 , pcInstance(std::move(other.pcInstance))
91 , bNeed(std::move(other.bNeed))
92 {
93 other.pcInstance = nullptr;
94 }
95
96
97 Material &operator=(Material &&other) AI_NO_EXCEPT {
98 if (this == &other) {
99 return *this;
100 }
101
102 D3DS::Material::operator=(std::move(other));
103
104 avSubMaterials = std::move(other.avSubMaterials);
105 pcInstance = std::move(other.pcInstance);
106 bNeed = std::move(other.bNeed);
107
108 other.pcInstance = nullptr;
109
110 return *this;
111 }
112
113
114 ~Material() {}
115

118 std::vector<Material> avSubMaterials;
119
121 aiMaterial* pcInstance;
122
124 bool bNeed;
