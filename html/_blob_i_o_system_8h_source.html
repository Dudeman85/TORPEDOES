68 BlobIOStream(BlobIOSystem* creator, const std::string& file, size_t initial = 4096)
69 : buffer()

77 }

81
82public:
83
84 // -------------------------------------------------------------------
85 aiExportDataBlob* GetBlob()
86 {
87 aiExportDataBlob* blob = new aiExportDataBlob();
88 blob->size = file_size;
89 blob->data = buffer;
90
91 buffer = NULL;
92
93 return blob;
94 }
95
96
97public:
98
99
100 // -------------------------------------------------------------------
101 virtual size_t Read( void *,
102 size_t,
103 size_t )
104 {
105 return 0;
106 }
107
108 // -------------------------------------------------------------------
109 virtual size_t Write(const void* pvBuffer,
110 size_t pSize,
111 size_t pCount)
112 {
113 pSize *= pCount;
114 if (cursor + pSize > cur_size) {
115 Grow(cursor + pSize);
116 }
117
118 memcpy(buffer+cursor, pvBuffer, pSize);
119 cursor += pSize;
120
121 file_size = std::max(file_size,cursor);
122 return pCount;
123 }
124
125 // -------------------------------------------------------------------
126 virtual aiReturn Seek(size_t pOffset,
127 aiOrigin pOrigin)
128 {
129 switch(pOrigin)
130 {
131 case aiOrigin_CUR:
132 cursor += pOffset;
133 break;
134
135 case aiOrigin_END:
136 cursor = file_size - pOffset;
137 break;
138
139 case aiOrigin_SET:
140 cursor = pOffset;
141 break;
142
143 default:
144 return AI_FAILURE;
145 }
146
147 if (cursor > file_size) {
148 Grow(cursor);
149 }
150
151 file_size = std::max(cursor,file_size);
152 return AI_SUCCESS;
153 }
154
155 // -------------------------------------------------------------------
156 virtual size_t Tell() const
157 {
158 return cursor;
159 }
160
161 // -------------------------------------------------------------------
162 virtual size_t FileSize() const
163 {
164 return file_size;
165 }
166
167 // -------------------------------------------------------------------
168 virtual void Flush()
169 {
170 // ignore
171 }
172
173
174
175private:
176
177 // -------------------------------------------------------------------
178 void Grow(size_t need = 0)
179 {
180 // 1.5 and phi are very heap-friendly growth factors (the first
181 // allows for frequent re-use of heap blocks, the second
182 // forms a fibonacci sequence with similar characteristics -
183 // since this heavily depends on the heap implementation
184 // and other factors as well, i'll just go with 1.5 since
185 // it is quicker to compute).
186 size_t new_size = std::max(initial, std::max( need, cur_size+(cur_size>>1) ));
187
188 const uint8_t* const old = buffer;
189 buffer = new uint8_t[new_size];
190
191 if (old) {
192 memcpy(buffer,old,cur_size);
193 delete[] old;
194 }
195
196 cur_size = new_size;
197 }
198
199private:
200
201 uint8_t* buffer;
202 size_t cur_size,file_size, cursor, initial;
203
204 const std::string file;
205 BlobIOSystem* const creator;
206};

228 for(BlobEntry& blobby : blobs) {
229 delete blobby.second;
230 }
231 }

233public:
234
235 // -------------------------------------------------------------------
236 const char* GetMagicFileName() const
237 {
238 return AI_BLOBIO_MAGIC;
239 }
240
241
242 // -------------------------------------------------------------------
243 aiExportDataBlob* GetBlobChain()
244 {
245 // one must be the master
246 aiExportDataBlob* master = NULL, *cur;
247 for(const BlobEntry& blobby : blobs) {
248 if (blobby.first == AI_BLOBIO_MAGIC) {
249 master = blobby.second;
250 break;
251 }
252 }
253 if (!master) {
254 ASSIMP_LOG_ERROR("BlobIOSystem: no data written or master file was not closed properly.");
255 return NULL;
256 }
257
258 master->name.Set("");
259
260 cur = master;
261 for(const BlobEntry& blobby : blobs) {
262 if (blobby.second == master) {
263 continue;
264 }
265
266 cur->next = blobby.second;
267 cur = cur->next;
268
269 // extract the file extension from the file written
270 const std::string::size_type s = blobby.first.find_first_of('.');
271 cur->name.Set(s == std::string::npos ? blobby.first : blobby.first.substr(s+1));
272 }
273
274 // give up blob ownership
275 blobs.clear();
276 return master;
277 }
278
279public:
280
281 // -------------------------------------------------------------------
282 virtual bool Exists( const char* pFile) const {
283 return created.find(std::string(pFile)) != created.end();
284 }
285
286
287 // -------------------------------------------------------------------
288 virtual char getOsSeparator() const {
289 return '/';
290 }
291
292
293 // -------------------------------------------------------------------
294 virtual IOStream* Open(const char* pFile,
295 const char* pMode)
296 {
297 if (pMode[0] != 'w') {
298 return NULL;
299 }
300
301 created.insert(std::string(pFile));
302 return new BlobIOStream(this,std::string(pFile));
303 }
304
305 // -------------------------------------------------------------------
306 virtual void Close( IOStream* pFile)
307 {
308 delete pFile;
309 }
310
311private:
312
313 // -------------------------------------------------------------------
314 void OnDestruct(const std::string& filename, BlobIOStream* child)
315 {
316 // we don't know in which the files are closed, so we
317 // can't reliably say that the first must be the master
318 // file ...
319 blobs.push_back( BlobEntry(filename,child->GetBlob()) );
320 }
321
322private:
323 std::set<std::string> created;
324 std::vector< BlobEntry > blobs;
325};
