198 inline void Write(Value& obj, Image& img, AssetWriter& w)
199 {
200 std::string uri;
201 if (w.mAsset.extensionsUsed.KHR_binary_glTF && img.bufferView) {
202 Value exts, ext;
203 exts.SetObject();
204 ext.SetObject();
205
206 ext.AddMember("bufferView", StringRef(img.bufferView->id), w.mAl);
207
208 if (!img.mimeType.empty())
209 ext.AddMember("mimeType", StringRef(img.mimeType), w.mAl);
210
211 exts.AddMember("KHR_binary_glTF", ext, w.mAl);
212 obj.AddMember("extensions", exts, w.mAl);
213 return;
214 }
215 else if (img.HasData()) {
216 uri = "data:" + (img.mimeType.empty() ? "application/octet-stream" : img.mimeType);
217 uri += ";base64,";
218 glTFCommon::Util::EncodeBase64(img.GetData(), img.GetDataLength(), uri);
219 }
220 else {
221 uri = img.uri;
222 }
223
224 obj.AddMember("uri", Value(uri, w.mAl).Move(), w.mAl);
225 }

275 inline void Write(Value& obj, Mesh& m, AssetWriter& w)
276 {
277 /********************* Name **********************/
278 obj.AddMember("name", m.name, w.mAl);
279
280 /**************** Mesh extensions ****************/
281 if(m.Extension.size() > 0)
282 {
283 Value json_extensions;
284
285 json_extensions.SetObject();
286 for(Mesh::SExtension* ptr_ext : m.Extension)
287 {
288 switch(ptr_ext->Type)
289 {
290#ifdef ASSIMP_IMPORTER_GLTF_USE_OPEN3DGC
291 case Mesh::SExtension::EType::Compression_Open3DGC:
292 {
293 Value json_comp_data;
294 Mesh::SCompression_Open3DGC* ptr_ext_comp = (Mesh::SCompression_Open3DGC*)ptr_ext;
295
296 // filling object "compressedData"
297 json_comp_data.SetObject();
298 json_comp_data.AddMember("buffer", ptr_ext_comp->Buffer, w.mAl);
299 json_comp_data.AddMember("byteOffset", static_cast<uint64_t>(ptr_ext_comp->Offset), w.mAl);
300 json_comp_data.AddMember("componentType", 5121, w.mAl);
301 json_comp_data.AddMember("type", "SCALAR", w.mAl);
302 json_comp_data.AddMember("count", static_cast<uint64_t>(ptr_ext_comp->Count), w.mAl);
303 if(ptr_ext_comp->Binary)
304 json_comp_data.AddMember("mode", "binary", w.mAl);
305 else
306 json_comp_data.AddMember("mode", "ascii", w.mAl);
307
308 json_comp_data.AddMember("indicesCount", static_cast<uint64_t>(ptr_ext_comp->IndicesCount), w.mAl);
309 json_comp_data.AddMember("verticesCount", static_cast<uint64_t>(ptr_ext_comp->VerticesCount), w.mAl);
310 // filling object "Open3DGC-compression"
311 Value json_o3dgc;
312
313 json_o3dgc.SetObject();
314 json_o3dgc.AddMember("compressedData", json_comp_data, w.mAl);
315 // add member to object "extensions"
316 json_extensions.AddMember("Open3DGC-compression", json_o3dgc, w.mAl);
317 }
318
319 break;
320#endif
321 default:
322 throw DeadlyImportError("GLTF: Can not write mesh: unknown mesh extension, only Open3DGC is supported.");
323 }// switch(ptr_ext->Type)
324 }// for(Mesh::SExtension* ptr_ext : m.Extension)
325
326 // Add extensions to mesh
327 obj.AddMember("extensions", json_extensions, w.mAl);
328 }// if(m.Extension.size() > 0)
329
330 /****************** Primitives *******************/
331 Value primitives;
332 primitives.SetArray();
333 primitives.Reserve(unsigned(m.primitives.size()), w.mAl);
334
335 for (size_t i = 0; i < m.primitives.size(); ++i) {
336 Mesh::Primitive& p = m.primitives[i];
337 Value prim;
338 prim.SetObject();
339 {
340 prim.AddMember("mode", Value(int(p.mode)).Move(), w.mAl);
341
342 if (p.material)
343 prim.AddMember("material", p.material->id, w.mAl);
344
345 if (p.indices)
346 prim.AddMember("indices", Value(p.indices->id, w.mAl).Move(), w.mAl);
347
348 Value attrs;
349 attrs.SetObject();
350 {
351 WriteAttrs(w, attrs, p.attributes.position, "POSITION");
352 WriteAttrs(w, attrs, p.attributes.normal, "NORMAL");
353 WriteAttrs(w, attrs, p.attributes.texcoord, "TEXCOORD", true);
354 WriteAttrs(w, attrs, p.attributes.color, "COLOR");
355 WriteAttrs(w, attrs, p.attributes.joint, "JOINT");
356 WriteAttrs(w, attrs, p.attributes.jointmatrix, "JOINTMATRIX");
357 WriteAttrs(w, attrs, p.attributes.weight, "WEIGHT");
358 }
359 prim.AddMember("attributes", attrs, w.mAl);
360 }
361 primitives.PushBack(prim, w.mAl);
362 }
363
364 obj.AddMember("primitives", primitives, w.mAl);
365 }

500 inline void AssetWriter::WriteFile(const char* path)
501 {
502 std::unique_ptr<IOStream> jsonOutFile(mAsset.OpenFile(path, "wt", true));
503
504 if (jsonOutFile == 0) {
505 throw DeadlyExportError("Could not open output file: " + std::string(path));
506 }
507
508 StringBuffer docBuffer;
509
510 PrettyWriter<StringBuffer> writer(docBuffer);
511 mDoc.Accept(writer);
512
513 if (jsonOutFile->Write(docBuffer.GetString(), docBuffer.GetSize(), 1) != 1) {
514 throw DeadlyExportError("Failed to write scene data!");
515 }
516
517 // Write buffer data to separate .bin files
518 for (unsigned int i = 0; i < mAsset.buffers.Size(); ++i) {
519 Ref<Buffer> b = mAsset.buffers.Get(i);
520
521 std::string binPath = b->GetURI();
522
523 std::unique_ptr<IOStream> binOutFile(mAsset.OpenFile(binPath, "wb", true));
524
525 if (binOutFile == 0) {
526 throw DeadlyExportError("Could not open output file: " + binPath);
527 }
528
529 if (b->byteLength > 0) {
530 if (binOutFile->Write(b->GetPointer(), b->byteLength, 1) != 1) {
531 throw DeadlyExportError("Failed to write binary file: " + binPath);
532 }
533 }
534 }

537 inline void AssetWriter::WriteGLBFile(const char* path)
538 {
539 std::unique_ptr<IOStream> outfile(mAsset.OpenFile(path, "wb", true));
540
541 if (outfile == 0) {
542 throw DeadlyExportError("Could not open output file: " + std::string(path));
543 }
544
545 // we will write the header later, skip its size
546 outfile->Seek(sizeof(GLB_Header), aiOrigin_SET);
547
548 StringBuffer docBuffer;
549 Writer<StringBuffer> writer(docBuffer);
550 mDoc.Accept(writer);
551
552 if (outfile->Write(docBuffer.GetString(), docBuffer.GetSize(), 1) != 1) {
553 throw DeadlyExportError("Failed to write scene data!");
554 }
555
556 WriteBinaryData(outfile.get(), docBuffer.GetSize());
557 }

559 inline void AssetWriter::WriteBinaryData(IOStream* outfile, size_t sceneLength)
560 {
561 //
562 // write the body data
563 //
564
565 size_t bodyLength = 0;
566 if (Ref<Buffer> b = mAsset.GetBodyBuffer()) {
567 bodyLength = b->byteLength;
568
569 if (bodyLength > 0) {
570 size_t bodyOffset = sizeof(GLB_Header) + sceneLength;
571 bodyOffset = (bodyOffset + 3) & ~3; // Round up to next multiple of 4
572
573 outfile->Seek(bodyOffset, aiOrigin_SET);
574
575 if (outfile->Write(b->GetPointer(), b->byteLength, 1) != 1) {
576 throw DeadlyExportError("Failed to write body data!");
577 }
578 }
579 }
580
581 //
582 // write the header
583 //
584
585 GLB_Header header;
586 memcpy(header.magic, AI_GLB_MAGIC_NUMBER, sizeof(header.magic));
587
588 header.version = 1;
589 AI_SWAP4(header.version);
590
591 header.length = uint32_t(sizeof(header) + sceneLength + bodyLength);
592 AI_SWAP4(header.length);
593
594 header.sceneLength = uint32_t(sceneLength);
595 AI_SWAP4(header.sceneLength);
596
597 header.sceneFormat = SceneFormat_JSON;
598 AI_SWAP4(header.sceneFormat);
599
600 outfile->Seek(0, aiOrigin_SET);
601
602 if (outfile->Write(&header, 1, sizeof(header)) != sizeof(header)) {
603 throw DeadlyExportError("Failed to write the header!");
604 }
605 }

633 template<class T>
634 void AssetWriter::WriteObjects(LazyDict<T>& d)
635 {
636 if (d.mObjs.empty()) return;
637
638 Value* container = &mDoc;
639
640 if (d.mExtId) {
641 Value* exts = FindObject(mDoc, "extensions");
642 if (!exts) {
643 mDoc.AddMember("extensions", Value().SetObject().Move(), mDoc.GetAllocator());
644 exts = FindObject(mDoc, "extensions");
645 }
646
647 if (!(container = FindObject(*exts, d.mExtId))) {
648 exts->AddMember(StringRef(d.mExtId), Value().SetObject().Move(), mDoc.GetAllocator());
649 container = FindObject(*exts, d.mExtId);
650 }
651 }
652
653 Value* dict;
654 if (!(dict = FindObject(*container, d.mDictId))) {
655 container->AddMember(StringRef(d.mDictId), Value().SetObject().Move(), mDoc.GetAllocator());
656 dict = FindObject(*container, d.mDictId);
657 }
658
659 for (size_t i = 0; i < d.mObjs.size(); ++i) {
660 if (d.mObjs[i]->IsSpecial()) continue;
661
662 Value obj;
663 obj.SetObject();
664
665 if (!d.mObjs[i]->name.empty()) {
666 obj.AddMember("name", StringRef(d.mObjs[i]->name.c_str()), mAl);
667 }
668
669 Write(obj, *d.mObjs[i], *this);
670
671 dict->AddMember(StringRef(d.mObjs[i]->id), obj, mAl);
672 }
673 }
