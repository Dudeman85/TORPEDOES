127 const Element* operator[] (const std::string& index) const {
128 ElementMap::const_iterator it = elements.find(index);
129 return it == elements.end() ? NULL : (*it).second;
130 }
131
132 const Element* FindElementCaseInsensitive(const std::string& elementName) const {
133 const char* elementNameCStr = elementName.c_str();
134 for (auto element = elements.begin(); element != elements.end(); ++element)
135 {
136 if (!ASSIMP_strincmp(element->first.c_str(), elementNameCStr, MAXLEN)) {
137 return element->second;
138 }
139 }
140 return NULL;
141 }
142
143 ElementCollection GetCollection(const std::string& index) const {
144 return elements.equal_range(index);
145 }
146
147 const ElementMap& Elements() const {
148 return elements;
149 }

151private:
152 ElementMap elements;
