56// NOTE: the functions below are mostly intended as replacement for
57// std::upper, std::lower, std::isupper, std::islower, std::isspace.
58// we don't bother of locales. We don't want them. We want reliable
59// (i.e. identical) results across all locales.
60
61// The functions below accept any character type, but know only
62// about ASCII. However, UTF-32 is the only safe ASCII superset to
63// use since it doesn't have multi-byte sequences.
64

134// ---------------------------------------------------------------------------------
135template <class char_t>
136AI_FORCE_INLINE
137bool SkipLine( const char_t* in, const char_t** out) {
138 while( *in != ( char_t )'\r' && *in != ( char_t )'\n' && *in != ( char_t )'\0' ) {
139 ++in;
140 }
141
142 // files are opened in binary mode. Ergo there are both NL and CR
143 while( *in == ( char_t )'\r' || *in == ( char_t )'\n' ) {
144 ++in;
145 }
146 *out = in;
147 return *in != (char_t)'\0';
148}

175// ---------------------------------------------------------------------------------
176template <class char_t>
177AI_FORCE_INLINE
178bool GetNextLine( const char_t*& buffer, char_t out[ BufferSize ] ) {
179 if( ( char_t )'\0' == *buffer ) {
180 return false;
181 }
182
183 char* _out = out;
184 char* const end = _out + BufferSize;
185 while( !IsLineEnd( *buffer ) && _out < end ) {
186 *_out++ = *buffer++;
187 }
188 *_out = (char_t)'\0';
189
190 while( IsLineEnd( *buffer ) && '\0' != *buffer ) {
191 ++buffer;
192 }
193
194 return true;
195}

204template <class char_t>
205AI_FORCE_INLINE
206bool TokenMatch(char_t*& in, const char* token, unsigned int len)
207{
208 if (!::strncmp(token,in,len) && IsSpaceOrNewLine(in[len])) {
209 if (in[len] != '\0') {
210 in += len+1;
211 } else {
212 // If EOF after the token make sure we don't go past end of buffer
213 in += len;
214 }
215 return true;
216 }
217
218 return false;
219}

226AI_FORCE_INLINE
227bool TokenMatchI(const char*& in, const char* token, unsigned int len) {
228 if (!ASSIMP_strincmp(token,in,len) && IsSpaceOrNewLine(in[len])) {
229 in += len+1;
230 return true;
231 }
232 return false;
233}

244// ---------------------------------------------------------------------------------
245AI_FORCE_INLINE
246std::string GetNextToken(const char*& in) {
247 SkipSpacesAndLineEnd(&in);
248 const char* cur = in;
249 while ( !IsSpaceOrNewLine( *in ) ) {
250 ++in;
251 }
252 return std::string(cur,(size_t)(in-cur));
253}
