258 // ---------------------------------------------------------------------
260 template <typename T>
261 void Put(T f) {
262 Intern :: Getter<SwapEndianess,T,RuntimeSwitch>() (&f, le);
263
264 if (cursor + sizeof(T) >= buffer.size()) {
265 buffer.resize(cursor + sizeof(T));
266 }
267
268 void* dest = &buffer[cursor];
269
270 // reinterpret_cast + assignment breaks strict aliasing rules
271 // and generally causes trouble on platforms such as ARM that
272 // do not silently ignore alignment faults.
273 ::memcpy(dest, &f, sizeof(T));
274 cursor += sizeof(T);
275 }
276
