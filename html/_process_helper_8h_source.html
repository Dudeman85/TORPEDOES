152// -------------------------------------------------------------------------------
153// Start points for ArrayBounds<T> for all supported Ts
154template <typename T>
155struct MinMaxChooser;
156
157template <> struct MinMaxChooser<float> {
158 void operator ()(float& min,float& max) {
159 max = -1e10f;
160 min = 1e10f;
161}};
162template <> struct MinMaxChooser<double> {
163 void operator ()(double& min,double& max) {
164 max = -1e10;
165 min = 1e10;
166}};
167template <> struct MinMaxChooser<unsigned int> {
168 void operator ()(unsigned int& min,unsigned int& max) {
169 max = 0;
170 min = (1u<<(sizeof(unsigned int)*8-1));
171}};
172
173template <typename T> struct MinMaxChooser< aiVector3t<T> > {
174 void operator ()(aiVector3t<T>& min,aiVector3t<T>& max) {
175 max = aiVector3t<T>(-1e10f,-1e10f,-1e10f);
176 min = aiVector3t<T>( 1e10f, 1e10f, 1e10f);
177}};
178template <typename T> struct MinMaxChooser< aiVector2t<T> > {
179 void operator ()(aiVector2t<T>& min,aiVector2t<T>& max) {
180 max = aiVector2t<T>(-1e10f,-1e10f);
181 min = aiVector2t<T>( 1e10f, 1e10f);
182 }};
183template <typename T> struct MinMaxChooser< aiColor4t<T> > {
184 void operator ()(aiColor4t<T>& min,aiColor4t<T>& max) {
185 max = aiColor4t<T>(-1e10f,-1e10f,-1e10f,-1e10f);
186 min = aiColor4t<T>( 1e10f, 1e10f, 1e10f, 1e10f);
187}};
188
189template <typename T> struct MinMaxChooser< aiQuaterniont<T> > {
190 void operator ()(aiQuaterniont<T>& min,aiQuaterniont<T>& max) {
191 max = aiQuaterniont<T>(-1e10f,-1e10f,-1e10f,-1e10f);
192 min = aiQuaterniont<T>( 1e10f, 1e10f, 1e10f, 1e10f);
193}};
194
195template <> struct MinMaxChooser<aiVectorKey> {
196 void operator ()(aiVectorKey& min,aiVectorKey& max) {
197 MinMaxChooser<double>()(min.mTime,max.mTime);
198 MinMaxChooser<aiVector3D>()(min.mValue,max.mValue);
199}};
200template <> struct MinMaxChooser<aiQuatKey> {
201 void operator ()(aiQuatKey& min,aiQuatKey& max) {
202 MinMaxChooser<double>()(min.mTime,max.mTime);
203 MinMaxChooser<aiQuaternion>()(min.mValue,max.mValue);
204}};
205
206template <> struct MinMaxChooser<aiVertexWeight> {
207 void operator ()(aiVertexWeight& min,aiVertexWeight& max) {
208 MinMaxChooser<unsigned int>()(min.mVertexId,max.mVertexId);
209 MinMaxChooser<float>()(min.mWeight,max.mWeight);
210}};

212// -------------------------------------------------------------------------------
219template <typename T>
220inline void ArrayBounds(const T* in, unsigned int size, T& min, T& max)
221{
222 MinMaxChooser<T> ()(min,max);
223 for (unsigned int i = 0; i < size;++i) {
224 min = std::min(in[i],min);
225 max = std::max(in[i],max);
226 }
227}

338// Utility postprocess step to share the spatial sort tree between
339// all steps which use it to speedup its computations.
340class ComputeSpatialSortProcess : public BaseProcess
341{
342 bool IsActive( unsigned int pFlags) const
343 {
344 return NULL != shared && 0 != (pFlags & (aiProcess_CalcTangentSpace |
345 aiProcess_GenNormals | aiProcess_JoinIdenticalVertices));
346 }
347
348 void Execute( aiScene* pScene)
349 {
350 typedef std::pair<SpatialSort, ai_real> _Type;
351 ASSIMP_LOG_DEBUG("Generate spatially-sorted vertex cache");
352
353 std::vector<_Type>* p = new std::vector<_Type>(pScene->mNumMeshes);
354 std::vector<_Type>::iterator it = p->begin();
355
356 for (unsigned int i = 0; i < pScene->mNumMeshes; ++i, ++it) {
357 aiMesh* mesh = pScene->mMeshes[i];
358 _Type& blubb = *it;
359 blubb.first.Fill(mesh->mVertices,mesh->mNumVertices,sizeof(aiVector3D));
360 blubb.second = ComputePositionEpsilon(mesh);
361 }
362
363 shared->AddProperty(AI_SPP_SPATIAL_SORT,p);
364 }
365};
366
367// -------------------------------------------------------------------------------
368// ... and the same again to cleanup the whole stuff
369class DestroySpatialSortProcess : public BaseProcess
370{
371 bool IsActive( unsigned int pFlags) const
372 {
373 return NULL != shared && 0 != (pFlags & (aiProcess_CalcTangentSpace |
374 aiProcess_GenNormals | aiProcess_JoinIdenticalVertices));
375 }
376
377 void Execute( aiScene* /*pScene*/)
378 {
379 shared->RemoveProperty(AI_SPP_SPATIAL_SORT);
380 }
381};
