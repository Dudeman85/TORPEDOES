81 void push_back(const T& element)
82 {
83 if (used + 1 > allocated)

85 // reallocate(used * 2 +1);

328 s32 binary_search(const T& element, s32 left, s32 right)
329 {
330 if (!used)
331 return -1;
332
333 sort();
334
335 s32 m;
336
337 do
338 {
339 m = (left+right)>>1;
340
341 if (element < data[m])
342 right = m - 1;
343 else
344 left = m + 1;
345
346 } while((element < data[m] || data[m] < element) && left<=right);
347
348 // this last line equals to:
349 // " while((element != array[m]) && left<=right);"
350 // but we only want to use the '<' operator.
351 // the same in next line, it is "(element == array[m])"
352
353 if (!(element < data[m]) && !(data[m] < element))
354 return m;
355
356 return -1;
357 }
