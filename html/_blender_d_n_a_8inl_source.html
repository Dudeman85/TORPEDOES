58{

61 throw Error((Formatter::format(),

103template <int error_policy, typename T, size_t M>
104void Structure :: ReadFieldArray(T (& out)[M], const char* name, const FileDatabase& db) const
105{
106 const StreamReaderAny::pos old = db.reader->GetCurrentPos();
107 try {
108 const Field& f = (*this)[name];
109 const Structure& s = db.dna[f.type];
110
111 // is the input actually an array?
112 if (!(f.flags & FieldFlag_Array)) {
113 throw Error((Formatter::format(),"Field `",name,"` of structure `",
114 this->name,"` ought to be an array of size ",M
115 ));
116 }
117
118 db.reader->IncPtr(f.offset);
119
120 // size conversions are always allowed, regardless of error_policy
121 unsigned int i = 0;
122 for(; i < std::min(f.array_sizes[0],M); ++i) {
123 s.Convert(out[i],db);
124 }
125 for(; i < M; ++i) {
126 _defaultInitializer<ErrorPolicy_Igno>()(out[i]);
127 }
128 }
129 catch (const Error& e) {
130 _defaultInitializer<error_policy>()(out,e.what());
131 }
132
133 // and recover the previous stream position
134 db.reader->SetCurrentPos(old);
135
136#ifndef ASSIMP_BUILD_BLENDER_NO_STATS
137 ++db.stats().fields_read;
138#endif
139}

187template <int error_policy, template <typename> class TOUT, typename T>
188bool Structure :: ReadFieldPtr(TOUT<T>& out, const char* name, const FileDatabase& db,
189 bool non_recursive /*= false*/) const
190{
191 const StreamReaderAny::pos old = db.reader->GetCurrentPos();
192 Pointer ptrval;
193 const Field* f;
194 try {
195 f = &(*this)[name];
196
197 // sanity check, should never happen if the genblenddna script is right
198 if (!(f->flags & FieldFlag_Pointer)) {
199 throw Error((Formatter::format(),"Field `",name,"` of structure `",
200 this->name,"` ought to be a pointer"));
201 }
202
203 db.reader->IncPtr(f->offset);
204 Convert(ptrval,db);
205 // actually it is meaningless on which Structure the Convert is called
206 // because the `Pointer` argument triggers a special implementation.
207 }
208 catch (const Error& e) {
209 _defaultInitializer<error_policy>()(out,e.what());
210
211 out.reset();
212 return false;
213 }
214
215 // resolve the pointer and load the corresponding structure
216 const bool res = ResolvePointer(out,ptrval,db,*f, non_recursive);
217
218 if(!non_recursive) {
219 // and recover the previous stream position
220 db.reader->SetCurrentPos(old);
221 }
222
223#ifndef ASSIMP_BUILD_BLENDER_NO_STATS
224 ++db.stats().fields_read;
225#endif
226
227 return res;
228}

231template <int error_policy, template <typename> class TOUT, typename T, size_t N>
232bool Structure :: ReadFieldPtr(TOUT<T> (&out)[N], const char* name,
233 const FileDatabase& db) const
234{
235 // XXX see if we can reduce this to call to the 'normal' ReadFieldPtr
236 const StreamReaderAny::pos old = db.reader->GetCurrentPos();
237 Pointer ptrval[N];
238 const Field* f;
239 try {
240 f = &(*this)[name];
241
242 // sanity check, should never happen if the genblenddna script is right
243 if ((FieldFlag_Pointer|FieldFlag_Pointer) != (f->flags & (FieldFlag_Pointer|FieldFlag_Pointer))) {
244 throw Error((Formatter::format(),"Field `",name,"` of structure `",
245 this->name,"` ought to be a pointer AND an array"));
246 }
247
248 db.reader->IncPtr(f->offset);
249
250 size_t i = 0;
251 for(; i < std::min(f->array_sizes[0],N); ++i) {
252 Convert(ptrval[i],db);
253 }
254 for(; i < N; ++i) {
255 _defaultInitializer<ErrorPolicy_Igno>()(ptrval[i]);
256 }
257
258 // actually it is meaningless on which Structure the Convert is called
259 // because the `Pointer` argument triggers a special implementation.
260 }
261 catch (const Error& e) {
262 _defaultInitializer<error_policy>()(out,e.what());
263 for(size_t i = 0; i < N; ++i) {
264 out[i].reset();
265 }
266 return false;
267 }
268
269 bool res = true;
270 for(size_t i = 0; i < N; ++i) {
271 // resolve the pointer and load the corresponding structure
272 res = ResolvePointer(out[i],ptrval[i],db,*f) && res;
273 }
274
275 // and recover the previous stream position
276 db.reader->SetCurrentPos(old);
277
278#ifndef ASSIMP_BUILD_BLENDER_NO_STATS
279 ++db.stats().fields_read;
280#endif
281 return res;
282}

312template <int error_policy>
313bool Structure::ReadCustomDataPtr(std::shared_ptr<ElemBase>&out, int cdtype, const char* name, const FileDatabase& db) const {
314
315 const StreamReaderAny::pos old = db.reader->GetCurrentPos();
316
317 Pointer ptrval;
318 const Field* f;
319 try {
320 f = &(*this)[name];
321
322 // sanity check, should never happen if the genblenddna script is right
323 if (!(f->flags & FieldFlag_Pointer)) {
324 throw Error((Formatter::format(), "Field `", name, "` of structure `",
325 this->name, "` ought to be a pointer"));
326 }
327
328 db.reader->IncPtr(f->offset);
329 Convert(ptrval, db);
330 // actually it is meaningless on which Structure the Convert is called
331 // because the `Pointer` argument triggers a special implementation.
332 }
333 catch (const Error& e) {
334 _defaultInitializer<error_policy>()(out, e.what());
335 out.reset();
336 }
337
338 bool readOk = true;
339 if (ptrval.val) {
340 // get block for ptr
341 const FileBlockHead* block = LocateFileBlockForAddress(ptrval, db);
342 db.reader->SetCurrentPos(block->start + static_cast<size_t>((ptrval.val - block->address.val)));
343 // read block->num instances of given type to out
344 readOk = readCustomData(out, cdtype, block->num, db);
345 }
346
347 // and recover the previous stream position
348 db.reader->SetCurrentPos(old);
349
350#ifndef ASSIMP_BUILD_BLENDER_NO_STATS
351 ++db.stats().fields_read;
352#endif
353
354 return readOk;
355}

413template <template <typename> class TOUT, typename T>
414bool Structure :: ResolvePointer(TOUT<T>& out, const Pointer & ptrval, const FileDatabase& db,
415 const Field& f,
416 bool non_recursive /*= false*/) const
417{
418 out.reset(); // ensure null pointers work
419 if (!ptrval.val) {
420 return false;
421 }
422 const Structure& s = db.dna[f.type];
423 // find the file block the pointer is pointing to
424 const FileBlockHead* block = LocateFileBlockForAddress(ptrval,db);
425
426 // also determine the target type from the block header
427 // and check if it matches the type which we expect.
428 const Structure& ss = db.dna[block->dna_index];
429 if (ss != s) {
430 throw Error((Formatter::format(),"Expected target to be of type `",s.name,
431 "` but seemingly it is a `",ss.name,"` instead"
432 ));
433 }
434
435 // try to retrieve the object from the cache
436 db.cache(out).get(s,out,ptrval);
437 if (out) {
438 return true;
439 }
440
441 // seek to this location, but save the previous stream pointer.
442 const StreamReaderAny::pos pold = db.reader->GetCurrentPos();
443 db.reader->SetCurrentPos(block->start+ static_cast<size_t>((ptrval.val - block->address.val) ));
444 // FIXME: basically, this could cause problems with 64 bit pointers on 32 bit systems.
445 // I really ought to improve StreamReader to work with 64 bit indices exclusively.
446
447 // continue conversion after allocating the required storage
448 size_t num = block->size / ss.size;
449 T* o = _allocate(out,num);
450
451 // cache the object before we convert it to avoid cyclic recursion.
452 db.cache(out).set(s,out,ptrval);
453
454 // if the non_recursive flag is set, we don't do anything but leave
455 // the cursor at the correct position to resolve the object.
456 if (!non_recursive) {
457 for (size_t i = 0; i < num; ++i,++o) {
458 s.Convert(*o,db);
459 }
460
461 db.reader->SetCurrentPos(pold);
462 }
463
464#ifndef ASSIMP_BUILD_BLENDER_NO_STATS
465 if(out) {
466 ++db.stats().pointers_resolved;
467 }
468#endif
469 return false;
470}

474inline bool Structure :: ResolvePointer( std::shared_ptr< FileOffset >& out, const Pointer & ptrval,
475 const FileDatabase& db,
476 const Field&,
477 bool) const
478{
479 // Currently used exclusively by PackedFile::data to represent
480 // a simple offset into the mapped BLEND file.
481 out.reset();
482 if (!ptrval.val) {
483 return false;
484 }
485
486 // find the file block the pointer is pointing to
487 const FileBlockHead* block = LocateFileBlockForAddress(ptrval,db);
488
489 out = std::shared_ptr< FileOffset > (new FileOffset());
490 out->val = block->start+ static_cast<size_t>((ptrval.val - block->address.val) );
491 return false;
492}

494//--------------------------------------------------------------------------------
495template <template <typename> class TOUT, typename T>
496bool Structure :: ResolvePointer(vector< TOUT<T> >& out, const Pointer & ptrval,
497 const FileDatabase& db,
498 const Field& f,
499 bool) const
500{
501 // This is a function overload, not a template specialization. According to
502 // the partial ordering rules, it should be selected by the compiler
503 // for array-of-pointer inputs, i.e. Object::mats.
504
505 out.reset();
506 if (!ptrval.val) {
507 return false;
508 }
509
510 // find the file block the pointer is pointing to
511 const FileBlockHead* block = LocateFileBlockForAddress(ptrval,db);
512 const size_t num = block->size / (db.i64bit?8:4);
513
514 // keep the old stream position
515 const StreamReaderAny::pos pold = db.reader->GetCurrentPos();
516 db.reader->SetCurrentPos(block->start+ static_cast<size_t>((ptrval.val - block->address.val) ));
517
518 bool res = false;
519 // allocate raw storage for the array
520 out.resize(num);
521 for (size_t i = 0; i< num; ++i) {
522 Pointer val;
523 Convert(val,db);
524
525 // and resolve the pointees
526 res = ResolvePointer(out[i],val,db,f) && res;
527 }
528
529 db.reader->SetCurrentPos(pold);
530 return res;
531}

533//--------------------------------------------------------------------------------
534template <> bool Structure :: ResolvePointer<std::shared_ptr,ElemBase>(std::shared_ptr<ElemBase>& out,
535 const Pointer & ptrval,
536 const FileDatabase& db,
537 const Field&,
538 bool
539) const
540{
541 // Special case when the data type needs to be determined at runtime.
542 // Less secure than in the `strongly-typed` case.
543
544 out.reset();
545 if (!ptrval.val) {
546 return false;
547 }
548
549 // find the file block the pointer is pointing to
550 const FileBlockHead* block = LocateFileBlockForAddress(ptrval,db);
551
552 // determine the target type from the block header
553 const Structure& s = db.dna[block->dna_index];
554
555 // try to retrieve the object from the cache
556 db.cache(out).get(s,out,ptrval);
557 if (out) {
558 return true;
559 }
560
561 // seek to this location, but save the previous stream pointer.
562 const StreamReaderAny::pos pold = db.reader->GetCurrentPos();
563 db.reader->SetCurrentPos(block->start+ static_cast<size_t>((ptrval.val - block->address.val) ));
564 // FIXME: basically, this could cause problems with 64 bit pointers on 32 bit systems.
565 // I really ought to improve StreamReader to work with 64 bit indices exclusively.
566
567 // continue conversion after allocating the required storage
568 DNA::FactoryPair builders = db.dna.GetBlobToStructureConverter(s,db);
569 if (!builders.first) {
570 // this might happen if DNA::RegisterConverters hasn't been called so far
571 // or if the target type is not contained in `our` DNA.
572 out.reset();
573 ASSIMP_LOG_WARN_F( "Failed to find a converter for the `",s.name,"` structure" );
574 return false;
575 }
576
577 // allocate the object hull
578 out = (s.*builders.first)();
579
580 // cache the object immediately to prevent infinite recursion in a
581 // circular list with a single element (i.e. a self-referencing element).
582 db.cache(out).set(s,out,ptrval);
583
584 // and do the actual conversion
585 (s.*builders.second)(out,db);
586 db.reader->SetCurrentPos(pold);
587
588 // store a pointer to the name string of the actual type
589 // in the object itself. This allows the conversion code
590 // to perform additional type checking.
591 out->dna_type = s.name.c_str();
592
593
594#ifndef ASSIMP_BUILD_BLENDER_NO_STATS
595 ++db.stats().pointers_resolved;
596#endif
597 return false;
598}

600//--------------------------------------------------------------------------------
601const FileBlockHead* Structure :: LocateFileBlockForAddress(const Pointer & ptrval, const FileDatabase& db) const
602{
603 // the file blocks appear in list sorted by
604 // with ascending base addresses so we can run a
605 // binary search to locate the pointer quickly.
606
607 // NOTE: Blender seems to distinguish between side-by-side
608 // data (stored in the same data block) and far pointers,
609 // which are only used for structures starting with an ID.
610 // We don't need to make this distinction, our algorithm
611 // works regardless where the data is stored.
612 vector<FileBlockHead>::const_iterator it = std::lower_bound(db.entries.begin(),db.entries.end(),ptrval);
613 if (it == db.entries.end()) {
614 // this is crucial, pointers may not be invalid.
615 // this is either a corrupted file or an attempted attack.
616 throw DeadlyImportError((Formatter::format(),"Failure resolving pointer 0x",
617 std::hex,ptrval.val,", no file block falls into this address range"
618 ));
619 }
620 if (ptrval.val >= (*it).address.val + (*it).size) {
621 throw DeadlyImportError((Formatter::format(),"Failure resolving pointer 0x",
622 std::hex,ptrval.val,", nearest file block starting at 0x",
623 (*it).address.val," ends at 0x",
624 (*it).address.val + (*it).size
625 ));
626 }
627 return &*it;
628}

630// ------------------------------------------------------------------------------------------------
631// NOTE: The MSVC debugger keeps showing up this annoying `a cast to a smaller data type has
632// caused a loss of data`-warning. Avoid this warning by a masking with an appropriate bitmask.
633
634template <typename T> struct signless;
635template <> struct signless<char> {typedef unsigned char type;};
636template <> struct signless<short> {typedef unsigned short type;};
637template <> struct signless<int> {typedef unsigned int type;};
638template <> struct signless<unsigned char> { typedef unsigned char type; };
639template <typename T>
640struct static_cast_silent {
641 template <typename V>
642 T operator()(V in) {
643 return static_cast<T>(in & static_cast<typename signless<T>::type>(-1));
644 }
645};
646
647template <> struct static_cast_silent<float> {
648 template <typename V> float operator()(V in) {
649 return static_cast<float> (in);
650 }
651};
652
653template <> struct static_cast_silent<double> {
654 template <typename V> double operator()(V in) {
655 return static_cast<double>(in);
656 }
657};
658
659// ------------------------------------------------------------------------------------------------
660template <typename T> inline void ConvertDispatcher(T& out, const Structure& in,const FileDatabase& db)
661{
662 if (in.name == "int") {
663 out = static_cast_silent<T>()(db.reader->GetU4());
664 }
665 else if (in.name == "short") {
666 out = static_cast_silent<T>()(db.reader->GetU2());
667 }
668 else if (in.name == "char") {
669 out = static_cast_silent<T>()(db.reader->GetU1());
670 }
671 else if (in.name == "float") {
672 out = static_cast<T>(db.reader->GetF4());
673 }
674 else if (in.name == "double") {
675 out = static_cast<T>(db.reader->GetF8());
676 }
677 else {
678 throw DeadlyImportError("Unknown source for conversion to primitive data type: "+in.name);
679 }
680}

740template <> inline void Structure :: Convert<float> (float& dest,const FileDatabase& db) const
