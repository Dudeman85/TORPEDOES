26 void Init()
27 {
28 //The default 3D model shader with bling-phong lighting
29 defaultShader = new Shader(
30 R"(
31 #version 330 core
32 layout(location = 0) in vec3 aPos;
33 layout(location = 1) in vec3 aNormal;
34 layout(location = 2) in vec2 aTexCoords;
35
36 out vec3 FragPos;
37 out vec3 Normal;
38 out vec2 TexCoords;
39
40 uniform mat4 model;
41 uniform mat4 view;
42 uniform mat4 projection;
43
44 void main()
45 {
46 TexCoords = aTexCoords;
47 FragPos = vec3 (model * vec4(aPos,1.0));
48 Normal = mat3(transpose(inverse(model))) *aNormal;
49
50 gl_Position = projection * view * model * vec4(aPos, 1.0);
51 }
52 )",
53 R"(
54 #version 330 core
55 out vec4 FragColor;
56
57 in vec2 TexCoords;
58 in vec3 Normal;
59 in vec3 FragPos;
60
61 uniform sampler2D texture_diffuse1;
62 uniform vec3 lightPos;
63 uniform vec3 lightColor;
64 uniform vec3 viewPos;
65
66 void main()
67 {
68 vec3 objectColor = texture(texture_diffuse1, TexCoords).xyz;
69 float ambientStrength = 0.3;
70 vec3 ambient = ambientStrength * lightColor;
71
72 vec3 norm = normalize(Normal);
73 vec3 lightDir = normalize(lightPos - FragPos);
74 float diff = max(dot(norm , lightDir), 0.0);
75 vec3 diffuse = diff * lightColor;
76
77 float specularStrength = 0.5;
78 vec3 viewDir = normalize(viewPos - FragPos);
79 vec3 reflectDir = reflect(-lightDir, norm);
80 float spec = pow(max(dot(viewDir, reflectDir), 0.0), 32);
81 vec3 specular = specularStrength * spec * lightColor;
82
83 vec3 result = (ambient + diffuse + specular ) * objectColor;
84 FragColor = vec4(result, 1.0);
85 }
86 )", false);
87 }
89 void Update(Camera* cam)
90 {
91 //For each entity
92 for (auto itr = entities.begin(); itr != entities.end();)
93 {
94 ecs::Entity entity = *itr++;
95
96 //Get relevant components
97 Transform& transform = ecs::GetComponent<Transform>(entity);
98 ModelRenderer& modelRenderer = ecs::GetComponent<ModelRenderer>(entity);
99
100 //If a shader has been specified for this sprite use it, else use the default
101 Shader* shader = defaultShader;
102 if (modelRenderer.shader)
103 shader = modelRenderer.shader;
104 shader->use();
105
106 //Create the model matrix, this is the same for each mesh so it only needs to be done once
107 glm::mat4 model = TransformSystem::GetGlobalTransformMatrix(entity);
108
109 //Model matrix
110 unsigned int modelLoc = glGetUniformLocation(shader->ID, "model");
111 glUniformMatrix4fv(modelLoc, 1, GL_FALSE, glm::value_ptr(model));
112 //View matrix
113 unsigned int viewLoc = glGetUniformLocation(shader->ID, "view");
114 glUniformMatrix4fv(viewLoc, 1, GL_FALSE, glm::value_ptr(cam->GetViewMatrix()));
115 //Projection matrix
116 unsigned int projLoc = glGetUniformLocation(shader->ID, "projection");
117 glUniformMatrix4fv(projLoc, 1, GL_FALSE, glm::value_ptr(cam->GetProjectionMatrix()));
118 //Light Color
119 unsigned int colorLoc = glGetUniformLocation(shader->ID, "lightColor");
120 glUniform3f(colorLoc, lightColor.x / 255, lightColor.y / 255, lightColor.z / 255);
121 //Light Position
122 unsigned int lightPosLoc = glGetUniformLocation(shader->ID, "lightPos");
123 glUniform3f(lightPosLoc, lightPos.x, lightPos.y, lightPos.z);
124 // camara pposition
125 unsigned int viewPosLoc = glGetUniformLocation(shader->ID, "viewPos");
126 glUniform3fv(viewPosLoc, 1, glm::value_ptr(cam->position));
127
128
129 //For each mesh in the model
130 for (unsigned int i = 0; i < modelRenderer.model->meshes.size(); i++)
131 {
132 Mesh mesh = modelRenderer.model->meshes[i];
133
134 //Texture uniforms are named: uniform sampler2D texture_diffuseN, or texture_specularN
135 //We can support up to 8 textures which have to be defined in the shader
136 unsigned int diffuseNr = 1;
137 unsigned int specularNr = 1;
138
139 //For each Texture in the mesh
140 for (unsigned int i = 0; i < mesh.textures.size(); i++)
141 {
142 //Activate proper texture unit before binding
143 glActiveTexture(GL_TEXTURE0 + i);
144
145 //Retrieve texture number and type (the N in texture_{type}N)
146 std::string number;
147 std::string name = mesh.textures[i]->type;
148 if (name == "texture_diffuse")
149 number = std::to_string(diffuseNr++);
150 else if (name == "texture_specular")
151 number = std::to_string(specularNr++);
152
153 //Set the uniform for the material texture
154 glUniform1i(glGetUniformLocation(shader->ID, (name + number).c_str()), i);
155
156 glBindTexture(GL_TEXTURE_2D, mesh.textures[i]->ID());
157 }
158
159 //Unbind texture
160 glActiveTexture(GL_TEXTURE0);
161
162 //Draw mesh
163 glBindVertexArray(mesh.VAO);
164 glDrawElements(GL_TRIANGLES, mesh.indices.size(), GL_UNSIGNED_INT, 0);
165 glBindVertexArray(0);
166 }
167 }
168 }
