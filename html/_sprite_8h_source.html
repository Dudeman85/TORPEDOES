161 void Update(Camera* cam)
162 {
163 //Sort the entities and tilemap by Z
164 std::set<float> layersToDraw;
165 std::map<float, std::vector<ecs::Entity>> sortedEntities;
166 if (tilemap)
167 layersToDraw.insert(tilemap->zLayers.begin(), tilemap->zLayers.end());
168
169 //UI elements are sorted seperately
170 std::set<float> uiLayersToDraw;
171 std::map<float, std::vector<ecs::Entity>> sortedUIElements;
172
173 //Sort the entities into sprite and UI layers
174 for (auto itr = entities.begin(); itr != entities.end();)
175 {
176 ecs::Entity entity = *itr++;
177 Transform& transform = ecs::GetComponent<Transform>(entity);
178 SpriteRenderer& renderer = ecs::GetComponent<SpriteRenderer>(entity);
179
180 //Seperate sprites and UI elements
181 if (!renderer.uiElement)
182 {
183 sortedEntities[transform.position.z].push_back(entity);
184 layersToDraw.insert(transform.position.z);
185 }
186 else
187 {
188 sortedUIElements[transform.position.z].push_back(entity);
189 uiLayersToDraw.insert(transform.position.z);
190 }
191 }
192
193 //Draw all sprites and tilemap by layer
194 for (const float& layer : layersToDraw)
195 {
196 if (tilemap)
197 tilemap->draw(layer);
198
199 //Bind the right VAO after tilemap
200 glBindVertexArray(VAO);
201
202 //Draw entities for this layer
203 for (const ecs::Entity& entity : sortedEntities[layer])
204 {
205 DrawEntity(entity, cam);
206 }
207 }
208
209 //Draw all UI elements by layer
210 for (const float& layer : uiLayersToDraw)
211 {
212 //Draw entities for this layer
213 for (const ecs::Entity& entity : sortedUIElements[layer])
214 {
215 DrawEntity(entity, cam);
216 }
217 }
218
219 //Unbind vertex array
220 glBindVertexArray(0);

223 //Draw an entity to the screen
224 void DrawEntity(ecs::Entity entity, Camera* cam)
225 {
226 //Get relevant components
227 SpriteRenderer& sprite = ecs::GetComponent<SpriteRenderer>(entity);
228 Transform& transform = ecs::GetComponent<Transform>(entity);
229
230 if (!sprite.enabled)
231 return;
232
233 //If a shader has been specified for this sprite use it, else use the default
234 Shader* shader = defaultShader;
235 if (sprite.shader)
236 shader = sprite.shader;
237 shader->use();
238
239 //Create the model matrix
240 glm::mat4 model = TransformSystem::GetGlobalTransformMatrix(entity);
241
242 //Give the shader the model matrix
243 unsigned int modelLoc = glGetUniformLocation(shader->ID, "model");
244 glUniformMatrix4fv(modelLoc, 1, GL_FALSE, glm::value_ptr(model));
245 //Get the view and projection locations
246 unsigned int projLoc = glGetUniformLocation(shader->ID, "projection");
247 unsigned int viewLoc = glGetUniformLocation(shader->ID, "view");
248
249 if (!sprite.uiElement)
250 {
251 //Give the shader the camera's view matrix
252 glUniformMatrix4fv(viewLoc, 1, GL_FALSE, glm::value_ptr(cam->GetViewMatrix()));
253
254 //Give the shader the camera's projection matrix
255 glUniformMatrix4fv(projLoc, 1, GL_FALSE, glm::value_ptr(cam->GetProjectionMatrix()));
256 }
257 else
258 {
259 //Clear the depth buffer to always draw UI elements on top
260 glClear(GL_DEPTH_BUFFER_BIT);
261
262 //Give the shader a constant view matrix
263 glUniformMatrix4fv(viewLoc, 1, GL_FALSE, glm::value_ptr(glm::mat4(1.f)));
264
265 //Give the shader a constant projection matrix
266 glUniformMatrix4fv(projLoc, 1, GL_FALSE, glm::value_ptr(glm::mat4(1.f)));
267 }
268
269 //Bind the texture
270 glActiveTexture(GL_TEXTURE0);
271 if (sprite.texture)
272 sprite.texture->Use();
273
274 //Draw the sprite
275 glDrawElements(GL_TRIANGLES, 6, GL_UNSIGNED_INT, 0);
276
277 //Unbind the texture
278 glBindTexture(GL_TEXTURE_2D, 0);
279 }

332 static void AdvanceFrame(ecs::Entity entity)
333 {
334 //Get the relevant components from entity
335 Animator& animator = ecs::GetComponent<Animator>(entity);
336 SpriteRenderer& sprite = ecs::GetComponent<SpriteRenderer>(entity);
337
338 //Change Sprites texture
339 sprite.texture = animator.animations[animator.currentAnimation].textures[animator.animationFrame];
340
341 animator.animationFrame++;
342 animator.animationTimer = 0;
343
344 //If end of animation has been reached go to start or end animation
345 if (animator.animationFrame >= animator.animations[animator.currentAnimation].length)

349
350 //End the animation if it is not set to repeat

353 animator.playingAnimation = false;
354 animator.currentAnimation = "";

356 }
