208// This section was added by K. Mammou
209 void ExpGolombEncode(unsigned int symbol,
210 int k,
211 Static_Bit_Model & bModel0,
212 Adaptive_Bit_Model & bModel1)
213 {
214 while(1)
215 {
216 if (symbol >= (unsigned int)(1<<k))
217 {
218 encode(1, bModel1);
219 symbol = symbol - (1<<k);
220 k++;
221 }
222 else
223 {
224 encode(0, bModel1); // now terminated zero of unary part
225 while (k--) // next binary part
226 {
227 encode((signed short)((symbol>>k)&1), bModel0);
228 }
229 break;
230 }
231 }
232 }

234
235 unsigned ExpGolombDecode(int k,
236 Static_Bit_Model & bModel0,
237 Adaptive_Bit_Model & bModel1)
238 {
239 unsigned int l;
240 int symbol = 0;
241 int binary_symbol = 0;
242 do
243 {
244 l=decode(bModel1);
245 if (l==1)
246 {
247 symbol += (1<<k);
248 k++;
249 }
250 }
251 while (l!=0);
252 while (k--) //next binary part
253 if (decode(bModel0)==1)
254 {
255 binary_symbol |= (1<<k);
256 }
257 return (unsigned int) (symbol+binary_symbol);
258 }
