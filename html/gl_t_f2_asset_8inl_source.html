205template<class T>
206unsigned int LazyDict<T>::Remove(const char* id)
207{
208 id = T::TranslateId(mAsset, id);
209
210 typename IdDict::iterator it = mObjsById.find(id);
211
212 if (it == mObjsById.end()) {
213 throw DeadlyExportError("GLTF: Object with id \"" + std::string(id) + "\" is not found");
214 }
215
216 const unsigned int index = it->second;
217
218 mAsset.mUsedIds[id] = false;
219 mObjsById.erase(id);
220 mObjsByOIndex.erase(index);
221 mObjs.erase(mObjs.begin() + index);
222
223 //update index of object in mObjs;
224 for (unsigned int i = index; i < mObjs.size(); ++i) {
225 T *obj = mObjs[i];
226
227 obj->index = i;
228 }
229
230 for (IdDict::iterator it = mObjsById.begin(); it != mObjsById.end(); ++it) {
231 if (it->second <= index) {
232 continue;
233 }
234
235 mObjsById[it->first] = it->second - 1;
236 }
237
238 for (Dict::iterator it = mObjsByOIndex.begin(); it != mObjsByOIndex.end(); ++it) {
239 if (it->second <= index) {
240 continue;
241 }
242
243 mObjsByOIndex[it->first] = it->second - 1;
244 }
245
246 return index;
247}

249template<class T>
250Ref<T> LazyDict<T>::Retrieve(unsigned int i)
251{
252
253 typename Dict::iterator it = mObjsByOIndex.find(i);
254 if (it != mObjsByOIndex.end()) {// already created?
255 return Ref<T>(mObjs, it->second);
256 }
257
258 // read it from the JSON object
259 if (!mDict) {
260 throw DeadlyImportError("GLTF: Missing section \"" + std::string(mDictId) + "\"");
261 }
262
263 if (!mDict->IsArray()) {
264 throw DeadlyImportError("GLTF: Field is not an array \"" + std::string(mDictId) + "\"");
265 }
266
267 Value &obj = (*mDict)[i];
268
269 if (!obj.IsObject()) {
270 throw DeadlyImportError("GLTF: Object at index \"" + to_string(i) + "\" is not a JSON object");
271 }
272
273 T* inst = new T();
274 inst->id = std::string(mDictId) + "_" + to_string(i);
275 inst->oIndex = i;
276 ReadMember(obj, "name", inst->name);
277 inst->Read(obj, mAsset);
278
279 return Add(inst);
280}

303{
304 unsigned int idx = unsigned(mObjs.size());
305 mObjs.push_back(obj);

321 inst->id = id;
322 inst->index = idx;
323 inst->oIndex = idx;

347inline void Buffer::Read(Value& obj, Asset& r)
348{
349 size_t statedLength = MemberOrDefault<size_t>(obj, "byteLength", 0);
350 byteLength = statedLength;
351
352 Value* it = FindString(obj, "uri");
353 if (!it) {
354 if (statedLength > 0) {
355 throw DeadlyImportError("GLTF: buffer with non-zero length missing the \"uri\" attribute");
356 }
357 return;
358 }
359
360 const char* uri = it->GetString();
361
362 glTFCommon::Util::DataURI dataURI;
363 if (ParseDataURI(uri, it->GetStringLength(), dataURI)) {
364 if (dataURI.base64) {
365 uint8_t* data = 0;
366 this->byteLength = glTFCommon::Util::DecodeBase64(dataURI.data, dataURI.dataLength, data);
367 this->mData.reset(data, std::default_delete<uint8_t[]>());
368
369 if (statedLength > 0 && this->byteLength != statedLength) {
370 throw DeadlyImportError("GLTF: buffer \"" + id + "\", expected " + to_string(statedLength) +
371 " bytes, but found " + to_string(dataURI.dataLength));
372 }
373 }
374 else { // assume raw data
375 if (statedLength != dataURI.dataLength) {
376 throw DeadlyImportError("GLTF: buffer \"" + id + "\", expected " + to_string(statedLength) +
377 " bytes, but found " + to_string(dataURI.dataLength));
378 }
379
380 this->mData.reset(new uint8_t[dataURI.dataLength], std::default_delete<uint8_t[]>());
381 memcpy( this->mData.get(), dataURI.data, dataURI.dataLength );
382 }
383 }
384 else { // Local file
385 if (byteLength > 0) {
386 std::string dir = !r.mCurrentAssetDir.empty() ? (r.mCurrentAssetDir + "/") : "";
387
388 IOStream* file = r.OpenFile(dir + uri, "rb");
389 if (file) {
390 bool ok = LoadFromStream(*file, byteLength);
391 delete file;
392
393 if (!ok)
394 throw DeadlyImportError("GLTF: error while reading referenced file \"" + std::string(uri) + "\"" );
395 }
396 else {
397 throw DeadlyImportError("GLTF: could not open referenced file \"" + std::string(uri) + "\"");
398 }
399 }
400 }
401}

403inline bool Buffer::LoadFromStream(IOStream& stream, size_t length, size_t baseOffset)
404{
405 byteLength = length ? length : stream.FileSize();
406
407 if (baseOffset) {
408 stream.Seek(baseOffset, aiOrigin_SET);
409 }
410
411 mData.reset(new uint8_t[byteLength], std::default_delete<uint8_t[]>());
412
413 if (stream.Read(mData.get(), byteLength, 1) != 1) {
414 return false;
415 }
416 return true;
417}

419inline void Buffer::EncodedRegion_Mark(const size_t pOffset, const size_t pEncodedData_Length, uint8_t* pDecodedData, const size_t pDecodedData_Length, const std::string& pID)
420{
421 // Check pointer to data
422 if(pDecodedData == nullptr) throw DeadlyImportError("GLTF: for marking encoded region pointer to decoded data must be provided.");
423
424 // Check offset
425 if(pOffset > byteLength)
426 {
427 const uint8_t val_size = 32;
428
429 char val[val_size];
430
431 ai_snprintf(val, val_size, "%llu", (long long)pOffset);
432 throw DeadlyImportError(std::string("GLTF: incorrect offset value (") + val + ") for marking encoded region.");
433 }
434
435 // Check length
436 if((pOffset + pEncodedData_Length) > byteLength)
437 {
438 const uint8_t val_size = 64;
439
440 char val[val_size];
441
442 ai_snprintf(val, val_size, "%llu, %llu", (long long)pOffset, (long long)pEncodedData_Length);
443 throw DeadlyImportError(std::string("GLTF: encoded region with offset/length (") + val + ") is out of range.");
444 }
445
446 // Add new region
447 EncodedRegion_List.push_back(new SEncodedRegion(pOffset, pEncodedData_Length, pDecodedData, pDecodedData_Length, pID));
448 // And set new value for "byteLength"
449 byteLength += (pDecodedData_Length - pEncodedData_Length);
450}

470inline
471bool Buffer::ReplaceData(const size_t pBufferData_Offset, const size_t pBufferData_Count, const uint8_t* pReplace_Data, const size_t pReplace_Count)
472{
473
474 if((pBufferData_Count == 0) || (pReplace_Count == 0) || (pReplace_Data == nullptr)) {
475 return false;
476 }
477
478 const size_t new_data_size = byteLength + pReplace_Count - pBufferData_Count;
479 uint8_t *new_data = new uint8_t[new_data_size];
480 // Copy data which place before replacing part.
481 ::memcpy(new_data, mData.get(), pBufferData_Offset);
482 // Copy new data.
483 ::memcpy(&new_data[pBufferData_Offset], pReplace_Data, pReplace_Count);
484 // Copy data which place after replacing part.
485 ::memcpy(&new_data[pBufferData_Offset + pReplace_Count], &mData.get()[pBufferData_Offset + pBufferData_Count], pBufferData_Offset);
486 // Apply new data
487 mData.reset(new_data, std::default_delete<uint8_t[]>());
488 byteLength = new_data_size;
489
490 return true;
491}

526inline void Buffer::Grow(size_t amount)
527{
528 if (amount <= 0) return;
529 if (capacity >= byteLength + amount)
530 {
531 byteLength += amount;
532 return;
533 }
534
535 // Shift operation is standard way to divide integer by 2, it doesn't cast it to float back and forth, also works for odd numbers,
536 // originally it would look like: static_cast<size_t>(capacity * 1.5f)
537 capacity = std::max(capacity + (capacity >> 1), byteLength + amount);
538
539 uint8_t* b = new uint8_t[capacity];
540 if (mData) memcpy(b, mData.get(), byteLength);
541 mData.reset(b, std::default_delete<uint8_t[]>());
542 byteLength += amount;
543}

617 inline void CopyData(size_t count,
618 const uint8_t* src, size_t src_stride,
619 uint8_t* dst, size_t dst_stride)
620 {
621 if (src_stride == dst_stride) {
622 memcpy(dst, src, count * src_stride);
623 }
624 else {
625 size_t sz = std::min(src_stride, dst_stride);
626 for (size_t i = 0; i < count; ++i) {
627 memcpy(dst, src, sz);
628 if (sz < dst_stride) {
629 memset(dst + sz, 0, dst_stride - sz);
630 }
631 src += src_stride;
632 dst += dst_stride;
633 }
634 }
635 }

638template<class T>
639bool Accessor::ExtractData(T*& outData)
640{
641 uint8_t* data = GetPointer();
642 if (!data) return false;
643
644 const size_t elemSize = GetElementSize();
645 const size_t totalSize = elemSize * count;
646
647 const size_t stride = bufferView && bufferView->byteStride ? bufferView->byteStride : elemSize;
648
649 const size_t targetElemSize = sizeof(T);
650 ai_assert(elemSize <= targetElemSize);
651
652 ai_assert(count*stride <= bufferView->byteLength);
653
654 outData = new T[count];
655 if (stride == elemSize && targetElemSize == elemSize) {
656 memcpy(outData, data, totalSize);
657 }
658 else {
659 for (size_t i = 0; i < count; ++i) {
660 memcpy(outData + i, data + i*stride, elemSize);
661 }
662 }
663
664 return true;
665}

667inline void Accessor::WriteData(size_t count, const void* src_buffer, size_t src_stride)
668{
669 uint8_t* buffer_ptr = bufferView->buffer->GetPointer();
670 size_t offset = byteOffset + bufferView->byteOffset;

673

675 uint8_t* dst = reinterpret_cast< uint8_t*>(buffer_ptr + offset);
676
677 ai_assert(dst + count*dst_stride <= buffer_ptr + bufferView->buffer->byteLength);
678 CopyData(count, src, src_stride, dst, dst_stride);
679}

725 }

729 }
730 }
731 else if (Value* bufferViewVal = FindUInt(obj, "bufferView")) {

1281inline void Asset::ReadBinaryHeader(IOStream& stream, std::vector<char>& sceneData)
1282{
1283 GLB_Header header;
1284 if (stream.Read(&header, sizeof(header), 1) != 1) {
1285 throw DeadlyImportError("GLTF: Unable to read the file header");
1286 }
1287
1288 if (strncmp((char*)header.magic, AI_GLB_MAGIC_NUMBER, sizeof(header.magic)) != 0) {
1289 throw DeadlyImportError("GLTF: Invalid binary glTF file");
1290 }
1291
1292 AI_SWAP4(header.version);
1293 asset.version = to_string(header.version);
1294 if (header.version != 2) {
1295 throw DeadlyImportError("GLTF: Unsupported binary glTF version");
1296 }
1297
1298 GLB_Chunk chunk;
1299 if (stream.Read(&chunk, sizeof(chunk), 1) != 1) {
1300 throw DeadlyImportError("GLTF: Unable to read JSON chunk");
1301 }
1302
1303 AI_SWAP4(chunk.chunkLength);
1304 AI_SWAP4(chunk.chunkType);
1305
1306 if (chunk.chunkType != ChunkType_JSON) {
1307 throw DeadlyImportError("GLTF: JSON chunk missing");
1308 }
1309
1310 // read the scene data
1311
1312 mSceneLength = chunk.chunkLength;
1313 sceneData.resize(mSceneLength + 1);
1314 sceneData[mSceneLength] = '\0';
1315
1316 if (stream.Read(&sceneData[0], 1, mSceneLength) != mSceneLength) {
1317 throw DeadlyImportError("GLTF: Could not read the file contents");
1318 }
1319
1320 uint32_t padding = ((chunk.chunkLength + 3) & ~3) - chunk.chunkLength;
1321 if (padding > 0) {
1322 stream.Seek(padding, aiOrigin_CUR);
1323 }
1324
1325 AI_SWAP4(header.length);
1326 mBodyOffset = 12 + 8 + chunk.chunkLength + padding + 8;
1327 if (header.length >= mBodyOffset) {
1328 if (stream.Read(&chunk, sizeof(chunk), 1) != 1) {
1329 throw DeadlyImportError("GLTF: Unable to read BIN chunk");
1330 }
1331
1332 AI_SWAP4(chunk.chunkLength);
1333 AI_SWAP4(chunk.chunkType);
1334
1335 if (chunk.chunkType != ChunkType_BIN) {
1336 throw DeadlyImportError("GLTF: BIN chunk missing");
1337 }
1338
1339 mBodyLength = chunk.chunkLength;
1340 }
1341 else {
1342 mBodyOffset = mBodyLength = 0;
1343 }

1346inline void Asset::Load(const std::string& pFile, bool isBinary)
1347{
1348 mCurrentAssetDir.clear();
1349 int pos = std::max(int(pFile.rfind('/')), int(pFile.rfind('\\')));
1350 if (pos != int(std::string::npos)) mCurrentAssetDir = pFile.substr(0, pos + 1);
1351
1352 shared_ptr<IOStream> stream(OpenFile(pFile.c_str(), "rb", true));
1353 if (!stream) {
1354 throw DeadlyImportError("GLTF: Could not open file for reading");
1355 }
1356
1357 // is binary? then read the header
1358 std::vector<char> sceneData;
1359 if (isBinary) {
1360 SetAsBinary(); // also creates the body buffer
1361 ReadBinaryHeader(*stream, sceneData);
1362 }
1363 else {
1364 mSceneLength = stream->FileSize();
1365 mBodyLength = 0;
1366
1367
1368 // read the scene data
1369
1370 sceneData.resize(mSceneLength + 1);
1371 sceneData[mSceneLength] = '\0';
1372
1373 if (stream->Read(&sceneData[0], 1, mSceneLength) != mSceneLength) {
1374 throw DeadlyImportError("GLTF: Could not read the file contents");
1375 }
1376 }
1377
1378
1379 // parse the JSON document
1380
1381 Document doc;
1382 doc.ParseInsitu(&sceneData[0]);
1383
