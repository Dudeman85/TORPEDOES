156 struct DistanceSorter {
157
158 DistanceSorter(const IfcVector3& base) : base(base) {}
159
160 bool operator () (const TempOpening& a, const TempOpening& b) const {

232struct FuzzyVectorCompare {
233
234 FuzzyVectorCompare(IfcFloat epsilon) : epsilon(epsilon) {}
235 bool operator()(const IfcVector3& a, const IfcVector3& b) {
236 return std::abs((a-b).SquareLength()) < epsilon;
237 }

243// ------------------------------------------------------------------------------------------------
244// Ordering predicate to totally order R^2 vectors first by x and then by y
245// ------------------------------------------------------------------------------------------------
246struct XYSorter {
247
248 // sort first by X coordinates, then by Y coordinates
249 bool operator () (const IfcVector2&a, const IfcVector2& b) const {
250 if (a.x == b.x) {
251 return a.y < b.y;
252 }
253 return a.x < b.x;
254 }
