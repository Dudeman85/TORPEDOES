74 static int SimpleSolveCollision(Collision collision)
75 {
76 //One of the entities does not have a rigidbody. Return <0 on failure;
77 if (!ecs::HasComponent<Rigidbody>(collision.a) || !ecs::HasComponent<Rigidbody>(collision.b))
78 return -1;
79 //Nothing needs to be done. Return >0 on trigger
80 if (collision.type == Collision::Type::trigger || collision.type == Collision::Type::tilemapTrigger)
81 return 1;
82 //Nothing needs to be done. Return 0 on success
83 if (collision.type == Collision::Type::miss)
84 return 0;
85
86 Rigidbody& rba = ecs::GetComponent<Rigidbody>(collision.a);
87
88 //Entity collision
89 if (collision.type == Collision::Type::collision)
90 {
91 Rigidbody& rbb = ecs::GetComponent<Rigidbody>(collision.b);
92
93 //Calculate new velocities
94 float j = -(1 + (rba.restitution + rbb.restitution) / 2) * (rba.velocity - rbb.velocity).Dot(collision.normal) / collision.normal.Dot(collision.normal * (1 / rba.mass + 1 / rbb.mass));
95 rba.velocity = rba.velocity + collision.normal * (j / rba.mass);
96 rbb.velocity = rbb.velocity - collision.normal * (j / rbb.mass);
97
98 //Move the entity by the minimum translation vector to make sure it is out of the collision
99 TransformSystem::Translate(collision.a, collision.mtv);
100
101 return 0;
102 }

107 //Solve a collision between two colliders
108 //Returns 0 on success, >0 on trigger, and <0 on failure
109 static int SolveTilemapCollision(std::vector<Collision> collisions)
110 {
111 //No collision, nothing needs to be done
112 if (collisions.empty())
113 return 0;
114
115 //Nothing needs to be done. Return >0 on trigger
116 if (collisions.front().type == Collision::Type::trigger || collisions.front().type == Collision::Type::tilemapTrigger)
117 return 1;
118 //Nothing needs to be done. Return 0 on success
119 if (collisions.front().type == Collision::Type::miss)
120 return 0;
121
122 ecs::Entity a = collisions.front().a;
123 Collision maxIntersect = Collision{.mtv = Vector2(0)};
124
125 int i = 0;
126 do
127 {
128 //Find the largest collision
129 for (Collision& collision : collisions)
130 {
131 if (collision.mtv.Length() > maxIntersect.mtv.Length())
132 {
133 if (tileProperties.count(collision.b) != 0)
134 {
135 if (!tileProperties[collision.b].trigger)
136 maxIntersect = collision;
137 }
138 else
139 {
140 maxIntersect = collision;
141 }
142 }
143 }
144
145 //Solve the largest collision
146 if (maxIntersect.type == Collision::Type::tilemapCollision)
147 {
148 TransformSystem::Translate(a, maxIntersect.mtv);
149 }
150
151 //Check collision again
152 collisions = ecs::GetSystem<CollisionSystem>()->CheckTilemapCollision(a);
153
154 i++;
155 } while (collisions.size() > 0 && i < tilemapIterationLimit);
156
157 //Calculate new velocities
158 Rigidbody& rba = ecs::GetComponent<Rigidbody>(a);
159 Vector3 newAVeclocity = rba.velocity - maxIntersect.normal * 2 * rba.velocity.Dot(maxIntersect.normal);
160 rba.velocity = newAVeclocity * rba.restitution;
161
162 return 0;
163 }
