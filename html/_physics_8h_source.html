74 static int SimpleSolveCollision(Collision collision)
75 {
76 //One of the entities does not have a rigidbody. Return <0 on failure;
77 if (!ecs::HasComponent<Rigidbody>(collision.a) || !ecs::HasComponent<Rigidbody>(collision.b))
78 return -1;
79 //Nothing needs to be done. Return >0 on trigger
80 if (collision.type == Collision::Type::trigger || collision.type == Collision::Type::tilemapTrigger)
81 return 1;
82 //Nothing needs to be done. Return 0 on success
83 if (collision.type == Collision::Type::miss)
84 return 0;
85
86 Rigidbody& rba = ecs::GetComponent<Rigidbody>(collision.a);
87
88 //Entity collision
89 if (collision.type == Collision::Type::collision)
90 {
91 Rigidbody& rbb = ecs::GetComponent<Rigidbody>(collision.b);
92
93 //Calculate new velocities
94 float j = -(1 + (rba.restitution + rbb.restitution) / 2) * (rba.velocity - rbb.velocity).Dot(collision.normal) / collision.normal.Dot(collision.normal * (1 / rba.mass + 1 / rbb.mass));
95 rba.velocity = rba.velocity + collision.normal * (j / rba.mass);
96 rbb.velocity = rbb.velocity - collision.normal * (j / rbb.mass);
97
98 //Move the entity by the minimum translation vector to make sure it is out of the collision
99 TransformSystem::Translate(collision.a, collision.mtv);
100
101 return 0;
102 }
103
104 return -1;
105 }

108 static int SolveTilemapCollision(std::vector<Collision> collisions)
109 {
110 //No collision, nothing needs to be done
111 if (collisions.empty())
112 return 0;
113
114 //Nothing needs to be done. Return >0 on trigger
115 if (collisions.front().type == Collision::Type::trigger || collisions.front().type == Collision::Type::tilemapTrigger)
116 return 1;
117 //Nothing needs to be done. Return 0 on success
118 if (collisions.front().type == Collision::Type::miss)
119 return 0;
120
121 ecs::Entity a = collisions.front().a;
122 Collision maxIntersect = Collision{.mtv = Vector2(0)};
123
124 int i = 0;
125 do
126 {
127 //Find the largest collision
128 for (Collision& collision : collisions)
129 {
130 if (collision.mtv.Length() > maxIntersect.mtv.Length())
131 {
132 if (tileProperties.count(collision.b) != 0)
133 {
134 if (!tileProperties[collision.b].trigger)
135 maxIntersect = collision;
136 }
137 else
138 {
139 maxIntersect = collision;
140 }
141 }
142 }
143
144 //Solve the largest collision
145 if (maxIntersect.type == Collision::Type::tilemapCollision)
146 {
147 TransformSystem::Translate(a, maxIntersect.mtv);
148 }
149
150 //Check collision again
151 collisions = ecs::GetSystem<CollisionSystem>()->CheckTilemapCollision(a);
152
153 i++;
154 } while (collisions.size() > 0 && i < tilemapIterationLimit);
155
156 //Calculate new velocities
157 Rigidbody& rba = ecs::GetComponent<Rigidbody>(a);
158 Vector3 newAVeclocity = rba.velocity - maxIntersect.normal * 2 * rba.velocity.Dot(maxIntersect.normal);
159 rba.velocity = newAVeclocity * rba.restitution;
160
161 return 0;
162 }
