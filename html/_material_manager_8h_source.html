122 // Diffuse-only materials are ignored.

140

149 //------------------------------------------------------------------
150 // Reset the state of the class
151 // Called whenever a new asset is loaded
152 inline void Reset()
153 {
154 this->m_iShaderCount = 0;
155 for( TextureCache::iterator it = sCachedTextures.begin(); it != sCachedTextures.end(); ++it ) {
156 ( *it ).second->Release();
157 }
158 sCachedTextures.clear();
159 }

171 //------------------------------------------------------------------

182 void HMtoNMIfNecessary( IDirect3DTexture9* piTexture,
183 IDirect3DTexture9** piTextureOut,
184 bool bWasOriginallyHM = true );
185

187 // Search for non-opaque pixels in a texture

190 // less than 255
191 //------------------------------------------------------------------
192 bool HasAlphaPixels( IDirect3DTexture9* piTexture );
193
194 private:
195
