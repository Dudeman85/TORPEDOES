111 /**************************************************************************
112 *
113 * Return the vertical metrics in font units for a given glyph.
114 * See function `tt_loader_set_pp' below for explanations.
115 */
116 FT_LOCAL_DEF( void )
117 TT_Get_VMetrics( TT_Face face,
118 FT_UInt idx,
119 FT_Pos yMax,
120 FT_Short* tsb,
121 FT_UShort* ah )
122 {
123 if ( face->vertical_info )
124 ( (SFNT_Service)face->sfnt )->get_metrics( face, 1, idx, tsb, ah );
125
126 else if ( face->os2.version != 0xFFFFU )
127 {
128 *tsb = (FT_Short)( face->os2.sTypoAscender - yMax );
129 *ah = (FT_UShort)FT_ABS( face->os2.sTypoAscender -
130 face->os2.sTypoDescender );
131 }
132
133 else
134 {
135 *tsb = (FT_Short)( face->horizontal.Ascender - yMax );
136 *ah = (FT_UShort)FT_ABS( face->horizontal.Ascender -
137 face->horizontal.Descender );
138 }
139
140#ifdef FT_DEBUG_LEVEL_TRACE
141 if ( !face->vertical_info )
142 FT_TRACE5(( " [vertical metrics missing, computing values]\n" ));
143#endif
144
145 FT_TRACE5(( " advance height (font units): %d\n", *ah ));
146 FT_TRACE5(( " top side bearing (font units): %d\n", *tsb ));
147 }

150 static FT_Error
151 tt_get_metrics( TT_Loader loader,
152 FT_UInt glyph_index )
153 {
154 TT_Face face = loader->face;
155#ifdef TT_SUPPORT_SUBPIXEL_HINTING_INFINALITY
156 TT_Driver driver = (TT_Driver)FT_FACE_DRIVER( face );
157#endif
158
159 FT_Error error;
160 FT_Stream stream = loader->stream;
161
162 FT_Short left_bearing = 0, top_bearing = 0;
163 FT_UShort advance_width = 0, advance_height = 0;
164
165 /* we must preserve the stream position */
166 /* (which gets altered by the metrics functions) */
167 FT_ULong pos = FT_STREAM_POS();
168
169
170 TT_Get_HMetrics( face, glyph_index,
171 &left_bearing,
172 &advance_width );
173 TT_Get_VMetrics( face, glyph_index,
174 loader->bbox.yMax,
175 &top_bearing,
176 &advance_height );
177
178 if ( FT_STREAM_SEEK( pos ) )
179 return error;
180
181 loader->left_bearing = left_bearing;
182 loader->advance = advance_width;
183 loader->top_bearing = top_bearing;
184 loader->vadvance = advance_height;
185
186#ifdef TT_SUPPORT_SUBPIXEL_HINTING_INFINALITY
187 if ( driver->interpreter_version == TT_INTERPRETER_VERSION_38 &&
188 loader->exec )
189 {
190 loader->exec->sph_tweak_flags = 0;
191
192 /* This may not be the right place for this, but it works... */
193 /* Note that we have to unconditionally load the tweaks since */
194 /* it is possible that glyphs individually switch ClearType's */
195 /* backward compatibility mode on and off. */
196 sph_set_tweaks( loader, glyph_index );
197 }
198#endif /* TT_SUPPORT_SUBPIXEL_HINTING_INFINALITY */
199
200#ifdef FT_CONFIG_OPTION_INCREMENTAL
201 /* With the incremental interface, these values are set by */
202 /* a call to `tt_get_metrics_incremental'. */
203 if ( face->root.internal->incremental_interface == NULL )
204#endif
205 {
206 if ( !loader->linear_def )
207 {
208 loader->linear_def = 1;
209 loader->linear = advance_width;
210 }
211 }
212
213 return FT_Err_Ok;
214 }

332 {

358 FT_CALLBACK_DEF( FT_Error )
359 TT_Load_Simple_Glyph( TT_Loader load )
360 {
361 FT_Error error;
362 FT_Byte* p = load->cursor;
363 FT_Byte* limit = load->limit;
364 FT_GlyphLoader gloader = load->gloader;
365 FT_Outline* outline = &gloader->current.outline;
366 FT_Int n_contours = load->n_contours;
367 FT_Int n_points;
368 FT_UShort n_ins;
369
370 FT_Byte *flag, *flag_limit;
371 FT_Byte c, count;
372 FT_Vector *vec, *vec_limit;
373 FT_Pos x, y;
374 FT_Short *cont, *cont_limit, last;
375
376
377 /* check that we can add the contours to the glyph */
378 error = FT_GLYPHLOADER_CHECK_POINTS( gloader, 0, n_contours );
379 if ( error )
380 goto Fail;
381
382 /* check space for contours array + instructions count */
383 if ( n_contours >= 0xFFF || p + 2 * n_contours + 2 > limit )
384 goto Invalid_Outline;
385
386 /* reading the contours' endpoints & number of points */
387 cont = outline->contours;
388 cont_limit = cont + n_contours;
389
390 last = -1;
391 for ( ; cont < cont_limit; cont++ )
392 {
393 *cont = FT_NEXT_SHORT( p );
394
395 if ( *cont <= last )
396 goto Invalid_Outline;
397
398 last = *cont;
399 }
400
401 n_points = last + 1;
402
403 FT_TRACE5(( " # of points: %d\n", n_points ));
404
405 /* note that we will add four phantom points later */
406 error = FT_GLYPHLOADER_CHECK_POINTS( gloader, n_points + 4, 0 );
407 if ( error )
408 goto Fail;
409
410 /* space checked above */
411 n_ins = FT_NEXT_USHORT( p );
412
413 FT_TRACE5(( " Instructions size: %u\n", n_ins ));
414
415 /* check instructions size */
416 if ( p + n_ins > limit )
417 {
418 FT_TRACE1(( "TT_Load_Simple_Glyph: excessive instruction count\n" ));
419 error = FT_THROW( Too_Many_Hints );
420 goto Fail;
421 }
422
423#ifdef TT_USE_BYTECODE_INTERPRETER
424
425 if ( IS_HINTED( load->load_flags ) )
426 {
427 TT_ExecContext exec = load->exec;
428 FT_Memory memory = exec->memory;
429
430
431 if ( exec->glyphSize )
432 FT_FREE( exec->glyphIns );
433 exec->glyphSize = 0;
434
435 /* we don't trust `maxSizeOfInstructions' in the `maxp' table */
436 /* and thus allocate the bytecode array size by ourselves */
437 if ( n_ins )
438 {
439 if ( FT_QNEW_ARRAY( exec->glyphIns, n_ins ) )
440 return error;
441
442 FT_MEM_COPY( exec->glyphIns, p, (FT_Long)n_ins );
443
444 exec->glyphSize = n_ins;
445 }
446 }
447
448#endif /* TT_USE_BYTECODE_INTERPRETER */
449
450 p += n_ins;
451
452 /* reading the point tags */
453 flag = (FT_Byte*)outline->tags;
454 flag_limit = flag + n_points;
455
456 FT_ASSERT( flag );
457
458 while ( flag < flag_limit )
459 {
460 if ( p + 1 > limit )
461 goto Invalid_Outline;
462
463 *flag++ = c = FT_NEXT_BYTE( p );
464 if ( c & REPEAT_FLAG )
465 {
466 if ( p + 1 > limit )
467 goto Invalid_Outline;
468
469 count = FT_NEXT_BYTE( p );
470 if ( flag + (FT_Int)count > flag_limit )
471 goto Invalid_Outline;
472
473 for ( ; count > 0; count-- )
474 *flag++ = c;
475 }
476 }
477
478 /* retain the overlap flag */
479 if ( n_points && outline->tags[0] & OVERLAP_SIMPLE )
480 gloader->base.outline.flags |= FT_OUTLINE_OVERLAP;
481
482 /* reading the X coordinates */
483
484 vec = outline->points;
485 vec_limit = vec + n_points;
486 flag = (FT_Byte*)outline->tags;
487 x = 0;
488
489 for ( ; vec < vec_limit; vec++, flag++ )
490 {
491 FT_Pos delta = 0;
492 FT_Byte f = *flag;
493
494
495 if ( f & X_SHORT_VECTOR )
496 {
497 if ( p + 1 > limit )
498 goto Invalid_Outline;
499
500 delta = (FT_Pos)FT_NEXT_BYTE( p );
501 if ( !( f & X_POSITIVE ) )
502 delta = -delta;
503 }
504 else if ( !( f & SAME_X ) )
505 {
506 if ( p + 2 > limit )
507 goto Invalid_Outline;
508
509 delta = (FT_Pos)FT_NEXT_SHORT( p );
510 }
511
512 x += delta;
513 vec->x = x;
514 }
515
516 /* reading the Y coordinates */
517
518 vec = outline->points;
519 vec_limit = vec + n_points;
520 flag = (FT_Byte*)outline->tags;
521 y = 0;
522
523 for ( ; vec < vec_limit; vec++, flag++ )
524 {
525 FT_Pos delta = 0;
526 FT_Byte f = *flag;
527
528
529 if ( f & Y_SHORT_VECTOR )
530 {
531 if ( p + 1 > limit )
532 goto Invalid_Outline;
533
534 delta = (FT_Pos)FT_NEXT_BYTE( p );
535 if ( !( f & Y_POSITIVE ) )
536 delta = -delta;
537 }
538 else if ( !( f & SAME_Y ) )
539 {
540 if ( p + 2 > limit )
541 goto Invalid_Outline;
542
543 delta = (FT_Pos)FT_NEXT_SHORT( p );
544 }
545
546 y += delta;
547 vec->y = y;
548
549 /* the cast is for stupid compilers */
550 *flag = (FT_Byte)( f & ON_CURVE_POINT );
551 }
552
553 outline->n_points = (FT_Short)n_points;
554 outline->n_contours = (FT_Short)n_contours;
555
556 load->cursor = p;
557
558 Fail:
559 return error;
560
561 Invalid_Outline:
562 error = FT_THROW( Invalid_Outline );
563 goto Fail;
564 }

567 FT_CALLBACK_DEF( FT_Error )
568 TT_Load_Composite_Glyph( TT_Loader loader )
569 {
570 FT_Error error;
571 FT_Byte* p = loader->cursor;
572 FT_Byte* limit = loader->limit;
573 FT_GlyphLoader gloader = loader->gloader;
574 FT_Long num_glyphs = loader->face->root.num_glyphs;
575 FT_SubGlyph subglyph;
576 FT_UInt num_subglyphs;
577
578
579 num_subglyphs = 0;
580
581 do
582 {
583 FT_Fixed xx, xy, yy, yx;
584 FT_UInt count;
585
586
587 /* check that we can load a new subglyph */
588 error = FT_GlyphLoader_CheckSubGlyphs( gloader, num_subglyphs + 1 );
589 if ( error )
590 goto Fail;
591
592 /* check space */
593 if ( p + 4 > limit )
594 goto Invalid_Composite;
595
596 subglyph = gloader->current.subglyphs + num_subglyphs;
597
598 subglyph->arg1 = subglyph->arg2 = 0;
599
600 subglyph->flags = FT_NEXT_USHORT( p );
601 subglyph->index = FT_NEXT_USHORT( p );
602
603 /* we reject composites that have components */
604 /* with invalid glyph indices */
605 if ( subglyph->index >= num_glyphs )
606 goto Invalid_Composite;
607
608 /* check space */
609 count = 2;
610 if ( subglyph->flags & ARGS_ARE_WORDS )
611 count += 2;
612 if ( subglyph->flags & WE_HAVE_A_SCALE )
613 count += 2;
614 else if ( subglyph->flags & WE_HAVE_AN_XY_SCALE )
615 count += 4;
616 else if ( subglyph->flags & WE_HAVE_A_2X2 )
617 count += 8;
618
619 if ( p + count > limit )
620 goto Invalid_Composite;
621
622 /* read arguments */
623 if ( subglyph->flags & ARGS_ARE_XY_VALUES )
624 {
625 if ( subglyph->flags & ARGS_ARE_WORDS )
626 {
627 subglyph->arg1 = FT_NEXT_SHORT( p );
628 subglyph->arg2 = FT_NEXT_SHORT( p );
629 }
630 else
631 {
632 subglyph->arg1 = FT_NEXT_CHAR( p );
633 subglyph->arg2 = FT_NEXT_CHAR( p );
634 }
635 }
636 else
637 {
638 if ( subglyph->flags & ARGS_ARE_WORDS )
639 {
640 subglyph->arg1 = (FT_Int)FT_NEXT_USHORT( p );
641 subglyph->arg2 = (FT_Int)FT_NEXT_USHORT( p );
642 }
643 else
644 {
645 subglyph->arg1 = (FT_Int)FT_NEXT_BYTE( p );
646 subglyph->arg2 = (FT_Int)FT_NEXT_BYTE( p );
647 }
648 }
649
650 /* read transform */
651 xx = yy = 0x10000L;
652 xy = yx = 0;
653
654 if ( subglyph->flags & WE_HAVE_A_SCALE )
655 {
656 xx = (FT_Fixed)FT_NEXT_SHORT( p ) * 4;
657 yy = xx;
658 }
659 else if ( subglyph->flags & WE_HAVE_AN_XY_SCALE )
660 {
661 xx = (FT_Fixed)FT_NEXT_SHORT( p ) * 4;
662 yy = (FT_Fixed)FT_NEXT_SHORT( p ) * 4;
663 }
664 else if ( subglyph->flags & WE_HAVE_A_2X2 )
665 {
666 xx = (FT_Fixed)FT_NEXT_SHORT( p ) * 4;
667 yx = (FT_Fixed)FT_NEXT_SHORT( p ) * 4;
668 xy = (FT_Fixed)FT_NEXT_SHORT( p ) * 4;
669 yy = (FT_Fixed)FT_NEXT_SHORT( p ) * 4;
670 }
671
672 subglyph->transform.xx = xx;
673 subglyph->transform.xy = xy;
674 subglyph->transform.yx = yx;
675 subglyph->transform.yy = yy;
676
677 num_subglyphs++;
678
679 } while ( subglyph->flags & MORE_COMPONENTS );
680
681 gloader->current.num_subglyphs = num_subglyphs;
682 FT_TRACE5(( " %d component%s\n",
683 num_subglyphs,
684 num_subglyphs > 1 ? "s" : "" ));
685
686#ifdef FT_DEBUG_LEVEL_TRACE
687 {
688 FT_UInt i;
689
690
691 subglyph = gloader->current.subglyphs;
692
693 for ( i = 0; i < num_subglyphs; i++ )
694 {
695 if ( num_subglyphs > 1 )
696 FT_TRACE7(( " subglyph %d:\n", i ));
697
698 FT_TRACE7(( " glyph index: %d\n", subglyph->index ));
699
700 if ( subglyph->flags & ARGS_ARE_XY_VALUES )
701 FT_TRACE7(( " offset: x=%d, y=%d\n",
702 subglyph->arg1,
703 subglyph->arg2 ));
704 else
705 FT_TRACE7(( " matching points: base=%d, component=%d\n",
706 subglyph->arg1,
707 subglyph->arg2 ));
708
709 if ( subglyph->flags & WE_HAVE_A_SCALE )
710 FT_TRACE7(( " scaling: %f\n",
711 (double)subglyph->transform.xx / 65536 ));
712 else if ( subglyph->flags & WE_HAVE_AN_XY_SCALE )
713 FT_TRACE7(( " scaling: x=%f, y=%f\n",
714 (double)subglyph->transform.xx / 65536,
715 (double)subglyph->transform.yy / 65536 ));
716 else if ( subglyph->flags & WE_HAVE_A_2X2 )
717 {
718 FT_TRACE7(( " scaling: xx=%f, yx=%f\n",
719 (double)subglyph->transform.xx / 65536,
720 (double)subglyph->transform.yx / 65536 ));
721 FT_TRACE7(( " xy=%f, yy=%f\n",
722 (double)subglyph->transform.xy / 65536,
723 (double)subglyph->transform.yy / 65536 ));
724 }
725
726 subglyph++;
727 }
728 }
729#endif /* FT_DEBUG_LEVEL_TRACE */
730
731#ifdef TT_USE_BYTECODE_INTERPRETER
732
733 {
734 FT_Stream stream = loader->stream;
735
736
737 /* we must undo the FT_FRAME_ENTER in order to point */
738 /* to the composite instructions, if we find some. */
739 /* We will process them later. */
740 /* */
741 loader->ins_pos = (FT_ULong)( FT_STREAM_POS() +
742 p - limit );
743 }
744
745#endif
746
747 loader->cursor = p;
748
749 Fail:
750 return error;
751
752 Invalid_Composite:
753 error = FT_THROW( Invalid_Composite );
754 goto Fail;
755 }

770 tt_prepare_zone( TT_GlyphZone zone,
771 FT_GlyphLoad load,
772 FT_UInt start_point,
773 FT_UInt start_contour )
774 {
775 zone->n_points = (FT_UShort)load->outline.n_points + 4 -

778 (FT_Short)start_contour;

820 FT_ARRAY_COPY( zone->org, zone->cur, zone->n_points );

847 FT_PIX_ROUND( zone->cur[zone->n_points - 2].y );

849 FT_PIX_ROUND( zone->cur[zone->n_points - 1].y );

855 FT_Error error;

880 {
881#endif
882 loader->pp1 = zone->cur[zone->n_points - 4];
883 loader->pp2 = zone->cur[zone->n_points - 3];

888#endif

894 FT_Outline_EmboldenXY( &loader->gloader->current.outline, -24, 0 );

1366
1367 /*
1368 * Calculate the phantom points
1369 *
1370 * Defining the right side bearing (rsb) as
1371 *
1372 * rsb = aw - (lsb + xmax - xmin)
1373 *
1374 * (with `aw' the advance width, `lsb' the left side bearing, and `xmin'
1375 * and `xmax' the glyph's minimum and maximum x value), the OpenType
1376 * specification defines the initial position of horizontal phantom points
1377 * as
1378 *
1379 * pp1 = (round(xmin - lsb), 0) ,
1380 * pp2 = (round(pp1 + aw), 0) .
1381 *
1382 * Note that the rounding to the grid (in the device space) is not
1383 * documented currently in the specification.
1384 *
1385 * However, the specification lacks the precise definition of vertical
1386 * phantom points. Greg Hitchcock provided the following explanation.
1387 *
1388 * - a `vmtx' table is present
1389 *
1390 * For any glyph, the minimum and maximum y values (`ymin' and `ymax')
1391 * are given in the `glyf' table, the top side bearing (tsb) and advance
1392 * height (ah) are given in the `vmtx' table. The bottom side bearing
1393 * (bsb) is then calculated as
1394 *
1395 * bsb = ah - (tsb + ymax - ymin) ,
1396 *
1397 * and the initial position of vertical phantom points is
1398 *
1399 * pp3 = (x, round(ymax + tsb)) ,
1400 * pp4 = (x, round(pp3 - ah)) .
1401 *
1402 * See below for value `x'.
1403 *
1404 * - no `vmtx' table in the font
1405 *
1406 * If there is an `OS/2' table, we set
1407 *
1408 * DefaultAscender = sTypoAscender ,
1409 * DefaultDescender = sTypoDescender ,
1410 *
1411 * otherwise we use data from the `hhea' table:
1412 *
1413 * DefaultAscender = Ascender ,
1414 * DefaultDescender = Descender .
1415 *
1416 * With these two variables we can now set
1417 *
1418 * ah = DefaultAscender - sDefaultDescender ,
1419 * tsb = DefaultAscender - yMax ,
1420 *
1421 * and proceed as if a `vmtx' table was present.
1422 *
1423 * Usually we have
1424 *
1425 * x = aw / 2 , (1)
1426 *
1427 * but there is one compatibility case where it can be set to
1428 *
1429 * x = -DefaultDescender -
1430 * ((DefaultAscender - DefaultDescender - aw) / 2) . (2)
1431 *
1432 * and another one with
1433 *
1434 * x = 0 . (3)
1435 *
1436 * In Windows, the history of those values is quite complicated,
1437 * depending on the hinting engine (that is, the graphics framework).
1438 *
1439 * framework from to formula
1440 * ----------------------------------------------------------
1441 * GDI Windows 98 current (1)
1442 * (Windows 2000 for NT)
1443 * GDI+ Windows XP Windows 7 (2)
1444 * GDI+ Windows 8 current (3)
1445 * DWrite Windows 7 current (3)
1446 *
1447 * For simplicity, FreeType uses (1) for grayscale subpixel hinting and
1448 * (3) for everything else.
1449 *
1450 */
1451 static void
1452 tt_loader_set_pp( TT_Loader loader )
1453 {
1454 FT_Bool subpixel_hinting = 0;
1455 FT_Bool grayscale = 0;
1456 FT_Bool use_aw_2 = 0;
1457
1458#ifdef TT_CONFIG_OPTION_SUBPIXEL_HINTING
1459 TT_Driver driver = (TT_Driver)FT_FACE_DRIVER( loader->face );
1460#endif
1461
1462
1463#ifdef TT_SUPPORT_SUBPIXEL_HINTING_INFINALITY
1464 if ( driver->interpreter_version == TT_INTERPRETER_VERSION_38 )
1465 {
1466 subpixel_hinting = loader->exec ? loader->exec->subpixel_hinting
1467 : 0;
1468 grayscale = loader->exec ? loader->exec->grayscale
1469 : 0;
1470 }
1471#endif
1472#ifdef TT_SUPPORT_SUBPIXEL_HINTING_MINIMAL
1473 if ( driver->interpreter_version == TT_INTERPRETER_VERSION_40 )
1474 {
1475 subpixel_hinting = loader->exec ? loader->exec->subpixel_hinting_lean
1476 : 0;
1477 grayscale = loader->exec ? loader->exec->grayscale_cleartype
1478 : 0;
1479 }
1480#endif
1481
1482 use_aw_2 = FT_BOOL( subpixel_hinting && grayscale );
1483
1484 loader->pp1.x = loader->bbox.xMin - loader->left_bearing;
1485 loader->pp1.y = 0;
1486 loader->pp2.x = loader->pp1.x + loader->advance;
1487 loader->pp2.y = 0;
1488
1489 loader->pp3.x = use_aw_2 ? loader->advance / 2 : 0;
1490 loader->pp3.y = loader->bbox.yMax + loader->top_bearing;
1491 loader->pp4.x = use_aw_2 ? loader->advance / 2 : 0;
1492 loader->pp4.y = loader->pp3.y - loader->vadvance;
1493 }

1519 /**************************************************************************
1520 *
1521 * @Function:
1522 * load_truetype_glyph
1523 *
1524 * @Description:
1525 * Loads a given truetype glyph. Handles composites and uses a
1526 * TT_Loader object.
1527 */
1528 static FT_Error
1529 load_truetype_glyph( TT_Loader loader,
1530 FT_UInt glyph_index,
1531 FT_UInt recurse_count,
1532 FT_Bool header_only )
1533 {
1534 FT_Error error = FT_Err_Ok;
1535 FT_Fixed x_scale, y_scale;
1536 FT_ULong offset;
1537 TT_Face face = loader->face;
1538 FT_GlyphLoader gloader = loader->gloader;
1539
1540 FT_Bool opened_frame = 0;
1541
1542#ifdef FT_CONFIG_OPTION_INCREMENTAL
1543 FT_StreamRec inc_stream;
1544 FT_Data glyph_data;
1545 FT_Bool glyph_data_loaded = 0;
1546#endif
1547
1548
1549#ifdef FT_DEBUG_LEVEL_TRACE
1550 if ( recurse_count )
1551 FT_TRACE5(( " nesting level: %d\n", recurse_count ));
1552#endif
1553
1554 /* some fonts have an incorrect value of `maxComponentDepth' */
1555 if ( recurse_count > face->max_profile.maxComponentDepth )
1556 {
1557 FT_TRACE1(( "load_truetype_glyph: maxComponentDepth set to %d\n",
1558 recurse_count ));
1559 face->max_profile.maxComponentDepth = (FT_UShort)recurse_count;
1560 }
1561
1562#ifndef FT_CONFIG_OPTION_INCREMENTAL
1563 /* check glyph index */
1564 if ( glyph_index >= (FT_UInt)face->root.num_glyphs )
1565 {
1566 error = FT_THROW( Invalid_Glyph_Index );
1567 goto Exit;
1568 }
1569#endif
1570
1571 loader->glyph_index = glyph_index;
1572
1573 if ( loader->load_flags & FT_LOAD_NO_SCALE )
1574 {
1575 x_scale = 0x10000L;
1576 y_scale = 0x10000L;
1577 }
1578 else
1579 {
1580 x_scale = loader->size->metrics->x_scale;
1581 y_scale = loader->size->metrics->y_scale;
1582 }
1583
1584 /* Set `offset' to the start of the glyph relative to the start of */
1585 /* the `glyf' table, and `byte_len' to the length of the glyph in */
1586 /* bytes. */
1587
1588#ifdef FT_CONFIG_OPTION_INCREMENTAL
1589
1590 /* If we are loading glyph data via the incremental interface, set */
1591 /* the loader stream to a memory stream reading the data returned */
1592 /* by the interface. */
1593 if ( face->root.internal->incremental_interface )
1594 {
1595 error = face->root.internal->incremental_interface->funcs->get_glyph_data(
1596 face->root.internal->incremental_interface->object,
1597 glyph_index, &glyph_data );
1598 if ( error )
1599 goto Exit;
1600
1601 glyph_data_loaded = 1;
1602 offset = 0;
1603 loader->byte_len = glyph_data.length;
1604
1605 FT_ZERO( &inc_stream );
1606 FT_Stream_OpenMemory( &inc_stream,
1607 glyph_data.pointer,
1608 glyph_data.length );
1609
1610 loader->stream = &inc_stream;
1611 }
1612 else
1613
1614#endif /* FT_CONFIG_OPTION_INCREMENTAL */
1615 {
1616 FT_ULong len;
1617
1618
1619 offset = tt_face_get_location( FT_FACE( face ), glyph_index, &len );
1620
1621 loader->byte_len = (FT_UInt)len;
1622 }
1623
1624 if ( loader->byte_len > 0 )
1625 {
1626#ifdef FT_CONFIG_OPTION_INCREMENTAL
1627 /* for the incremental interface, `glyf_offset' is always zero */
1628 if ( !face->glyf_offset &&
1629 !face->root.internal->incremental_interface )
1630#else
1631 if ( !face->glyf_offset )
1632#endif /* FT_CONFIG_OPTION_INCREMENTAL */
1633 {
1634 FT_TRACE2(( "no `glyf' table but non-zero `loca' entry\n" ));
1635 error = FT_THROW( Invalid_Table );
1636 goto Exit;
1637 }
1638
1639 error = face->access_glyph_frame( loader, glyph_index,
1640 face->glyf_offset + offset,
1641 loader->byte_len );
1642 if ( error )
1643 goto Exit;
1644
1645 /* read glyph header first */
1646 error = face->read_glyph_header( loader );
1647
1648 face->forget_glyph_frame( loader );
1649
1650 if ( error )
1651 goto Exit;
1652 }
1653
1654 /* a space glyph */
1655 if ( loader->byte_len == 0 || loader->n_contours == 0 )
1656 {
1657 loader->bbox.xMin = 0;
1658 loader->bbox.xMax = 0;
1659 loader->bbox.yMin = 0;
1660 loader->bbox.yMax = 0;
1661 }
1662
1663 /* the metrics must be computed after loading the glyph header */
1664 /* since we need the glyph's `yMax' value in case the vertical */
1665 /* metrics must be emulated */
1666 error = tt_get_metrics( loader, glyph_index );
1667 if ( error )
1668 goto Exit;
1669
1670 if ( header_only )
1671 goto Exit;
1672
1673 if ( loader->byte_len == 0 || loader->n_contours == 0 )
1674 {
1675#ifdef FT_CONFIG_OPTION_INCREMENTAL
1676 tt_get_metrics_incremental( loader, glyph_index );
1677#endif
1678 tt_loader_set_pp( loader );
1679
1680
1681#ifdef TT_CONFIG_OPTION_GX_VAR_SUPPORT
1682
1683 if ( FT_IS_NAMED_INSTANCE( FT_FACE( face ) ) ||
1684 FT_IS_VARIATION( FT_FACE( face ) ) )
1685 {
1686 /* a small outline structure with four elements for */
1687 /* communication with `TT_Vary_Apply_Glyph_Deltas' */
1688 FT_Vector points[4];
1689 FT_Outline outline;
1690
1691 /* unrounded values */
1692 FT_Vector unrounded[4] = { {0, 0}, {0, 0}, {0, 0}, {0, 0} };
1693
1694
1695 points[0] = loader->pp1;
1696 points[1] = loader->pp2;
1697 points[2] = loader->pp3;
1698 points[3] = loader->pp4;
1699
1700 outline.n_points = 0;
1701 outline.n_contours = 0;
1702 outline.points = points;
1703 outline.tags = NULL;
1704 outline.contours = NULL;
1705
1706 /* this must be done before scaling */
1707 error = TT_Vary_Apply_Glyph_Deltas( loader,
1708 &outline,
1709 unrounded );
1710 if ( error )
1711 goto Exit;
1712 }
1713
1714#endif /* TT_CONFIG_OPTION_GX_VAR_SUPPORT */
1715
1716 /* scale phantom points, if necessary; */
1717 /* they get rounded in `TT_Hint_Glyph' */
1718 if ( ( loader->load_flags & FT_LOAD_NO_SCALE ) == 0 )
1719 {
1720 loader->pp1.x = FT_MulFix( loader->pp1.x, x_scale );
1721 loader->pp2.x = FT_MulFix( loader->pp2.x, x_scale );
1722 /* pp1.y and pp2.y are always zero */
1723
1724 loader->pp3.x = FT_MulFix( loader->pp3.x, x_scale );
1725 loader->pp3.y = FT_MulFix( loader->pp3.y, y_scale );
1726 loader->pp4.x = FT_MulFix( loader->pp4.x, x_scale );
1727 loader->pp4.y = FT_MulFix( loader->pp4.y, y_scale );
1728 }
1729
1730 error = FT_Err_Ok;
1731 goto Exit;
1732 }
1733
1734#ifdef FT_CONFIG_OPTION_INCREMENTAL
1735 tt_get_metrics_incremental( loader, glyph_index );
1736#endif
1737 tt_loader_set_pp( loader );
1738
1739
1740 /***********************************************************************/
1741 /***********************************************************************/
1742 /***********************************************************************/
1743
1744 /* we now open a frame again, right after the glyph header */
1745 /* (which consists of 10 bytes) */
1746 error = face->access_glyph_frame( loader, glyph_index,
1747 face->glyf_offset + offset + 10,
1748 loader->byte_len - 10 );
1749 if ( error )
1750 goto Exit;
1751
1752 opened_frame = 1;
1753
1754 /* if it is a simple glyph, load it */
1755
1756 if ( loader->n_contours > 0 )
1757 {
1758 error = face->read_simple_glyph( loader );
1759 if ( error )
1760 goto Exit;
1761
1762 /* all data have been read */
1763 face->forget_glyph_frame( loader );
1764 opened_frame = 0;
1765
1766 error = TT_Process_Simple_Glyph( loader );
1767 if ( error )
1768 goto Exit;
1769
1770 FT_GlyphLoader_Add( gloader );
1771 }
1772
1773 /***********************************************************************/
1774 /***********************************************************************/
1775 /***********************************************************************/
1776
1777 /* otherwise, load a composite! */
1778 else if ( loader->n_contours < 0 )
1779 {
1780 FT_Memory memory = face->root.memory;
1781
1782 FT_UInt start_point;
1783 FT_UInt start_contour;
1784 FT_ULong ins_pos; /* position of composite instructions, if any */
1785
1786 FT_ListNode node, node2;
1787
1788
1789 /* normalize the `n_contours' value */
1790 loader->n_contours = -1;
1791
1792 /*
1793 * We store the glyph index directly in the `node->data' pointer,
1794 * following the glib solution (cf. macro `GUINT_TO_POINTER') with a
1795 * double cast to make this portable. Note, however, that this needs
1796 * pointers with a width of at least 32 bits.
1797 */
1798
1799 /* clear the nodes filled by sibling chains */
1800 node = ft_list_get_node_at( &loader->composites, recurse_count );
1801 for ( node2 = node; node2; node2 = node2->next )
1802 node2->data = (void*)-1;
1803
1804 /* check whether we already have a composite glyph with this index */
1805 if ( FT_List_Find( &loader->composites,
1806 FT_UINT_TO_POINTER( glyph_index ) ) )
1807 {
1808 FT_TRACE1(( "TT_Load_Composite_Glyph:"
1809 " infinite recursion detected\n" ));
1810 error = FT_THROW( Invalid_Composite );
1811 goto Exit;
1812 }
1813
1814 else if ( node )
1815 node->data = FT_UINT_TO_POINTER( glyph_index );
1816
1817 else
1818 {
1819 if ( FT_QNEW( node ) )
1820 goto Exit;
1821 node->data = FT_UINT_TO_POINTER( glyph_index );
1822 FT_List_Add( &loader->composites, node );
1823 }
1824
1825 start_point = (FT_UInt)gloader->base.outline.n_points;
1826 start_contour = (FT_UInt)gloader->base.outline.n_contours;
1827
1828 /* for each subglyph, read composite header */
1829 error = face->read_composite_glyph( loader );
1830 if ( error )
1831 goto Exit;
1832
1833 /* store the offset of instructions */
1834 ins_pos = loader->ins_pos;
1835
1836 /* all data we need are read */
1837 face->forget_glyph_frame( loader );
1838 opened_frame = 0;
1839
1840#ifdef TT_CONFIG_OPTION_GX_VAR_SUPPORT
1841
1842 if ( FT_IS_NAMED_INSTANCE( FT_FACE( face ) ) ||
1843 FT_IS_VARIATION( FT_FACE( face ) ) )
1844 {
1845 short i, limit;
1846 FT_SubGlyph subglyph;
1847
1848 FT_Outline outline = { 0, 0, NULL, NULL, NULL, 0 };
1849 FT_Vector* unrounded = NULL;
1850
1851
1852 limit = (short)gloader->current.num_subglyphs;
1853
1854 /* construct an outline structure for */
1855 /* communication with `TT_Vary_Apply_Glyph_Deltas' */
1856 if ( FT_QNEW_ARRAY( outline.points, limit + 4 ) ||
1857 FT_QNEW_ARRAY( outline.tags, limit ) ||
1858 FT_QNEW_ARRAY( outline.contours, limit ) ||
1859 FT_QNEW_ARRAY( unrounded, limit + 4 ) )
1860 goto Exit1;
1861
1862 outline.n_contours = outline.n_points = limit;
1863
1864 subglyph = gloader->current.subglyphs;
1865
1866 for ( i = 0; i < limit; i++, subglyph++ )
1867 {
1868 /* applying deltas for anchor points doesn't make sense, */
1869 /* but we don't have to specially check this since */
1870 /* unused delta values are zero anyways */
1871 outline.points[i].x = subglyph->arg1;
1872 outline.points[i].y = subglyph->arg2;
1873 outline.tags[i] = ON_CURVE_POINT;
1874 outline.contours[i] = i;
1875 }
1876
1877 outline.points[i++] = loader->pp1;
1878 outline.points[i++] = loader->pp2;
1879 outline.points[i++] = loader->pp3;
1880 outline.points[i ] = loader->pp4;
1881
1882 /* this call provides additional offsets */
1883 /* for each component's translation */
1884 if ( FT_SET_ERROR( TT_Vary_Apply_Glyph_Deltas( loader,
1885 &outline,
1886 unrounded ) ) )
1887 goto Exit1;
1888
1889 subglyph = gloader->current.subglyphs;
1890
1891 for ( i = 0; i < limit; i++, subglyph++ )
1892 {
1893 if ( subglyph->flags & ARGS_ARE_XY_VALUES )
1894 {
1895 subglyph->arg1 = (FT_Int16)outline.points[i].x;
1896 subglyph->arg2 = (FT_Int16)outline.points[i].y;
1897 }
1898 }
1899
1900 Exit1:
1901 FT_FREE( outline.points );
1902 FT_FREE( outline.tags );
1903 FT_FREE( outline.contours );
1904 FT_FREE( unrounded );
1905
1906 if ( error )
1907 goto Exit;
1908 }
1909
1910#endif /* TT_CONFIG_OPTION_GX_VAR_SUPPORT */
1911
1912 /* scale phantom points, if necessary; */
1913 /* they get rounded in `TT_Hint_Glyph' */
1914 if ( ( loader->load_flags & FT_LOAD_NO_SCALE ) == 0 )
1915 {
1916 loader->pp1.x = FT_MulFix( loader->pp1.x, x_scale );
1917 loader->pp2.x = FT_MulFix( loader->pp2.x, x_scale );
1918 /* pp1.y and pp2.y are always zero */
1919
1920 loader->pp3.x = FT_MulFix( loader->pp3.x, x_scale );
1921 loader->pp3.y = FT_MulFix( loader->pp3.y, y_scale );
1922 loader->pp4.x = FT_MulFix( loader->pp4.x, x_scale );
1923 loader->pp4.y = FT_MulFix( loader->pp4.y, y_scale );
1924 }
1925
1926 /* if the flag FT_LOAD_NO_RECURSE is set, we return the subglyph */
1927 /* `as is' in the glyph slot (the client application will be */
1928 /* responsible for interpreting these data)... */
1929 if ( loader->load_flags & FT_LOAD_NO_RECURSE )
1930 {
1931 FT_GlyphLoader_Add( gloader );
1932 loader->glyph->format = FT_GLYPH_FORMAT_COMPOSITE;
1933
1934 goto Exit;
1935 }
1936
1937 /*********************************************************************/
1938 /*********************************************************************/
1939 /*********************************************************************/
1940
1941 {
1942 FT_UInt n, num_base_points;
1943 FT_SubGlyph subglyph = NULL;
1944
1945 FT_UInt num_points = start_point;
1946 FT_UInt num_subglyphs = gloader->current.num_subglyphs;
1947 FT_UInt num_base_subgs = gloader->base.num_subglyphs;
1948
1949 FT_Stream old_stream = loader->stream;
1950 FT_UInt old_byte_len = loader->byte_len;
1951
1952
1953 FT_GlyphLoader_Add( gloader );
1954
1955 /* read each subglyph independently */
1956 for ( n = 0; n < num_subglyphs; n++ )
1957 {
1958 FT_Vector pp[4];
1959
1960 FT_Int linear_hadvance;
1961 FT_Int linear_vadvance;
1962
1963
1964 /* Each time we call `load_truetype_glyph' in this loop, the */
1965 /* value of `gloader.base.subglyphs' can change due to table */
1966 /* reallocations. We thus need to recompute the subglyph */
1967 /* pointer on each iteration. */
1968 subglyph = gloader->base.subglyphs + num_base_subgs + n;
1969
1970 pp[0] = loader->pp1;
1971 pp[1] = loader->pp2;
1972 pp[2] = loader->pp3;
1973 pp[3] = loader->pp4;
1974
1975 linear_hadvance = loader->linear;
1976 linear_vadvance = loader->vadvance;
1977
1978 num_base_points = (FT_UInt)gloader->base.outline.n_points;
1979
1980 error = load_truetype_glyph( loader,
1981 (FT_UInt)subglyph->index,
1982 recurse_count + 1,
1983 FALSE );
1984 if ( error )
1985 goto Exit;
1986
1987 /* restore subglyph pointer */
1988 subglyph = gloader->base.subglyphs + num_base_subgs + n;
1989
1990 /* restore phantom points if necessary */
1991 if ( !( subglyph->flags & USE_MY_METRICS ) )
1992 {
1993 loader->pp1 = pp[0];
1994 loader->pp2 = pp[1];
1995 loader->pp3 = pp[2];
1996 loader->pp4 = pp[3];
1997
1998 loader->linear = linear_hadvance;
1999 loader->vadvance = linear_vadvance;
2000 }
2001
2002 num_points = (FT_UInt)gloader->base.outline.n_points;
2003
2004 if ( num_points == num_base_points )
2005 continue;
2006
2007 /* gloader->base.outline consists of three parts: */
2008 /* */
2009 /* 0 ----> start_point ----> num_base_points ----> n_points */
2010 /* (1) (2) (3) */
2011 /* */
2012 /* (1) points that exist from the beginning */
2013 /* (2) component points that have been loaded so far */
2014 /* (3) points of the newly loaded component */
2015 error = TT_Process_Composite_Component( loader,
2016 subglyph,
2017 start_point,
2018 num_base_points );
2019 if ( error )
2020 goto Exit;
2021 }
2022
2023 loader->stream = old_stream;
2024 loader->byte_len = old_byte_len;
2025
2026 /* process the glyph */
2027 loader->ins_pos = ins_pos;
2028 if ( IS_HINTED( loader->load_flags ) &&
2029#ifdef TT_USE_BYTECODE_INTERPRETER
2030 subglyph &&
2031 subglyph->flags & WE_HAVE_INSTR &&
2032#endif
2033 num_points > start_point )
2034 {
2035 error = TT_Process_Composite_Glyph( loader,
2036 start_point,
2037 start_contour );
2038 if ( error )
2039 goto Exit;
2040 }
2041 }
2042
2043 /* retain the overlap flag */
2044 if ( gloader->base.num_subglyphs &&
2045 gloader->base.subglyphs[0].flags & OVERLAP_COMPOUND )
2046 gloader->base.outline.flags |= FT_OUTLINE_OVERLAP;
2047 }
2048
2049 /***********************************************************************/
2050 /***********************************************************************/
2051 /***********************************************************************/
2052
2053 Exit:
2054
2055 if ( opened_frame )
2056 face->forget_glyph_frame( loader );
2057
2058#ifdef FT_CONFIG_OPTION_INCREMENTAL
2059
2060 if ( glyph_data_loaded )
2061 face->root.internal->incremental_interface->funcs->free_glyph_data(
2062 face->root.internal->incremental_interface->object,
2063 &glyph_data );
2064
2065#endif
2066
2067 return error;
2068 }

2071 static FT_Error
2072 compute_glyph_metrics( TT_Loader loader,
2073 FT_UInt glyph_index )
2074 {
2075 TT_Face face = loader->face;
2076 TT_Size size = loader->size;
2077 TT_GlyphSlot glyph = loader->glyph;
2078 FT_BBox bbox;
2079 FT_Fixed y_scale;
2080
2081
2082 y_scale = 0x10000L;
2083 if ( ( loader->load_flags & FT_LOAD_NO_SCALE ) == 0 )
2084 y_scale = size->metrics->y_scale;
2085
2086 if ( glyph->format != FT_GLYPH_FORMAT_COMPOSITE )
2087 FT_Outline_Get_CBox( &glyph->outline, &bbox );
2088 else
2089 bbox = loader->bbox;
2090
2091 /* get the device-independent horizontal advance; it is scaled later */
2092 /* by the base layer. */
2093 glyph->linearHoriAdvance = loader->linear;
2094
2095 glyph->metrics.horiBearingX = bbox.xMin;
2096 glyph->metrics.horiBearingY = bbox.yMax;
2097 if ( loader->widthp )
2098 glyph->metrics.horiAdvance = loader->widthp[glyph_index] * 64;
2099 else
2100 glyph->metrics.horiAdvance = SUB_LONG( loader->pp2.x, loader->pp1.x );
2101
2102 /* set glyph dimensions */
2103 glyph->metrics.width = SUB_LONG( bbox.xMax, bbox.xMin );
2104 glyph->metrics.height = SUB_LONG( bbox.yMax, bbox.yMin );
2105
2106 /* Now take care of vertical metrics. In the case where there is */
2107 /* no vertical information within the font (relatively common), */
2108 /* create some metrics manually */
2109 {
2110 FT_Pos top; /* scaled vertical top side bearing */
2111 FT_Pos advance; /* scaled vertical advance height */
2112
2113
2114 /* Get the unscaled top bearing and advance height. */
2115 if ( face->vertical_info &&
2116 face->vertical.number_Of_VMetrics > 0 )
2117 {
2118 top = (FT_Short)FT_DivFix( SUB_LONG( loader->pp3.y, bbox.yMax ),
2119 y_scale );
2120
2121 if ( loader->pp3.y <= loader->pp4.y )
2122 advance = 0;
2123 else
2124 advance = (FT_UShort)FT_DivFix( SUB_LONG( loader->pp3.y,
2125 loader->pp4.y ),
2126 y_scale );
2127 }
2128 else
2129 {
2130 FT_Pos height;
2131
2132
2133 /* XXX Compute top side bearing and advance height in */
2134 /* Get_VMetrics instead of here. */
2135
2136 /* NOTE: The OS/2 values are the only `portable' ones, */
2137 /* which is why we use them, if there is an OS/2 */
2138 /* table in the font. Otherwise, we use the */
2139 /* values defined in the horizontal header. */
2140
2141 height = (FT_Short)FT_DivFix( SUB_LONG( bbox.yMax,
2142 bbox.yMin ),
2143 y_scale );
2144 if ( face->os2.version != 0xFFFFU )
2145 advance = (FT_Pos)( face->os2.sTypoAscender -
2146 face->os2.sTypoDescender );
2147 else
2148 advance = (FT_Pos)( face->horizontal.Ascender -
2149 face->horizontal.Descender );
2150
2151 top = ( advance - height ) / 2;
2152 }
2153
2154#ifdef FT_CONFIG_OPTION_INCREMENTAL
2155 {
2156 FT_Incremental_InterfaceRec* incr;
2157 FT_Incremental_MetricsRec incr_metrics;
2158 FT_Error error;
2159
2160
2161 incr = face->root.internal->incremental_interface;
2162
2163 /* If this is an incrementally loaded font see if there are */
2164 /* overriding metrics for this glyph. */
2165 if ( incr && incr->funcs->get_glyph_metrics )
2166 {
2167 incr_metrics.bearing_x = 0;
2168 incr_metrics.bearing_y = top;
2169 incr_metrics.advance = advance;
2170
2171 error = incr->funcs->get_glyph_metrics( incr->object,
2172 glyph_index,
2173 TRUE,
2174 &incr_metrics );
2175 if ( error )
2176 return error;
2177
2178 top = incr_metrics.bearing_y;
2179 advance = incr_metrics.advance;
2180 }
