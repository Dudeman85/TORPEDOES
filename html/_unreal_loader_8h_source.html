120 // UNREAL vertex compression
121inline void CompressVertex(const aiVector3D& v, uint32_t& out)
122{
123 union {
124 Vertex n;
125 int32_t t;
126 };
127 n.X = (int32_t)v.x;
128 n.Y = (int32_t)v.y;
129 n.Z = (int32_t)v.z;
130 ::memcpy( &out, &t, sizeof(int32_t));
131 //out = t;
132}
133
134 // UNREAL vertex decompression
135inline void DecompressVertex(aiVector3D& v, int32_t in)
136{
137 union {
138 Vertex n;
139 int32_t i;
140 };
141 i = in;
142
143 v.x = (float)n.X;
144 v.y = (float)n.Y;
145 v.z = (float)n.Z;
146}
