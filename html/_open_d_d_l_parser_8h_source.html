44bool isEmbeddedCommentOpenTag( T *in, T *end ) {

47 }

61template<class T>
62inline
63T *lookForNextToken( T *in, T *end ) {
64 while( ( in != end ) && ( isSpace( *in ) || isNewLine( *in ) || ',' == *in ) ) {
65 in++;
66 }
67 return in;
68}

75template<class T>
76inline
77T *getNextToken( T *in, T *end ) {
78 T *tmp( in );
79 in = lookForNextToken( in, end );
80 if( tmp == in ) {
81 in++;
82 }
83 return in;
84}
