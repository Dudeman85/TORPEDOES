38 template<class T>
39 O3DGCErrorCode SC3DMCDecoder<T>::DecodeHeader(IndexedFaceSet<T> & ifs,
40 const BinaryStream & bstream)
41 {
42 unsigned long iterator0 = m_iterator;
43 unsigned long start_code = bstream.ReadUInt32(m_iterator, O3DGC_STREAM_TYPE_BINARY);
44 if (start_code != O3DGC_SC3DMC_START_CODE)
45 {
46 m_iterator = iterator0;
47 start_code = bstream.ReadUInt32(m_iterator, O3DGC_STREAM_TYPE_ASCII);
48 if (start_code != O3DGC_SC3DMC_START_CODE)
49 {
50 return O3DGC_ERROR_CORRUPTED_STREAM;
51 }
52 else
53 {
54 m_streamType = O3DGC_STREAM_TYPE_ASCII;
55 }
56 }
57 else
58 {
59 m_streamType = O3DGC_STREAM_TYPE_BINARY;
60 }
61
62 m_streamSize = bstream.ReadUInt32(m_iterator, m_streamType);
63 m_params.SetEncodeMode( (O3DGCSC3DMCEncodingMode) bstream.ReadUChar(m_iterator, m_streamType));
64
65 ifs.SetCreaseAngle((Real) bstream.ReadFloat32(m_iterator, m_streamType));
66
67 unsigned char mask = bstream.ReadUChar(m_iterator, m_streamType);
68
69 ifs.SetCCW ((mask & 1) == 1);
70 ifs.SetSolid ((mask & 2) == 1);
71 ifs.SetConvex ((mask & 4) == 1);
72 ifs.SetIsTriangularMesh((mask & 8) == 1);
73 //bool markerBit0 = (mask & 16 ) == 1;
74 //bool markerBit1 = (mask & 32 ) == 1;
75 //bool markerBit2 = (mask & 64 ) == 1;
76 //bool markerBit3 = (mask & 128) == 1;
77
78 ifs.SetNCoord (bstream.ReadUInt32(m_iterator, m_streamType));
79 ifs.SetNNormal (bstream.ReadUInt32(m_iterator, m_streamType));
80
81
82 ifs.SetNumFloatAttributes(bstream.ReadUInt32(m_iterator, m_streamType));
83 ifs.SetNumIntAttributes (bstream.ReadUInt32(m_iterator, m_streamType));
84
85 if (ifs.GetNCoord() > 0)
86 {
87 ifs.SetNCoordIndex(bstream.ReadUInt32(m_iterator, m_streamType));
88 for(int j=0 ; j<3 ; ++j)
89 {
90 ifs.SetCoordMin(j, (Real) bstream.ReadFloat32(m_iterator, m_streamType));
91 ifs.SetCoordMax(j, (Real) bstream.ReadFloat32(m_iterator, m_streamType));
92 }
93 m_params.SetCoordQuantBits( bstream.ReadUChar(m_iterator, m_streamType) );
94 }
95 if (ifs.GetNNormal() > 0)
96 {
97 ifs.SetNNormalIndex(bstream.ReadUInt32(m_iterator, m_streamType));
98 for(int j=0 ; j<3 ; ++j)
99 {
100 ifs.SetNormalMin(j, (Real) bstream.ReadFloat32(m_iterator, m_streamType));
101 ifs.SetNormalMax(j, (Real) bstream.ReadFloat32(m_iterator, m_streamType));
102 }
103 ifs.SetNormalPerVertex(bstream.ReadUChar(m_iterator, m_streamType) == 1);
104 m_params.SetNormalQuantBits(bstream.ReadUChar(m_iterator, m_streamType));
105 }
106
107 for(unsigned long a = 0; a < ifs.GetNumFloatAttributes(); ++a)
108 {
109 ifs.SetNFloatAttribute(a, bstream.ReadUInt32(m_iterator, m_streamType));
110 if (ifs.GetNFloatAttribute(a) > 0)
111 {
112 ifs.SetNFloatAttributeIndex(a, bstream.ReadUInt32(m_iterator, m_streamType));
113 unsigned char d = bstream.ReadUChar(m_iterator, m_streamType);
114 ifs.SetFloatAttributeDim(a, d);
115 for(unsigned char j = 0 ; j < d ; ++j)
116 {
117 ifs.SetFloatAttributeMin(a, j, (Real) bstream.ReadFloat32(m_iterator, m_streamType));
118 ifs.SetFloatAttributeMax(a, j, (Real) bstream.ReadFloat32(m_iterator, m_streamType));
119 }
120 ifs.SetFloatAttributePerVertex(a, bstream.ReadUChar(m_iterator, m_streamType) == 1);
121 ifs.SetFloatAttributeType(a, (O3DGCIFSFloatAttributeType) bstream.ReadUChar(m_iterator, m_streamType));
122 m_params.SetFloatAttributeQuantBits(a, bstream.ReadUChar(m_iterator, m_streamType));
123 }
124 }
125 for(unsigned long a = 0; a < ifs.GetNumIntAttributes(); ++a)
126 {
127 ifs.SetNIntAttribute(a, bstream.ReadUInt32(m_iterator, m_streamType));
128 if (ifs.GetNIntAttribute(a) > 0)
129 {
130 ifs.SetNIntAttributeIndex(a, bstream.ReadUInt32(m_iterator, m_streamType));
131 ifs.SetIntAttributeDim(a, bstream.ReadUChar(m_iterator, m_streamType));
132 ifs.SetIntAttributePerVertex(a, bstream.ReadUChar(m_iterator, m_streamType) == 1);
133 ifs.SetIntAttributeType(a, (O3DGCIFSIntAttributeType) bstream.ReadUChar(m_iterator, m_streamType));
134 }
135 }
136 return O3DGC_OK;
137 }
138 template<class T>
139 O3DGCErrorCode SC3DMCDecoder<T>::DecodePayload(IndexedFaceSet<T> & ifs,
140 const BinaryStream & bstream)
141 {
142 O3DGCErrorCode ret = O3DGC_OK;
143#ifdef DEBUG_VERBOSE
144 g_fileDebugSC3DMCDec = fopen("tfans_dec_main.txt", "w");
145#endif //DEBUG_VERBOSE
146
147 m_triangleListDecoder.SetStreamType(m_streamType);
148 m_stats.m_streamSizeCoordIndex = m_iterator;
149 Timer timer;
150 timer.Tic();
151 m_triangleListDecoder.Decode(ifs.GetCoordIndex(), ifs.GetNCoordIndex(), ifs.GetNCoord(), bstream, m_iterator);
152 timer.Toc();
153 m_stats.m_timeCoordIndex = timer.GetElapsedTime();
154 m_stats.m_streamSizeCoordIndex = m_iterator - m_stats.m_streamSizeCoordIndex;
155
156 // decode coord
157 m_stats.m_streamSizeCoord = m_iterator;
158 timer.Tic();
159 if (ifs.GetNCoord() > 0)
160 {
161 ret = DecodeFloatArray(ifs.GetCoord(), ifs.GetNCoord(), 3, 3, ifs.GetCoordMin(), ifs.GetCoordMax(),
162 m_params.GetCoordQuantBits(), ifs, m_params.GetCoordPredMode(), bstream);
163 }
164 if (ret != O3DGC_OK)
165 {
166 return ret;
167 }
168 timer.Toc();
169 m_stats.m_timeCoord = timer.GetElapsedTime();
170 m_stats.m_streamSizeCoord = m_iterator - m_stats.m_streamSizeCoord;
171
172 // decode Normal
173 m_stats.m_streamSizeNormal = m_iterator;
174 timer.Tic();
175 if (ifs.GetNNormal() > 0)
176 {
177 DecodeFloatArray(ifs.GetNormal(), ifs.GetNNormal(), 3, 3, ifs.GetNormalMin(), ifs.GetNormalMax(),
178 m_params.GetNormalQuantBits(), ifs, m_params.GetNormalPredMode(), bstream);
179 }
180 if (ret != O3DGC_OK)
181 {
182 return ret;
183 }
184 timer.Toc();
185 m_stats.m_timeNormal = timer.GetElapsedTime();
186 m_stats.m_streamSizeNormal = m_iterator - m_stats.m_streamSizeNormal;
187
188 // decode FloatAttributes
189 for(unsigned long a = 0; a < ifs.GetNumFloatAttributes(); ++a)
190 {
191 m_stats.m_streamSizeFloatAttribute[a] = m_iterator;
192 timer.Tic();
193 DecodeFloatArray(ifs.GetFloatAttribute(a), ifs.GetNFloatAttribute(a), ifs.GetFloatAttributeDim(a), ifs.GetFloatAttributeDim(a),
194 ifs.GetFloatAttributeMin(a), ifs.GetFloatAttributeMax(a),
195 m_params.GetFloatAttributeQuantBits(a), ifs, m_params.GetFloatAttributePredMode(a), bstream);
196 timer.Toc();
197 m_stats.m_timeFloatAttribute[a] = timer.GetElapsedTime();
198 m_stats.m_streamSizeFloatAttribute[a] = m_iterator - m_stats.m_streamSizeFloatAttribute[a];
199 }
200 if (ret != O3DGC_OK)
201 {
202 return ret;
203 }
204
205 // decode IntAttributes
206 for(unsigned long a = 0; a < ifs.GetNumIntAttributes(); ++a)
207 {
208 m_stats.m_streamSizeIntAttribute[a] = m_iterator;
209 timer.Tic();
210 DecodeIntArray(ifs.GetIntAttribute(a), ifs.GetNIntAttribute(a), ifs.GetIntAttributeDim(a), ifs.GetIntAttributeDim(a),
211 ifs, m_params.GetIntAttributePredMode(a), bstream);
212 timer.Toc();
213 m_stats.m_timeIntAttribute[a] = timer.GetElapsedTime();
214 m_stats.m_streamSizeIntAttribute[a] = m_iterator - m_stats.m_streamSizeIntAttribute[a];
215 }
216 if (ret != O3DGC_OK)
217 {
218 return ret;
219 }
220
221 timer.Tic();
222 m_triangleListDecoder.Reorder();
223 timer.Toc();
224 m_stats.m_timeReorder = timer.GetElapsedTime();
225
226#ifdef DEBUG_VERBOSE
227 fclose(g_fileDebugSC3DMCDec);
228#endif //DEBUG_VERBOSE
229 return ret;
230 }

239 {

241 long predResidual;
242 SC3DMCPredictor m_neighbors [O3DGC_SC3DMC_MAX_PREDICTION_NEIGHBORS];
243 Arithmetic_Codec acd;

248

251 const long nvert = (long) numIntArray;
252 unsigned char * buffer = 0;
253 unsigned long start = m_iterator;
254 unsigned long streamSize = bstream.ReadUInt32(m_iterator, m_streamType); // bitsream size
255 unsigned char mask = bstream.ReadUChar(m_iterator, m_streamType);
256 O3DGCSC3DMCBinarization binarization = (O3DGCSC3DMCBinarization)((mask >> 4) & 7);
257 predMode = (O3DGCSC3DMCPredictionMode)(mask & 7);
258 streamSize -= (m_iterator - start);
259 unsigned long iteratorPred = m_iterator + streamSize;
260 unsigned int exp_k = 0;
261 unsigned int M = 0;
262 if (m_streamType != O3DGC_STREAM_TYPE_ASCII)
263 {
264 if (binarization != O3DGC_SC3DMC_BINARIZATION_AC_EGC)
265 {
266 return O3DGC_ERROR_CORRUPTED_STREAM;
267 }
268 bstream.GetBuffer(m_iterator, buffer);
269 m_iterator += streamSize;
270 acd.set_buffer(streamSize, buffer);
271 acd.start_decoder();
272 exp_k = acd.ExpGolombDecode(0, bModel0, bModel1);
273 M = acd.ExpGolombDecode(0, bModel0, bModel1);
274 }
275 else
276 {
277 if (binarization != O3DGC_SC3DMC_BINARIZATION_ASCII)
278 {
279 return O3DGC_ERROR_CORRUPTED_STREAM;
280 }
281 bstream.ReadUInt32(iteratorPred, m_streamType); // predictors bitsream size
282 }
283 Adaptive_Data_Model mModelValues(M+2);
284
285#ifdef DEBUG_VERBOSE
286 printf("IntArray (%i, %i)\n", numIntArray, dimIntArray);
287 fprintf(g_fileDebugSC3DMCDec, "IntArray (%i, %i)\n", numIntArray, dimIntArray);
288#endif //DEBUG_VERBOSE
289
290 for (long v=0; v < nvert; ++v)
291 {
292 nPred = 0;
293 if ( v2T.GetNumNeighbors(v) > 0 &&
294 predMode != O3DGC_SC3DMC_NO_PREDICTION)
295 {
296 int u0 = v2T.Begin(v);
297 int u1 = v2T.End(v);
298 for (long u = u0; u < u1; u++)
299 {
300 long ta = v2T.GetNeighbor(u);
301 if (ta < 0)
302 {
303 break;
304 }
305 for(long k = 0; k < 3; ++k)
306 {
307 long w = triangles[ta*3 + k];
308 if ( w < v )
309 {
310 SC3DMCTriplet id = {-1, -1, w};
311 unsigned long p = Insert(id, nPred, m_neighbors);
312 if (p != 0xFFFFFFFF)
313 {
314 for (unsigned long i = 0; i < dimIntArray; i++)
315 {
316 m_neighbors[p].m_pred[i] = intArray[w*stride+i];
317 }
318 }
319 }
320 }
321 }
322 }
323 if (nPred > 1)
324 {
325#ifdef DEBUG_VERBOSE1
326 printf("\t\t vm %i\n", v);
327 fprintf(g_fileDebugSC3DMCDec, "\t\t vm %i\n", v);
328 for (unsigned long p = 0; p < nPred; ++p)
329 {
330 printf("\t\t pred a = %i b = %i c = %i \n", m_neighbors[p].m_id.m_a, m_neighbors[p].m_id.m_b, m_neighbors[p].m_id.m_c);
331 fprintf(g_fileDebugSC3DMCDec, "\t\t pred a = %i b = %i c = %i \n", m_neighbors[p].m_id.m_a, m_neighbors[p].m_id.m_b, m_neighbors[p].m_id.m_c);
332 for (unsigned long i = 0; i < dimIntArray; ++i)
333 {
334 printf("\t\t\t %i\n", m_neighbors[p].m_pred[i]);
335 fprintf(g_fileDebugSC3DMCDec, "\t\t\t %i\n", m_neighbors[p].m_pred[i]);
336 }
337 }
338#endif //DEBUG_VERBOSE
339 unsigned long bestPred;
340 if (m_streamType == O3DGC_STREAM_TYPE_ASCII)
341 {
342 bestPred = bstream.ReadUCharASCII(iteratorPred);
343 }
344 else
345 {
346 bestPred = acd.decode(mModelPreds);
347 }
348#ifdef DEBUG_VERBOSE1
349 printf("best (%i, %i, %i) \t pos %i\n", m_neighbors[bestPred].m_id.m_a, m_neighbors[bestPred].m_id.m_b, m_neighbors[bestPred].m_id.m_c, bestPred);
350 fprintf(g_fileDebugSC3DMCDec, "best (%i, %i, %i) \t pos %i\n", m_neighbors[bestPred].m_id.m_a, m_neighbors[bestPred].m_id.m_b, m_neighbors[bestPred].m_id.m_c, bestPred);
351#endif //DEBUG_VERBOSE
352 for (unsigned long i = 0; i < dimIntArray; i++)
353 {
354 if (m_streamType == O3DGC_STREAM_TYPE_ASCII)
355 {
356 predResidual = bstream.ReadIntASCII(m_iterator);
357 }
358 else
359 {
360 predResidual = DecodeIntACEGC(acd, mModelValues, bModel0, bModel1, exp_k, M);
361 }
362 intArray[v*stride+i] = predResidual + m_neighbors[bestPred].m_pred[i];
363#ifdef DEBUG_VERBOSE
364 printf("%i \t %i \t [%i]\n", v*dimIntArray+i, predResidual, m_neighbors[bestPred].m_pred[i]);
365 fprintf(g_fileDebugSC3DMCDec, "%i \t %i \t [%i]\n", v*dimIntArray+i, predResidual, m_neighbors[bestPred].m_pred[i]);
366#endif //DEBUG_VERBOSE
367 }
368 }
369 else if (v > 0 && predMode != O3DGC_SC3DMC_NO_PREDICTION)
370 {
371 for (unsigned long i = 0; i < dimIntArray; i++)
372 {
373 if (m_streamType == O3DGC_STREAM_TYPE_ASCII)
374 {
375 predResidual = bstream.ReadIntASCII(m_iterator);
376 }
377 else
378 {
379 predResidual = DecodeIntACEGC(acd, mModelValues, bModel0, bModel1, exp_k, M);
380 }
381 intArray[v*stride+i] = predResidual + intArray[(v-1)*stride+i];
382#ifdef DEBUG_VERBOSE
383 printf("%i \t %i\n", v*dimIntArray+i, predResidual);
384 fprintf(g_fileDebugSC3DMCDec, "%i \t %i\n", v*dimIntArray+i, predResidual);
385#endif //DEBUG_VERBOSE
386 }
387 }
388 else
389 {
390 for (unsigned long i = 0; i < dimIntArray; i++)
391 {
392 if (m_streamType == O3DGC_STREAM_TYPE_ASCII)
393 {
394 predResidual = bstream.ReadUIntASCII(m_iterator);
395 }
396 else
397 {
398 predResidual = DecodeUIntACEGC(acd, mModelValues, bModel0, bModel1, exp_k, M);
399 }
400 intArray[v*stride+i] = predResidual;
401#ifdef DEBUG_VERBOSE
402 printf("%i \t %i\n", v*dimIntArray+i, predResidual);
403 fprintf(g_fileDebugSC3DMCDec, "%i \t %i\n", v*dimIntArray+i, predResidual);
404#endif //DEBUG_VERBOSE
405 }
406 }
407 }
408 m_iterator = iteratorPred;
409#ifdef DEBUG_VERBOSE
410 fflush(g_fileDebugSC3DMCDec);
411#endif //DEBUG_VERBOSE
412 return O3DGC_OK;

414 template <class T>
415 O3DGCErrorCode SC3DMCDecoder<T>::ProcessNormals(const IndexedFaceSet<T> & ifs)
416 {
417 const long nvert = (long) ifs.GetNNormal();
418 const unsigned long normalSize = ifs.GetNNormal() * 2;
419 if (m_normalsSize < normalSize)
420 {
421 delete [] m_normals;
422 m_normalsSize = normalSize;
423 m_normals = new Real [normalSize];
424 }
425 const AdjacencyInfo & v2T = m_triangleListDecoder.GetVertexToTriangle();
426 const T * const triangles = ifs.GetCoordIndex();
427 Vec3<long> p1, p2, p3, n0, nt;
428 long na0 = 0, nb0 = 0;
429 Real rna0, rnb0, norm0;
430 char ni0 = 0, ni1 = 0;
431 long a, b, c;
432 for (long v=0; v < nvert; ++v)
433 {
434 n0.X() = 0;
435 n0.Y() = 0;
436 n0.Z() = 0;
437 int u0 = v2T.Begin(v);
438 int u1 = v2T.End(v);
439 for (long u = u0; u < u1; u++)
440 {
441 long ta = v2T.GetNeighbor(u);
442 if (ta == -1)
443 {
444 break;
445 }
446 a = triangles[ta*3 + 0];
447 b = triangles[ta*3 + 1];
448 c = triangles[ta*3 + 2];
449 p1.X() = m_quantFloatArray[3*a];
450 p1.Y() = m_quantFloatArray[3*a+1];
451 p1.Z() = m_quantFloatArray[3*a+2];
452 p2.X() = m_quantFloatArray[3*b];
453 p2.Y() = m_quantFloatArray[3*b+1];
454 p2.Z() = m_quantFloatArray[3*b+2];
455 p3.X() = m_quantFloatArray[3*c];
456 p3.Y() = m_quantFloatArray[3*c+1];
457 p3.Z() = m_quantFloatArray[3*c+2];
458 nt = (p2-p1)^(p3-p1);
459 n0 += nt;
460 }
461 norm0 = (Real) n0.GetNorm();
462 if (norm0 == 0.0)
463 {
464 norm0 = 1.0;
465 }
466 SphereToCube(n0.X(), n0.Y(), n0.Z(), na0, nb0, ni0);
467
468
469 rna0 = na0 / norm0;
470 rnb0 = nb0 / norm0;
471 ni1 = ni0 + m_orientation[v];
472 m_orientation[v] = ni1;
473 if ( (ni1 >> 1) != (ni0 >> 1) )
474 {
475 rna0 = Real(0.0);
476 rnb0 = Real(0.0);
477 }
478 m_normals[2*v] = rna0;
479 m_normals[2*v+1] = rnb0;
480
481#ifdef DEBUG_VERBOSE1
482 printf("n0 \t %i \t %i \t %i \t %i (%f, %f)\n", v, n0.X(), n0.Y(), n0.Z(), rna0, rnb0);
483 fprintf(g_fileDebugSC3DMCDec, "n0 \t %i \t %i \t %i \t %i (%f, %f)\n", v, n0.X(), n0.Y(), n0.Z(), rna0, rnb0);
484#endif //DEBUG_VERBOSE
485
486 }
487 return O3DGC_OK;

512 const long nvert = (long) numFloatArray;

514 unsigned char * buffer = 0;

544 }
545 Adaptive_Data_Model mModelValues(M+2);
546
547
548 if (predMode == O3DGC_SC3DMC_SURF_NORMALS_PREDICTION)
549 {
550 m_orientation.Allocate(size);
551 m_orientation.Clear();
552 if (m_streamType == O3DGC_STREAM_TYPE_ASCII)

560 {
561 Adaptive_Data_Model dModel(12);
