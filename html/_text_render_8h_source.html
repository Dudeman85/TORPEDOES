39 out vec2 TexCoords;
40 uniform mat4 projection;
41 uniform mat4 view;

50 #version 330 core

55

64 void Update(Camera* cam)
65 {
66 for (auto const& entity : entities)
67 {
68 Transform& transform = ecs::GetComponent<Transform>(entity);
69 TextRenderer& textRenderer = ecs::GetComponent<TextRenderer>(entity);
70 if (!textRenderer.font)
71 {
72 printf("ERROR: No font given!");
73 continue;
74 }
75 m_shader->use();
76
77 //Create the model matrix
78 glm::mat4 model = glm::mat4(1.0f);
79 //Position
80 model = glm::translate(model, transform.position.ToGlm());
81 //X, Y, Z euler rotations
82 model = glm::rotate(model, glm::radians(transform.rotation.x), glm::vec3(1.0f, 0.0f, 0.0f));
83 model = glm::rotate(model, glm::radians(transform.rotation.y), glm::vec3(0.0f, 1.0f, 0.0f));
84 model = glm::rotate(model, glm::radians(transform.rotation.z), glm::vec3(0.0f, 0.0f, 1.0f));
85 //Scale
86 model = glm::scale(model, transform.scale.ToGlm());
87
88 //Give the shader the model matrix
89 unsigned int modelLoc = glGetUniformLocation(m_shader->ID, "model");
90 glUniformMatrix4fv(modelLoc, 1, GL_FALSE, glm::value_ptr(model));
91
92 //Get the view and projection locations
93 unsigned int viewLoc = glGetUniformLocation(m_shader->ID, "view");
94 unsigned int projLoc = glGetUniformLocation(m_shader->ID, "projection");
95
96 if (!textRenderer.uiElement)
97 {
98 //Give the shader the camera's view matrix
99 glUniformMatrix4fv(viewLoc, 1, GL_FALSE, glm::value_ptr(cam->GetViewMatrix()));
100
101 //Give the shader the camera's projection matrix
102 glUniformMatrix4fv(projLoc, 1, GL_FALSE, glm::value_ptr(cam->GetProjectionMatrix()));
103 }
104 else
105 {
106 //Give the shader a constant view matrix
107 glUniformMatrix4fv(viewLoc, 1, GL_FALSE, glm::value_ptr(glm::mat4(1.0f)));
108
109 //Give the shader a constant projection matrix
110 glUniformMatrix4fv(projLoc, 1, GL_FALSE, glm::value_ptr(glm::mat4(1.0f)));
111 }
112
113 glUniform3f(glGetUniformLocation(m_shader->ID, "textColor"), textRenderer.color.x, textRenderer.color.y, textRenderer.color.z);
114 glActiveTexture(GL_TEXTURE0);
115 glBindVertexArray(textRenderer.font->VAO);
116
117 // Renders text one letter at a time
118 string::const_iterator c;
119 float x = textRenderer.offset.x;
120 float y = textRenderer.offset.y;
121 for (c = textRenderer.text.begin(); c != textRenderer.text.end(); ++c)
122 {
123 Character ch = textRenderer.font->characters[*c];
124
125 float xpos = x + ch.Bearing.x * textRenderer.scale.x;
126 float ypos = y - (ch.Size.y - ch.Bearing.y) * textRenderer.scale.y;
127
128 float w = ch.Size.x * textRenderer.scale.x;
129 float h = ch.Size.y * textRenderer.scale.y;
130
131 float vertices[6][4] = {
132 {xpos, ypos + h, 0.0f, 0.0f},
133 {xpos, ypos, 0.0f, 1.0f},
134 {xpos + w, ypos, 1.0f, 1.0f},
135
136 {xpos, ypos + h, 0.0f, 0.0f},
137 {xpos + w, ypos, 1.0f, 1.0f},
138 {xpos + w, ypos + h, 1.0f, 0.0f}
139 };
140
141 glBindTexture(GL_TEXTURE_2D, ch.TextureID);
142 glBindBuffer(GL_ARRAY_BUFFER, textRenderer.font->VBO);
143 glBufferSubData(GL_ARRAY_BUFFER, 0, sizeof(vertices), vertices);
144 glBindBuffer(GL_ARRAY_BUFFER, 0);
145 glDrawArrays(GL_TRIANGLES, 0, 6);
146 x += (ch.Advance >> 6) * textRenderer.scale.x;
147 }
148
149 glBindVertexArray(0);
150 glBindTexture(GL_TEXTURE_2D, 0);
151 }
152 }
