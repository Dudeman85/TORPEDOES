56 void update()
57 {
58 InputState newState = InputState::None;
59 pressed = false;
60 released = false;
61 newChange = false;
62
63 for (auto outputState : outputStates)
64 {
65 switch (*outputState)
66 {
67 case input::InputState::Released:
68 pressed = true;
69 break;
70 case input::InputState::Pressed:
71 released = true;
72 break;
73 case input::InputState::NewRelease:
74 released = true;
75 newChange = true;
76 case input::InputState::NewPress:
77 pressed = true;
78 newChange = true;
79 break;
80 case input::InputState::NewReleaseNewPress:
81 case input::InputState::NewPressNewRelease:
82 // Key was newly released and newly pressed last update
83 pressed = true;
84 released = true;
85 newChange = true;
86 break;
87 default:
88 break;
89 }
90 if (newChange && pressed && released)
91 {
92 return; // All states are true, testing cannot wield any changes
93 }
94 }
95 }

97 protected:
98 bool pressed = false; // Whether key is pressed the update before this poll
99 bool released = false; // Whether key is released the update before this poll
100 bool newChange = false; // Whether key was changed the update before this poll

119 // Polls the change in the key, returns whether to poll next update
120 void update()
121 {
122 if (buttonInputState == InputState::None)
123 {
124 switch (buttonOutputState)
125 {
126 case input::InputState::Released:
127 case input::InputState::Pressed:
128 // Key was not updated, keep the same state
129 break;
130 case input::InputState::NewPress:
131 case input::InputState::NewReleaseNewPress:
132 // Key was newly pressed last update, now it's just pressed
133 buttonOutputState = InputState::Pressed;
134 break;
135 case input::InputState::NewRelease:
136 case input::InputState::NewPressNewRelease:
137 // Key was newly released last update, now it's just released
138 buttonOutputState = InputState::Released;
139 break;
140 default:
141 break;
142 }
143 }
144 else
145 {
146 // Set outputState to be InputState
147 buttonOutputState = buttonInputState;
148 // Reset inputState
149 buttonInputState = InputState::None;
150 }
151 }
152
153 // Pressed/released logic
154 void callback(bool pressed)
155 {
156 if (pressed)
157 {
158 // Button is currently being pressed, what about other inputs this update cycle?
159 switch (buttonInputState)
160 {
161 case input::InputState::None: // No input before: ignore
162 case input::InputState::Released: // Should not be possible: ignore
163 case input::InputState::Pressed: // Should not be possible: ignore
164 case input::InputState::NewPress: // Same input: ignore
165 // There were no conflicting inputs: Set as new press:
166 buttonInputState = InputState::NewPress;
167 break;
168 case input::InputState::NewRelease:
169 case input::InputState::NewReleaseNewPress:
170 case input::InputState::NewPressNewRelease:
171 // Both inputs: Set as new new release, then new press:
172 buttonInputState = InputState::NewReleaseNewPress;
173 break;
174 default:
175 break;
176 }
177 }
178 else
179 {
180 // Button is currently not being pressed, what about other inputs this update cycle?
181 switch (buttonInputState)
182 {
183 case input::InputState::None: // No input before: ignore
184 case input::InputState::Released: // Should not be possible: ignore
185 case input::InputState::Pressed: // Should not be possible: ignore
186 case input::InputState::NewRelease: // Same input: ignore
187 // There were no conflicting inputs: Set as new release:
188 buttonInputState = InputState::NewRelease;
189 break;
190 case input::InputState::NewPress:
191 case input::InputState::NewReleaseNewPress:
192 case input::InputState::NewPressNewRelease:
193 // Both inputs: Set as new new press, then new release:
194 buttonInputState = InputState::NewPressNewRelease;
195 break;

197 break;
