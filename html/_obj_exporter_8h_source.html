151 template <class T, class Compare = std::less<T>>
152 class indexMap {
153 int mNextIndex;
154 typedef std::map<T, int, Compare> dataType;
155 dataType vecMap;
156
157 public:
158 indexMap()
159 : mNextIndex(1) {
160 // empty
161 }
162
163 int getIndex(const T& key) {
164 typename dataType::iterator vertIt = vecMap.find(key);
165 // vertex already exists, so reference it
166 if(vertIt != vecMap.end()){
167 return vertIt->second;
168 }
169 return vecMap[key] = mNextIndex++;
170 };
171
172 void getKeys( std::vector<T>& keys ) {
173 keys.resize(vecMap.size());
174 for(typename dataType::iterator it = vecMap.begin(); it != vecMap.end(); ++it){
175 keys[it->second-1] = it->first;
176 }
177 };
178 };
